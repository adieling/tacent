1// tChunk.h

3// The generic chunk based file format for all Tacent projects. This interface can be used for creating and reading 
4// arbitrary data files. At the bottom of this file are all the global chunk ids. There are 268 million available, so 
5// just add chunk IDs as necessary. The format supports alignment of data from 4 bytes to 512 bytes in powers of 2. 
6// 
7// A chunk consists of: 
8// 1) A 4 byte specifier. From most-significant bit to least: 
9// 1 bit: Is-container boolean. 
10// 3 bits: Alignment shift. 
11// 28 bits: Chunk ID. 
12// 2) A 4 byte unsigned int (big-endian? or plat-dependent?). This is the chunk data size not including any 
13// leading alignment padding. 
14// 3) Zero or more alignment padding bytes. 
15// 4) The data. 
16// 5) Any necessary ending alignment pad to achieve 4-byte alignment. 
17// 
18// Multiple chunks within a file are consecutive and chunks may contain sub chunks. Regardless of the data alignment 
19// size, all chunks start at 4-byte aligned addresses. Data alignment requirements may be stricter. 
20// 
21// Technically chunks may contain both data and sub-chunks but if you do it this way your chunk reading code will need 
22// to be smart enough to know where the data and sub-chunks are. By convention we do _not_ do this. A chunk with 
23// sub-chunks should not contain any other straight data, only other sub-chunks. 
24// 
25// A chunk ID of 0x00000000 is considered invalid so don't use it. It is also the ID returned by the uber-chunk 
26// (tChunkReader object). 
27// 
28// To maintain good file compatibility, chunks whose ID are unknown should be ignored and not treated as an error. 
29// 
30// If the contents of a chunk need to change you'd better use a new chunk ID. If you really like the name of the chunk 
31// you can keep it though. Simply rename it (e.g. from tChunkID_NICENAME to tChunkID_OBSOLETE_NICENAME) while keeping 
32// the ID the same. You can now use the chunk name (tChunkID_NICENAME) elsewhere with a new ID. The obsolete chunk, of 
33// course, must contain the same stuff as the original nice name chunk.

364// Globally chunk IDs. 
365// 
366// The chunk ID may be viewed as bits in the following order: 
367// 
368// b31 b30 b29 b28 b27 b26 b25 b24 b23 ... b0 
369// --- ----------- --------------- ---------- 
370// con align major id minor id 
371// 
372// The first bit (b31) is the container indicator (1 = container, 0 = data). The next three bits (b30, b29, b28) are 
373// reserved and are used for storing alignment information. Up to 512 byte alignment is supported. The 3 bits [0, 7] 
374// represent the shift amount - 2. That is, alignment = (3-bit-value + 2)^2. 
375// 
376// 0 -> 1<<(0+2) = 4 byte alignment. 
377// 1 -> 1<<(1+2) = 8 byte alignment. 
378// 2 -> 1<<(2+2) = 16 byte alignemnt. 
379// ... 
380// 7 -> 1<<(7+2) = 512 byte alignment. 
381// 
382// The reason we add the 2 is because the chunk ID and size must be 4-byte aligned so we are guaranteed 4B alignment 
383// from the get go. Specifying 1 or 2 byte alignment is therefore pointless. This allows us to increase our max 
384// alignment from 128 Bytes to 512 Bytes. 
385// 
386// The next nybble, bits 27 to 24, is the major id and must be 0 for all core (basic) chunks. Each major system (scene, 
387// physics, sound, rendering, ai, camera, gameplay, ui, etc) gets its own major chunk id. The major ID nybble plus the 
388// remaining 24 bits (minor ID) form the chunk ID. The currently assigned major IDs are: 
389// 
390// 0 : Core Foundation chunk IDs. 
391// 1 : Scene. Used by exporter and pipeline tools for things like meshes, models, materials, lights, etc. 
392// 2 : Rendering. These are used by the runtime rendering engine and the tools that generate data for it. 
393// 3 : Gameplay. Things like trigger volumes, regions, POIs, COIs, way points, etc. 
394// 4 : Physics. Rigid bodies, collision meshes, etc. 
395// 5 : AI. Paths etc. 
396// 6 : UI. Frontend and interface chunks like screens, buttons, boxes, and other widgets. 
397// 7 : Sound. Emitters, samples, etc. 
398// 8 : Camera. Camera paths, triggers, etc used by the runtime. 
399// 9 : Unassigned. 
400// A-E: For product (app or game) specific chunks. It's fine if more than one product uses the same ID here. 
401// A : Tactile. 
402// F: Generic. Don't know where it belongs. Use this major ID. 
403// 
404// In terms of chunk IDs the first nybble should be set to 0x8 for container chunks (chunks that only contain more 
405// chunks for their data). For chunks that contain data only, the first nybble should be 0x0. Only data chunks may have 
406// alignment. Everything else is 4-byte aligned. 
407// 
408// Note the use of 'old' style (not enum class) enums for chunk IDs. Since IDs more closely represent const integers, 
409// the type-safety of 'enum class' makes things more awkward. Casting to int every time would be onerous. 
410// 
411namespace tChunkID

699inline tChunk::tChunk(uint8* chunk) : 
700 Chunk(chunk), 
701 LastChunk(0) 
702{ 
703 uint32 idaa = *((uint32*)Chunk); 
704 int shift = ((idaa & 0x70000000) >> 28) + 2; 
705 int align = 1 << shift; 
706 
707 // We are safe to use uint32 cast here even for 64 bit pointers because the alignment requirements are much 
708 // smaller. That is, align is much less than 2^32. 
709 int bytesAfterAlign = uint32(uint64(chunk+8)) % align; 
710 int pad = bytesAfterAlign ? (align - bytesAfterAlign) : 0; 
711 ItemData = (chunk+8) + pad; 
712}

758inline int tChunk::GetDataSizeRaw() const 
759{ 
760 if (!IsValid()) 
761 return 0; 
762 
763 uint8* data = Chunk+8; 
764 uint32 idaa = *((uint32*)Chunk); 
765 int shift = ((idaa & 0x70000000) >> 28) + 2; 
766 int align = 1 << shift; 
767 
768 int bytesAfterAlign = uint32(uint64(data)) % align; 
769 int prologPad = bytesAfterAlign ? (align - bytesAfterAlign) : 0; 
770 int dataSize = GetDataSize(); 
771 data += prologPad + dataSize; 
772 
773 int epilogPad = (uint32(uint64(data)) % 4) ? (4 - (uint32(uint64(data)) % 4)) : 0; 
774 return prologPad + dataSize + epilogPad;
