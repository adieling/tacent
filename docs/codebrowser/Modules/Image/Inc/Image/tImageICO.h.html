6// a) Loads all frames of an ico, not just the biggest one. 
7// b) Supports embedded png images.

113inline tFrame* tImage::tImageICO::StealFrame(int frameNum) 
114{ 
115 tFrame* p = GetFrame(frameNum); 
116 if (!p) 
117 return nullptr; 
118 
119 return Frames.Remove(item: p); 
120}

123inline void tImage::tImageICO::StealFrames(tList<tFrame>& frames) 
124{ 
125 while (tFrame* frame = Frames.Remove()) 
126 frames.Append(item: frame); 
127}

130inline tFrame* tImage::tImageICO::GetFrame(int frameNum) 
131{ 
132 if ((frameNum >= Frames.GetNumItems()) || (frameNum < 0)) 
133 return nullptr; 
134 
135 tFrame* f = Frames.First(); 
136 while (frameNum--) 
137 f = f->Next(); 
138 
139 return f; 
140}

143inline void tImageICO::Clear() 
144{ 
145 while (tFrame* frame = Frames.Remove())

147 
148 tBaseImage::Clear();

152inline tPixelFormat tImageICO::GetBestSrcPixelFormat() const 
153{ 
154 tPixelFormat bestFormat = tPixelFormat::Invalid; 
155 for (tFrame* frame = Frames.First(); frame; frame = frame->Next()) 
156 { 
157 if (frame->PixelFormatSrc == tPixelFormat::Invalid) 
158 continue; 
159 
160 // Early exit as can't do better than 32-bit for an ico file. 
161 if (frame->PixelFormatSrc == tPixelFormat::R8G8B8A8) 
162 return tPixelFormat::R8G8B8A8; 
163 
164 // Otherwise use the bpp metric to determine the 'best'. 
165 else if (tGetBitsPerPixelFloat(frame->PixelFormatSrc) > tGetBitsPerPixelFloat(bestFormat)) 
166 bestFormat = frame->PixelFormatSrc; 
167 } 
168  
169 return bestFormat; 
170}
