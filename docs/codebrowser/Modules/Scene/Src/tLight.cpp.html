20void tLight::Clear() 
21{ 
22 tObject::Clear(); 
23 Type = tType::Point; 
24 AttenNearStart = 0.0f; 
25 AttenNearEnd = 0.0f; 
26 AttenFarStart = 0.0f; 
27 AttenFarEnd = 0.0f;

31void tLight::Save(tChunkWriter& chunk) const 
32{ 
33 chunk.Begin(chunkID: tChunkID::Scene_Light); 
34 { 
35 tObject::Save(chunk); 
36 
37 chunk.Begin(chunkID: tChunkID::Scene_LightType); 
38 { 
39 chunk.Write(obj: Type); 
40 } 
41 chunk.End(); 
42 
43 chunk.Begin(chunkID: tChunkID::Scene_LightParameters); 
44 { 
45 chunk.Write(obj: AttenNearStart); 
46 chunk.Write(obj: AttenNearEnd); 
47 chunk.Write(obj: AttenFarStart); 
48 chunk.Write(obj: AttenFarEnd); 
49 } 
50 chunk.End(); 
51 } 
52 chunk.End(); 
53}

56void tLight::Load(const tChunk& lightChunk) 
57{ 
58 tAssert(lightChunk.ID() == tChunkID::Scene_Light); 
59 Clear(); 
60 
61 for (tChunk chunk = lightChunk.First(); chunk.Valid(); chunk = chunk.Next()) 
62 { 
63 switch (chunk.ID()) 
64 { 
65 case tChunkID::Scene_Object: 
66 tObject::Load(chunk); 
67 break; 
68 
69 case tChunkID::Scene_LightType: 
70 chunk.GetItem(item&: Type); 
71 break; 
72 
73 case tChunkID::Scene_LightParameters: 
74 chunk.GetItem(item&: AttenNearStart); 
75 chunk.GetItem(item&: AttenNearEnd); 
76 chunk.GetItem(item&: AttenFarStart); 
77 chunk.GetItem(item&: AttenFarEnd); 
78 break; 
79 } 
80 } 
81}
