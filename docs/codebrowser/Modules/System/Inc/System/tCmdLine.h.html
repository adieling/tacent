1// tCmdLine.h 
2// 
3// Parses a command line. A command line takes the form: 
4// program [arg1 arg2 arg3 ...] 
5// 
6// ARGUMENTS: 
7// Arguments are separated by spaces. An argument must be enclosed in quotes (single or double) if it has spaces in it 
8// or you want the argument to start with a hyphen literal. Hat (^) escape sequences can be used to put either type of 
9// quote inside. If you need to specify file paths you may use forward or back slashes. An ARGUMENT is either an OPTION 
10// or PARAMETER. 
11// 
12// OPTIONS: 
13// An option is simply an argument that starts with a hyphen (-). An option has a short syntax and a long syntax. 
14// Short syntax is a - followed by a single non-hyphen character. The long form is -- followed by a word. All options 
15// support either long, short, or both forms. Options may have 0 or more arguments separated by spaces. Options can be 
16// specified in any order. Short form options may be combined: Eg. -al expands to -a -l. 
17// 
18// FLAGS: 
19// If an option takes zero arguments it is called a flag. You can only test for a FLAGs presence or lack thereof. 
20// 
21// PARAMETERS: 
22// A parameter is simply an argument that is not one of the available options. It can be read as a string and parsed 
23// however is needed (converted to an integer, float etc.) Order is important when specifying parameters. If you need a 
24// hyphen in a parameter at the start you will need put the parameter in quotes. For example, a filename _can_ start 
25// with -. Note that arguments that start with a hyphen but are not recognized as a valid option just get turned into 
26// parameters. This means interpreting a hyphen directly instead of as an option specifier will happen automatically if 
27// there are no options matching what comes after the hyphen. Eg. 'tool -.85 --add 33 -87.98 --notpresent' work just 
28// fine as long as there are no options that have a short form with digits or a decimal. In this example the -.85 will 
29// be the first parameter, --notpresent will be the second. The --add is assumed to take in two number arguments. 
30// 
31// ESCAPES: 
32// In some cases you may need a particular character to appear inside an argument. For example you 
33// may need a single or double quote to apprear inside a parameter. The hat (^) followed by the character you need is 
34// used for this purpose. Eg: ^^ yields ^ | ^' yields ' | ^" yields " 
35// 
36// VARIABLE ARGUMENTS: 
37// A variable number of space-separated parameters may be specified if the tool supports them. The parsing system will 
38// collect them all up if the parameter number is unset (-1). 
39// A variable number of option arguments is not directly supported due to the more complex parsing that would be needed. 
40// The same result is achieved by entering the same option more than once. 
41// Eg. tool -I /patha/include/ -I /pathb/include 
42// 
43// EXAMPLE: 
44// mycopy -R --overwrite fileA.txt -pat fileB.txt --log log.txt 
45// 
46// The fileA.txt and fileB.txt in the above example are parameters (assuming the overwrite option is a flag). fileA.txt 
47// is the first parameter and fileB.txt is the second. 
48// 
49// The '--log log.txt' is an option with a single argument, log.txt. Options that are Flags may be combined. The -pat 
50// in the example expands to -p -a -t. It is suggested only to combine flag options as only the last option would get 
51// any arguments. 
52// 
53// DESIGN: 
54// A powerful feature of the design of this parsing system is separation of concerns. In a typical system the knowledge 
55// of all the different command line parameters and options is needed in a single place, often in main() where argc and 
56// argv are passed in. These values need to somehow be passed all over the place in a large system. With tCmdLine you 
57// specify which options and parameters you care about only in the cpp file you are working in. 
58// 
59// To use the command line class, you start by registering your options and parameters. This is done using the tOption 
60// and tParam types to create static objects. After main calls the parse function, your objects get populated 
61// appropriately. For example, 
62// 
63// FileA.cpp: 
64// tParam FromFile(1, "FromFile"); // The 1 means this is the first parameter. The description is optional. 
65// tParam ToFile(2, "ToFile"); // The 2 means this is the second parameter. The description is optional. 
66// tOption("log", 'l', 1, "Specify log file"); // The 1 means there is one option argument to --log or -l. 
67// 
68// FileB.cpp: 
69// tOption ProgramOption('p', 0, "Program mode."); 
70// tOption AllOption('a', "ALL", 0, "Include all widgets."); 
71// tOption TimeOption("time", 't', 0, "Print timestamp."); 
72// 
73// Main.cpp: 
74// tParse(argc, argv); 
75// 
76// Internal Processing. The first step is the expansion of combined single hyphen options. Next the parameters and 
77// options are parsed out. For each registered tOption and tParam object, its members are set to reflect the current 
78// command line when the tParse call is made. You may have more than one tOption that responds to the same option name. 
79// You may have more than one tParam that responds to the same parameter number.

100 // Parameter number starts at 1. Set it to which parameter you want from the command line. For example, set to 
101 // 2 if you want this object to receive the 2nd parameter. If you want ALL command-line paramters collected

103 // and desc are optional and are used when printing the tool usage. Exclude means exclude from the usage print.

107 tString Get() const { return Values.IsEmpty() ? tString() : *Values.First(); } 
108 void Set(const tString& value) { if (Values.First()) Values.First()->Set(value); else Values.Append(item: new tStringItem(value)); } 
109 bool IsPresent() const { return !Values.IsEmpty(); } 
110 operator bool() const { return IsPresent(); } 
111 
112 // 1 based. 0 means all. 
113 int ParamNumber; 
114 
115 // This usually has a single item (if ParamNumber >= 1). Only if ParamNumber == 0 does this get populated with 
116 // every parameter in the command line. There may be an arbitrary number of them in this case. This list is not 
117 // in static-zero initialization mode because it is never populated before main (there is no init order issue). 
118 tList<tStringItem> Values; 
119 tString Name; 
120 tString Description; 
121 bool ExcludeFromUsage; 
122 };

217inline bool tCmdLine::tOption::GetArgs(tList<tStringItem>& args) const 
218{ 
219 if (!IsPresent()) 
220 return false; 
221 
222 for (tStringItem* srcArg = Args.First(); srcArg; srcArg = srcArg->Next()) 
223 args.Append(item: new tStringItem(*srcArg)); 
224 
225 return true; 
226}
