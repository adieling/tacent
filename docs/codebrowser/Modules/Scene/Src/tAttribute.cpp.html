21void tAttribute::Save(tChunkWriter& chunk) const 
22{ 
23 if (!IsValid()) 
24 return; 
25 
26 chunk.Begin(chunkID: tChunkID::Scene_Attribute); 
27 { 
28 chunk.Begin(chunkID: tChunkID::Scene_AttributeName); 
29 { 
30 chunk.Write(s: Name); 
31 } 
32 chunk.End(); 
33 
34 chunk.Begin(chunkID: tChunkID::Scene_AttributeType); 
35 { 
36 chunk.Write(obj: Type); 
37 } 
38 chunk.End(); 
39 
40 // Values must be switched on type so that endianness corrections are made. 
41 chunk.Begin(chunkID: tChunkID::Scene_AttributeValue); 
42 { 
43 switch (Type) 
44 { 
45 case tType::Bool: 
46 chunk.Write(b: GetAsBool()); 
47 break; 
48 
49 case tType::Int: 
50 chunk.Write(obj: GetAsInt()); 
51 break; 
52 
53 case tType::Colour: 
54 chunk.Write(c: GetAsColour()); 
55 break; 
56 
57 case tType::Float: 
58 chunk.Write(obj: GetAsFloat()); 
59 break; 
60 }; 
61 } 
62 chunk.End(); 
63 } 
64 chunk.End(); 
65}

68void tAttribute::Load(const tChunk& attribChunk) 
69{ 
70 tAssert(attribChunk.GetID() == tChunkID::Scene_Attribute); 
71 Clear(); 
72 
73 for (tChunk chunk = attribChunk.First(); chunk.IsValid(); chunk = chunk.Next()) 
74 { 
75 switch (chunk.GetID()) 
76 { 
77 case tChunkID::Scene_AttributeName: 
78 chunk.GetItem(item&: Name); 
79 break; 
80 
81 case tChunkID::Scene_AttributeType: 
82 chunk.GetItem(item&: Type); 
83 break; 
84 
85 case tChunkID::Scene_AttributeValue: 
86 tAssert(Type != tType::Invalid); 
87 
88 // We switch on the type here so that the tChunk system can handle any endian issues. 
89 switch (Type) 
90 { 
91 case tType::Bool: 
92 { 
93 bool val; 
94 chunk.GetItem(item&: val); 
95 Value = val ? 1 : 0; 
96 break; 
97 } 
98 
99 case tType::Int: 
100 { 
101 int val; 
102 chunk.GetItem(item&: val); 
103 Value = int(val); 
104 break; 
105 } 
106 
107 case tType::Colour: 
108 { 
109 tColour4b val; 
110 chunk.GetItem(item&: val); 
111 Value = *((uint32*)(&val)); 
112 break; 
113 } 
114 
115 case tType::Float: 
116 { 
117 float val; 
118 chunk.GetItem(item&: val); 
119 Value = *((uint32*)(&val)); 
120 break; 
121 } 
122 } 
123 break; 
124 } 
125 } 
126}

141void tAttributes::Save(tChunkWriter& chunk) const 
142{ 
143 chunk.Begin(chunkID: tChunkID::Scene_AttributeList); 
144 for (tItList<tAttribute>::Iter attr = Attributes.First(); attr; ++attr)

150void tAttributes::Load(const tChunk& chunk) 
151{ 
152 Clear(); 
153 tAssert(chunk.GetID() == tChunkID::Scene_AttributeList); 
154 for (tChunk attr = chunk.First(); attr.Valid(); attr = attr.Next()) 
155 Attributes.Append(obj: new tAttribute(attr));

159bool tAttributes::operator==(const tAttributes& attribs) const 
160{ 
161 if (Count() != attribs.Count()) 
162 return false; 
163 
164 for (tItList<tAttribute>::Iter attr = Attributes.First(); attr; ++attr) 
165 { 
166 if (!attribs.Contains(attrib: *attr)) 
167 return false; 
168 } 
169 
170 return true; 
171}
