51// 
52// RGBA pixel 
53// 
54 
55struct Rgba 
56{ 
57 half r; 
58 half g; 
59 half b; 
60 half a; 
61  
62 Rgba () {} 
63 Rgba (half r, half g, half b, half a = 1.f): r (r), g (g), b (b), a (a) {} 
64};

67// 
68// Channels in an RGBA file 
69// 
70 
71enum RgbaChannels 
72{ 
73 WRITE_R = 0x01, // Red 
74 WRITE_G = 0x02, // Green 
75 WRITE_B = 0x04, // Blue 
76 WRITE_A = 0x08, // Alpha 
77 
78 WRITE_Y = 0x10, // Luminance, for black-and-white images, 
79 // or in combination with chroma 
80 
81 WRITE_C = 0x20, // Chroma (two subsampled channels, RY and BY, 
82 // supported only for scanline-based files) 
83 
84 WRITE_RGB = 0x07, // Red, green, blue 
85 WRITE_RGBA = 0x0f, // Red, green, blue, alpha 
86 
87 WRITE_YC = 0x30, // Luminance, chroma 
88 WRITE_YA = 0x18, // Luminance, alpha 
89 WRITE_YCA = 0x38 // Luminance, chroma, alpha 
90};
