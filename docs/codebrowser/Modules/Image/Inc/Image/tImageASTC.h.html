3// This class knows how to load and save ARM's Adaptive Scalable Texture Compression (.astc) files. The pixel data is 
4// stored in a tLayer. If decode was requested the layer will store raw pixel data. The layer may be 'stolen'. IF it 
5// is the tImageASTC is invalid afterwards. This is purely for performance.

27class tImageASTC : public tBaseImage 
28{ 
29public: 
30 enum LoadFlag 
31 { 
32 LoadFlag_Decode = 1 << 0, // Decode the astc texture data into RGBA 32 bit. If not set, the pixel data will remain unmodified. 
33 
34 // The remaining flags only apply when decode flag set. ReverseRowOrder is guaranteed to work if decoding, and 
35 // guaranteed to not work if not decoding. 
36 LoadFlag_ReverseRowOrder = 1 << 1, // OpenGL uses the lower left as the orig DirectX uses the upper left. Set flag for OpenGL. 
37 LoadFlag_GammaCompression = 1 << 2, // Gamma-correct. Gamma compression using an encoding gamma of 1/2.2. You may override the 2.2 default. 
38 LoadFlag_SRGBCompression = 1 << 3, // Same as above but uses the official sRGB transformation. Linear -> sRGB. Approx encoding gamma of 1/2.4 for part of curve. 
39 LoadFlag_AutoGamma = 1 << 4, // Determines whether to apply sRGB compression based on colour profile. Call GetColourProfile to see if it applied. 
40 LoadFlag_ToneMapExposure = 1 << 5, // Apply exposure value when loading the astc. 
41 LoadFlags_Default = LoadFlag_Decode | LoadFlag_ReverseRowOrder | LoadFlag_AutoGamma 
42 }; 
43 
44 struct LoadParams 
45 { 
46 LoadParams() { Reset(); } 
47 LoadParams(const LoadParams& src) : Flags(src.Flags), Profile(src.Profile), Gamma(src.Gamma), Exposure(src.Exposure) { } 
48 
49 void Reset() { Flags = LoadFlags_Default; Profile = tColourProfile::LDRsRGB_LDRlA; Gamma = tMath::DefaultGamma; Exposure = 1.0f; } 
50 LoadParams& operator=(const LoadParams& src) { Flags = src.Flags; Profile = src.Profile; Gamma = src.Gamma; Exposure = src.Exposure; return *this; } 
51 
52 uint32 Flags; 
53 tColourProfile Profile; // Used if decoding and LoadFlag_AutoGamma is set. 
54 float Gamma; // Used if decoding and LoadFlag_GammaCompression is set. 
55 float Exposure; // Used if decoding and LoadFlag_ToneMapExposure is set. 
56 }; 
57 
58 // Creates an invalid tImageASTC. You must call Load manually. 
59 tImageASTC() { } 
60 tImageASTC(const tString& astcFile, const LoadParams& params = LoadParams()) { Load(astcFile, params); } 
61 
62 // The data is copied out of astcFileInMemory. Go ahead and delete[] after if you want. 
63 tImageASTC(const uint8* astcFileInMemory, int numBytes, const LoadParams& params = LoadParams()) { Load(astcFileInMemory, numBytes, params); } 
64 
65 // This one sets from a supplied pixel array. If steal is true it takes ownership of the pixels pointer. Otherwise 
66 // it just copies the data out. 
67 tImageASTC(tPixel4b* pixels, int width, int height, bool steal = false) { Set(pixels, width, height, steal); } 
68 
69 // Sets from a single frame. 
70 tImageASTC(tFrame* frame, bool steal = true) { Set(frame, steal); } 
71 
72 // Constructs from a tPicture. 
73 tImageASTC(tPicture& picture, bool steal = true) { Set(picture, steal); } 
74 
75 virtual ~tImageASTC() { Clear(); } 
76 
77 // Clears the current tImageASTC before loading. Returns success. If false returned, object is invalid. 
78 bool Load(const tString& astcFile, const LoadParams& params = LoadParams()); 
79 bool Load(const uint8* astcFileInMemory, int numBytes, const LoadParams& params = LoadParams()); 
80 
81 // This one sets from a supplied pixel array. If steal is true it takes ownership of the pixels pointer. Otherwise 
82 // it just copies the data out. 
83 bool Set(tPixel4b* pixels, int width, int height, bool steal = false) override; 
84 
85 // Sets from a single frame. After this is called the layer data will be in R8G8B8A8. 
86 bool Set(tFrame*, bool steal = true) override; 
87 
88 // Sets from a tPicture. 
89 bool Set(tPicture& picture, bool steal = true) override; 
90 
91 // After this call no memory will be consumed by the object and it will be invalid. 
92 void Clear() override; 
93 bool IsValid() const override { return (Layer && Layer->IsValid()); } 
94 
95 int GetWidth() const { return Layer ? Layer->Width : 0; } 
96 int GetHeight() const { return Layer ? Layer->Height : 0; } 
97 
98 // All pixels must be opaque (alpha = 255) for this to return true. Always returns false if the object is not in the 
99 // R8G8B8A8 pixel-format (i.e. not decoded) since all ASTC pixel formats support alpha. 
100 bool IsOpaque() const; 
101 
102 // After the steal call you are the owner of the layer and must eventually delete it. This tImageASTC object is 
103 // invalid afterwards. 
104 tLayer* StealLayer() { tLayer* layer = Layer; Layer = nullptr; return layer; } 
105 tLayer* GetLayer() const { return Layer; } 
106 tFrame* GetFrame(bool steal = true) override; 
107 
108private: 
109 // We store the data in a tLayer because that's the container we use for pixel data than may be in any format. 
110 // The user of tImageASTC is not required to decode, so we can't just use a tPixel array. 
111 tLayer* Layer = nullptr; 
112};
