18// The tThrow header should not depend on any header that may need to throw an exception. This is why we can't inline 
19// the tError member functions. They would depend on tPrint which includes tFile, but tFile throws and needs tThrow. 
20// The same header included twice in the include chain doesn't work. 
21#include <Foundation/tString.h> 
22 
23 
24// All errors that can be thrown by any library module are derived from this. By catching exceptions of the tError 
25// type all possibilities can be covered. Conversely, derived classes can be caught for more control over who 
26// handles what type of error. 
27struct tError 
28{ 
29 tError(const char* format, ...); 
30 tError(const tString& message); 
31 tError(); 
32 tString Message; 
33}; 
34 
35 
36struct tWarning 
37{ 
38 tWarning(const char* format, ...); 
39 tWarning(const tString& message); 
40 tWarning(); 
41 tString Message; 
42};
