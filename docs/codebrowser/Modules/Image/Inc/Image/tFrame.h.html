3// A tFrame is a container for an array of tPixels (in RGBA format) along with some minimal satellite information 
4// including width, height, and duration. The tFrame class is primarily used by tImage formats that support more 
5// than one frame in a single image file (like gif, tiff, apng, and webp). A tFrame differs from a tLayer in that 
6// they are much simpler and do not support multiple pixel formats.

71inline bool tFrame::Set(const tFrame& frame) 
72{ 
73 Clear(); 
74 
75 // If frame is not valid this one gets returned with defaults (also invalid). 
76 if ((&frame == this) || !frame.IsValid()) 
77 return false; 
78 
79 Width = frame.Width; 
80 Height = frame.Height; 
81 Duration = frame.Duration; 
82 PixelFormatSrc = frame.PixelFormatSrc; 
83 
84 tAssert((frame.Width > 0) && (frame.Height > 0) && frame.Pixels); 
85 Pixels = new tPixel4b[Width*Height]; 
86 tStd::tMemcpy(dest: Pixels, src: frame.Pixels, numBytes: Width*Height*sizeof(tPixel4b)); 
87 
88 return true; 
89}

92inline bool tFrame::Set(const tPixel4b* srcPixels, int width, int height, float duration) 
93{ 
94 Clear(); 
95 if (!srcPixels || (width <= 0) || (height <= 0)) 
96 return false; 
97 
98 Width = width; 
99 Height = height; 
100 Duration = duration; 
101 PixelFormatSrc = tPixelFormat::R8G8B8A8; 
102 
103 Pixels = new tPixel4b[Width*Height]; 
104 tStd::tMemcpy(dest: Pixels, src: srcPixels, numBytes: Width*Height*sizeof(tPixel4b)); 
105 return true; 
106}

109inline bool tFrame::StealFrom(tFrame& frame) 
110{ 
111 // If frame is not valid this one gets returned with defaults (also invalid). 
112 if ((&frame == this) || !frame.IsValid()) 
113 return false; 
114 
115 Width = frame.Width; 
116 Height = frame.Height; 
117 Duration = frame.Duration; 
118 PixelFormatSrc = frame.PixelFormatSrc; 
119 Pixels = frame.Pixels; 
120 frame.Pixels = nullptr; // Frame is left invalid. 
121 return true; 
122}

125inline bool tFrame::StealFrom(tPixel4b* src, int width, int height, float duration) 
126{ 
127 if (!src || (width <= 0) || (height <= 0)) 
128 return false; 
129 
130 Width = width; 
131 Height = height; 
132 Duration = duration; 
133 PixelFormatSrc = tPixelFormat::R8G8B8A8; 
134 Pixels = src; 
135 return true;

150inline void tFrame::ReverseRows() 
151{ 
152 int numPixels = Width * Height; 
153 tPixel4b* origPixels = Pixels; 
154 Pixels = new tPixel4b[numPixels]; 
155 
156 int bytesPerRow = Width*sizeof(tPixel4b); 
157 for (int y = Height-1; y >= 0; y--) 
158 tStd::tMemcpy(dest: (uint8*)Pixels + ((Height-1)-y)*bytesPerRow, src: (uint8*)origPixels + y*bytesPerRow, numBytes: bytesPerRow); 
159 
160 delete[] origPixels; 
161}

164inline bool tFrame::IsOpaque() const 
165{ 
166 for (int p = 0; p < (Width * Height); p++) 
167 if (Pixels[p].A < 255) 
168 return false; 
169 return true;
