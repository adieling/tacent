21const float Pi = 3.14159265358979323f; // 180 degrees.

30// This is a practical epsilon that can be used in many circumstances. One one millionth. 
31const float Epsilon = 0.000001f; 
32const double EpsilonDbl = 0.000001; 
33 
34// This is the smallest effective positive float such that 1.0f+EpsilonEff != 1.0f. 
35const float EpsilonEff = 1.192092896e-07f;

37// This is the smallest positive value that can be represented by a single precision floating point. 
38const float EpsilonRep = 1.175494351e-38f;

41// The infinities below are not the special IEEE bit patterns that represent infinity. They are the very largest and 
42// smallest values that may be represented without use of the special bit patterns. 
43const float PosInfinity = MaxFloat; 
44const float NegInfinity = -MaxFloat;

46// C++11 should promote 2147483648 to long long int, but VS currently turns it into and unsigned int. That's why we're 
47// using the hex string literal representation for now. 
48const int32 MinInt32 = 0x80000000; // Decimal value -2147483648. 
49const int32 MaxInt32 = 2147483647; // Hex value 0x7FFFFFFF. 
50const uint32 MinUint32 = 0; 
51const uint32 MaxUint32 = 0xFFFFFFFF;
