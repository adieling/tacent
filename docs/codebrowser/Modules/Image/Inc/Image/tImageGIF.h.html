68 // This function returns true on success. If any required condition is not met false is returned. gifFile is the 
69 // file to save to. It must end with a .gif extension. 
70 // 
71 // Format must be one of the PALNBIT formats where N E [1,8]. i.e. Palette sizes 2, 4, 8, 16, 32, 64, 128, or 256. 
72 // 
73 // Method should be set to one of the 4 available quantization methods: fixed, neuquant, wu, or scolorq. 
74 // 
75 // Loop only applies to multi-frame/animated gifs. 0 to loop forever. >0 to loop a specific number of times. 
76 // 
77 // Gif files support binary alpha only. 
78 // If AlphaThreshold is 255 (special case), the saved gif will be be opaque even if not all pixel alphas are max. 
79 // If AlphaThreshold is E [0, 255), any pixel with alpha <= AlphaThreshold is considered transparent. Pixel alpha 
80 // values > AlphaThreshold are considered opaque. If PAL1BIT is chosen as the pixel format (2 palette entries), 
81 // AlphaThreshold is forced to 255 (fully opaque). This is because gif transparency uses a palette entry, and colour 
82 // quantization on a single colour is not useful. 
83 // If AlphaThreshold is -1 (auto), then the frame pixels are inspected for transparency. If all frames are fully 
84 // opaque, an opaque gif will be saved (AlphaThreshold at 255). If any frame has a non-max pixel alpha, an 
85 // AlphaThreshold of 127 is used. 
86 // 
87 // OverrideframeDuration is in 1/100 seconds. Set to >= 0 to override all frames. Note that values of 0 or 1 get 
88 // min-clamped to 2 during save since many viewers do not handle values below 2 properly. If OverrideFrameDuration 
89 // is < 0, the individual frames' duration is used after being converted from seconds to 1/100th of seconds. 
90 struct SaveParams 
91 { 
92 SaveParams() { Reset(); } 
93 SaveParams(const SaveParams& src) : Format(src.Format), Method(src.Method), Loop(src.Loop), AlphaThreshold(src.AlphaThreshold), OverrideFrameDuration(src.OverrideFrameDuration), DitherLevel(src.DitherLevel), FilterSize(src.FilterSize), SampleFactor(src.SampleFactor) { } 
94 void Reset() { Format = tPixelFormat::PAL8BIT; Method = tQuantize::Method::Wu; Loop = 0; AlphaThreshold = -1; OverrideFrameDuration = -1; DitherLevel = 0.0; FilterSize = 3; SampleFactor = 1; } 
95 SaveParams& operator=(const SaveParams& src) { Format = src.Format; Method = src.Method; Loop = src.Loop; AlphaThreshold = src.AlphaThreshold; OverrideFrameDuration = src.OverrideFrameDuration; DitherLevel = src.DitherLevel; FilterSize = src.FilterSize; SampleFactor = src.SampleFactor; return *this; } 
96 
97 tPixelFormat Format; // See comment above. Must be one of the PALNBIT formats wher N is E [1, 8]. 
98 tQuantize::Method Method; // See comment above. Choose one of the 4 available colour quantization methods. 
99 int Loop; // See comment above. Animated only. 0 = infinite (default). >0 = that many times. 
100 int AlphaThreshold; // See comment above. -1 = auto. 255 = opaque. else A<=threshold = transp pixel. 
101 int OverrideFrameDuration; // See comment above. -1 = use frame duration. >=0 = Set all to this many 1/100 sec. 
102 
103 double DitherLevel; // For Method::Spatial only. 0.0 = auto. >0.0 = manual dither amount. 
104 int FilterSize; // For Method::Spatial only. Must be 1, 3, or 5. Default is 3. 
105 
106 int SampleFactor; // For Method::Neu only. 1 = whole image learning. 10 = 1/10th image used. Max is 30. 
107 };

156inline tFrame* tImage::tImageGIF::StealFrame(int frameNum) 
157{ 
158 tFrame* f = GetFrame(frameNum); 
159 if (!f) 
160 return nullptr; 
161 
162 return Frames.Remove(item: f); 
163} 
164 
165 
166inline void tImage::tImageGIF::StealFrames(tList<tFrame>& frames) 
167{ 
168 while (tFrame* frame = Frames.Remove()) 
169 frames.Append(item: frame); 
170}

173inline tFrame* tImage::tImageGIF::GetFrame(int frameNum) 
174{ 
175 if ((frameNum >= Frames.GetNumItems()) || (frameNum < 0)) 
176 return nullptr; 
177 
178 tFrame* f = Frames.First(); 
179 while (frameNum--) 
180 f = f->Next(); 
181 
182 return f; 
183}

186inline void tImageGIF::Clear() 
187{ 
188 Width = 0; 
189 Height = 0; 
190 while (tFrame* frame = Frames.Remove())
