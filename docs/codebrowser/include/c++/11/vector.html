106 template<typename _Tp, typename _Alloc, typename _Predicate> 
107 inline typename vector<_Tp, _Alloc>::size_type 
108 erase_if(vector<_Tp, _Alloc>& __cont, _Predicate __pred) 
109 { 
110 const auto __osz = __cont.size(); 
111 __cont.erase(std::remove_if(__cont.begin(), __cont.end(), __pred), 
112 __cont.end()); 
113 return __osz - __cont.size(); 
114 } 
115 
116 template<typename _Tp, typename _Alloc, typename _Up> 
117 inline typename vector<_Tp, _Alloc>::size_type 
118 erase(vector<_Tp, _Alloc>& __cont, const _Up& __value) 
119 { 
120 const auto __osz = __cont.size(); 
121 __cont.erase(std::remove(__cont.begin(), __cont.end(), __value), 
122 __cont.end()); 
123 return __osz - __cont.size(); 
124 }
