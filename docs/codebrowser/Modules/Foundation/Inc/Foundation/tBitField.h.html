3// A tBitField is a fixed size array of bits. Similar to STL bitset class. An tBitField needs to know how many bits 
4// will be stored at compile time and there is no possibility to grow or dynamically change that number. All bitwise 
5// operators are overloaded appropriately. This class is perfect for flags where a uint32 or uint64 is not enough. 
6// 
7// Comparisons 
8// tBitArray - Use when you want to store a large number of bits and you don't know how many at compile-time. 
9// This type os primatily for storage and access to a large number of bits. Not many bitwise or 
10// mathematical operators. 
11// tBitField - Use when know how many bits at compile-time and you want bitwise logic opertors like and, or, xor, 
12// shift, not, etc. Good for storing a fixed number of flags or channels etc. 
13// tFixInt - Use when you want full mathematical operations like any built-in integral type. Size must be known at 
14// compile time and must be a multiple of 32 bits. You get + - / * etc as well as all bitwise logic ops.

30// The tBitField class. NumBits represents the number of bits that can be stored by the instance. There are no 
31// conditions on the value of NumBits used and long as it is a whole number. If NumBits is a multiple of 32, the 
32// tBitField will be supported by tPrintf so just call something like: tsPrintf(s, "%032|128X", bitvar128) to convert it 
33// to a string. The memory image size taken up will always be a multiple of 4 bytes. ex: sizeof(tBitField<16>) = 4 and 
34// sizeof(tBitField<33>) = 8. You can still use tPrintf on a 33-bit bit-field, just be aware of the size. Any padding 
35// bits are guaranteed to be clear in the internal representation and when saved to a chunk (disk) format.

173// The tbitNNN types represent convenient bit-field sizes. They can represent large sets of bits and even allow bit 
174// operations. They are a little slower that native 32 or 64 bit integers and do not support many math operations. For 
175// full arithmetic support in a 128-bit or larger integer consider using the tFixInt class. Note that since bit-fields 
176// do not support arithmetic, they should be considered unsigned at all times. For example, the right shift >> operator 
177// does not sign extend. 
178typedef tBitField<128> tbit128; 
179typedef tBitField<256> tbit256; 
180typedef tBitField<512> tbit512;

186template<int N> inline void tBitField<N>::Set(const char* hexStr) 
187{ 
188 if (!hexStr) 
189 { 
190 ClearAll(); 
191 return; 
192 } 
193 
194 tBitField< NumElements*32 > val; 
195 tAssert(NumElements == val.GetNumElements()); 
196 tString hex(hexStr); 
197 hex.ToUpper(); 
198 hex.ExtractLeft(prefix: "0X"); 
199 int len = hex.Length(); 
200 int nybIdx = 0; 
201 for (int i = len-1; i >= 0; i--) 
202 { 
203 char nybChar = hex[i]; 
204 switch (nybChar) 
205 { 
206 case '0': val.SetNybble(nybIdx++, 0x0); break; 
207 case '1': val.SetNybble(nybIdx++, 0x1); break; 
208 case '2': val.SetNybble(nybIdx++, 0x2); break; 
209 case '3': val.SetNybble(nybIdx++, 0x3); break; 
210 case '4': val.SetNybble(nybIdx++, 0x4); break; 
211 case '5': val.SetNybble(nybIdx++, 0x5); break; 
212 case '6': val.SetNybble(nybIdx++, 0x6); break; 
213 case '7': val.SetNybble(nybIdx++, 0x7); break; 
214 case '8': val.SetNybble(nybIdx++, 0x8); break; 
215 case '9': val.SetNybble(nybIdx++, 0x9); break; 
216 case 'A': val.SetNybble(nybIdx++, 0xA); break; 
217 case 'B': val.SetNybble(nybIdx++, 0xB); break; 
218 case 'C': val.SetNybble(nybIdx++, 0xC); break; 
219 case 'D': val.SetNybble(nybIdx++, 0xD); break; 
220 case 'E': val.SetNybble(nybIdx++, 0xE); break; 
221 case 'F': val.SetNybble(nybIdx++, 0xF); break; 
222 } 
223 // We simply skip unrecognized characters. 
224 } 
225  
226 for (int i = 0; i < NumElements; i++) 
227 ElemData[i] = val.GetElement(i); 
228 
229 ClearPadBits(); 
230}

235 if (!binaryStr)

241 tBitField< NumElements*32 > val;

243 tString bin(binaryStr);

255 
256 // We simply skip unrecognized characters. 
257 }

260 ElemData[i] = val.GetElement(i); 
261 
262 ClearPadBits(); 
263}

384template<int N> inline tString tBitField<N>::GetAsHexString() const 
385{ 
386 tBitField< NumElements*32 > val; 
387 tAssert(NumElements == val.GetNumElements()); 
388 
389 for (int i = 0; i < NumElements; i++) 
390 val.SetElement(i, ElemData[i]); 
391 
392 int numNybbles = 8*NumElements; 
393 tString result(numNybbles); 
394 for (int ny = 0; ny < numNybbles; ny++) 
395 { 
396 uint8 nybble = val.GetNybble(ny); 
397 int n = numNybbles-1-ny; 
398 switch (nybble) 
399 { 
400 case 0x00: result[n] = '0'; break; 
401 case 0x01: result[n] = '1'; break; 
402 case 0x02: result[n] = '2'; break; 
403 case 0x03: result[n] = '3'; break; 
404 case 0x04: result[n] = '4'; break; 
405 case 0x05: result[n] = '5'; break; 
406 case 0x06: result[n] = '6'; break; 
407 case 0x07: result[n] = '7'; break; 
408 case 0x08: result[n] = '8'; break; 
409 case 0x09: result[n] = '9'; break; 
410 case 0x0A: result[n] = 'A'; break; 
411 case 0x0B: result[n] = 'B'; break; 
412 case 0x0C: result[n] = 'C'; break; 
413 case 0x0D: result[n] = 'D'; break; 
414 case 0x0E: result[n] = 'E'; break; 
415 case 0x0F: result[n] = 'F'; break; 
416 } 
417 } 
418 result.RemoveLeading(theseChars: "0");

488template<int N> inline bool tBitField<N>::AreAll(bool v) const 
489{ 
490 // To test all clear we rely on any extra bits being cleared as well. 
491 if (!v) 
492 { 
493 for (int i = 0; i < NumElements; i++) 
494 if (ElemData[i] != 0x00000000) 
495 return false; 
496 return true; 
497 } 
498 
499 for (int i = 0; i < NumElements-1; i++) 
500 if (ElemData[i] != 0xFFFFFFFF) 
501 return false; 
502 
503 // The last slot in the array may not be full. 
504 int last = N & 0x1F; 
505 uint32 maxFull = (last ? (1<<last) : 0) - 1; 
506 
507 return (ElemData[NumElements-1] == maxFull) ? true : false; 
508}

511template<int N> inline int tBitField<N>::CountBits(bool v) const 
512{ 
513 // First compute the total number set. 
514 int numSet = 0; 
515 for (int i = 0; i < NumElements; ++i) 
516 { 
517 uint32 v = ElemData[i]; // Count the number of bits set in v. 
518 uint32 c; // c accumulates the total bits set in v. 
519 for (c = 0; v; c++) 
520 v &= v - 1; // Clear the least significant bit set. 
521 numSet += c; 
522 } 
523 
524 // Now numSet is correct. If that's what we were asked, we're done. If not, we just subtract. 
525 if (v) 
526 return numSet; 
527 
528 return N - numSet; 
529}

543template<int N> inline void tBitField<N>::SetByte(int n, uint8 b) 
544{ 
545 int numBytes = (N / 8) + ((N % 8) ? 1 : 0); 
546 tAssert(n < numBytes); 
547 int idx = n / 4; 
548 int shift = (n % 4) << 3; 
549 uint32 elem = ElemData[idx]; 
550 elem = ~(0xFF << shift) & elem; // Clear the byte. 
551 ElemData[idx] = elem | (b << shift); // Set the byte.
