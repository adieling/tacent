3// A simple palettized image. Comprised of Width x Height pixel data storing indexes into a palette. The palette is 
4// simply an array of tPixels (RGB). Index resolution is determined by the pixel format (1 to 8 bits). The number of 
5// palette entries (colours) is 2 ^ the index-resolution.

25// A simple palettized image class supporting 1 to 8 bits per pixel-index. Origin is at the bottom-left and rows are 
26// ordered left to right moving up the image. This palette only contains RGB values (no alpha). Formats like gif handle 
27// (binary) alpha separately, and colour quantizers work on RGB values, so no alpha for tPaletteImage. 
28class tPaletteImage 
29{ 
30public: 
31 // Creates an invalid image. 
32 tPaletteImage() { } 
33 
34 tPaletteImage(const tPaletteImage& src) { Set(src); } 
35 
36 // The pixel format must be one of the PALNNIDX formats. The palette size is determined by the pixel format. This 
37 // constructor creates a palette with all black colours and every pixel indexing the first palette entry. Note that 
38 // internally there may be padding of the pixel-data for some palette pixel formats. It pads to 8 bits. For example, 
39 // if the pixel format is PAL1BIT (1 bit per pixel) and you have a 9x10 image, you will need 90 bits. That requires 
40 // 12 8-bit chunks costing 96 bits total. The last 6 bits get padded with 0. 
41 tPaletteImage(tPixelFormat fmt, int width, int height) { Set(fmt, width, height); } 
42 
43 // This is similar to above except you can construct a full image with palette and pixel-data. Caller is responsibe 
44 // for making sure a) the palette is the correct length, and b) the pixel-data indexes are the correct size and 
45 // there are width*height of them. The supplied pixelData array should include any necessary padding. eg. a 10x10 
46 // PAL1BIT image should be 13 bytes of data with 4 bits padded at the end. The data is copied out of the supplied 
47 // arrays. You can use tGetBitsPerPixel(tPixelFormat) to give you the size in bits of each pixel-index. Call tPow2 
48 // with that value to give you the palette length needed. 
49 tPaletteImage(tPixelFormat fmt, int width, int height, const uint8* pixelData, const tColour3b* palette) { Set(fmt, width, height, pixelData, palette); } 
50 
51 // This is the workhorse constructor because it needs to quantize the present colours to create the palette. 
52 // Quantizing, or rather doing a good job of quantizing, is quite complex. The NeuQuant algorithm uses a neural net 
53 // to accomplish this and gives good results. Alpha is ignored in the pixel array. 
54 tPaletteImage(tPixelFormat fmt, int width, int height, const tPixel4b* pixels, tQuantize::Method quantMethod) { Set(fmt, width, height, pixels, quantMethod); } 
55 
56 // Same as above but processes pixel data in RGB directly. 
57 tPaletteImage(tPixelFormat fmt, int width, int height, const tPixel3b* pixels, tQuantize::Method quantMethod) { Set(fmt, width, height, pixels, quantMethod); } 
58 
59 virtual ~tPaletteImage() { Clear(); } 
60 
61 // Frees internal layer data and makes the layer invalid. 
62 void Clear() { PixelFormat = tPixelFormat::Invalid; Width = Height = 0; delete[] PixelData; PixelData = nullptr; delete[] Palette; Palette = nullptr; } 
63 
64 // See the corresponding constructor comments for the set calls. 
65 bool Set(const tPaletteImage&); 
66 bool Set(tPixelFormat, int width, int height); 
67 bool Set(tPixelFormat, int width, int height, const uint8* pixelData, const tColour3b* palette); 
68 bool Set(tPixelFormat, int width, int height, const tPixel4b* pixels, tQuantize::Method); 
69 bool Set(tPixelFormat, int width, int height, const tPixel3b* pixels, tQuantize::Method); 
70 
71 // Populates the supplied pixel array. It is up to you to make sure there is enough room for width*height tPixels in 
72 // the supplied array. Returns success. You'll get false if either this image is invalid or null is passed in. 
73 bool Get(tPixel4b* pixels); 
74 
75 // Same as above but populates an RGB array. 
76 bool Get(tPixel3b* pixels); 
77 
78 bool IsValid() const { return (PixelData && Palette && Width && Height && tIsPaletteFormat(format: PixelFormat)); } 
79 
80 // Returns the size of the data in bytes by reading the Width, Height, and PixelFormat. Includes padding. The 
81 // returned value is always a multiple of 4 butes. 
82 int GetDataSize() const; 
83 
84 // Returns the size of the palette in tColour3b's. 
85 int GetPaletteSize() const; 
86 
87 tPixelFormat PixelFormat = tPixelFormat::Invalid; 
88 int Width = 0; 
89 int Height = 0; 
90 uint8* PixelData = nullptr; 
91 tColour3b* Palette = nullptr; 
92};
