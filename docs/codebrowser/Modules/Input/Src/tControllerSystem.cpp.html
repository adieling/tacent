46tControllerSystem::tControllerSystem(int pollingPeriod, int detectionPeriod) 
47{ 
48 Gamepads.reserve(n: int(tGamepadID::NumGamepads)); 
49 Gamepads.emplace_back(args: "Gamepad1", args: tGamepadID::GP0); 
50 Gamepads.emplace_back(args: "Gamepad2", args: tGamepadID::GP1); 
51 Gamepads.emplace_back(args: "Gamepad3", args: tGamepadID::GP2); 
52 Gamepads.emplace_back(args: "Gamepad4", args: tGamepadID::GP3); 
53 
54 // If auto-detect is turned on there is no need to set the PollingPeriod. 
55 if (pollingPeriod > 0) 
56 PollingPeriod = pollingPeriod; 
57 else 
58 PollingPeriodAutoDetect = true; 
59 
60 if (detectionPeriod > 0) 
61 DetectPeriod = detectionPeriod; 
62 
63 #ifdef PLATFORM_WINDOWS 
64 // This is better than LoadLibrary(TEXT("XInput1_4.dll") in two ways: 
65 // 1) The module is already loaded using the import library. 
66 // 2) We use the xinput.h header define in case xinput is ever updated. 
67 HMODULE moduleHandle = GetModuleHandle(XINPUT_DLL); 
68 XInputGetCapabilitiesEx = (_XInputGetCapabilitiesEx)GetProcAddress(moduleHandle, (char*)108); 
69 #endif 
70 
71 DetectThread = std::thread(&tControllerSystem::Detect, this); 
72}

75tControllerSystem::~tControllerSystem() 
76{ 
77 { 
78 std::lock_guard<std::mutex> lock(Mutex); 
79 DetectExitRequested = true; 
80 } 
81 // Notify that we want to cooperatively stop the polling thread. Notify one (thread) should be sufficient. 
82 // Notify_all would also work but is overkill since only one (polling) thread is waiting. By using a condition 
83 // variable we've made it so we don't have to wait for the current polling cycle sleep to complete. 
84 DetectExitCondition.notify_one(); 
85 
86 // The join just blocks until the polling thread has finished -- and has responded to the notify above. 
87 DetectThread.join(); 
88}

91void tControllerSystem::Detect() 
92{ 
93 while (1) 
94 { 
95 // Detect controllers connected and disconnected. 
96 #ifdef PLATFORM_WINDOWS 
97 
98 for (int g = 0; g < int(tGamepadID::NumGamepads); g++) 
99 { 
100 // XInputGetState is generally faster for detecting device connectedness. 
101 WinXInputState state; 
102 tStd::tMemclr(&state, sizeof(WinXInputState)); 
103 WinDWord result = XInputGetState(g, &state); 
104 
105 if (result == WinErrorSuccess) 
106 { 
107 // If we're already polling then move on. 
108 if (Gamepads[g].IsPolling()) 
109 continue; 
110 
111 // Controller connected. Can go ahead and try to figure out polling rate to use. 
112 int pollingPeriod = PollingPeriod; 
113 if (PollingPeriodAutoDetect) 
114 { 
115 // In auto detect mode we try to determin the polling rate from the hardware. 
116 // Do not modify pollingPeriod if detection fails. 
117 int optimalPollingPeriod = DetermineGamepadPollingPeriodFromHardwareInfo(tGamepadID(g)); 
118 if (optimalPollingPeriod) 
119 pollingPeriod = optimalPollingPeriod; 
120 } 
121 tAssert(pollingPeriod > 0); 
122 
123 // Now we need to start polling the controller and queue a message that a controller has 
124 // been connected for the main Update to pick up. 
125 Gamepads[g].StartPolling(pollingPeriod, tGamepadID(g)); 
126 } 
127 else 
128 { 
129 // Either WinErrorDeviceNotConnected or some other error. Either way treat controller as disconnected. 
130 if (Gamepads[g].IsPolling()) 
131 { 
132 // Now we need to stop polling and queue a disconnect message to for the main update to pick up. 
133 Gamepads[g].StopPolling(); 
134 } 
135 } 
136 } 
137 #endif 
138 
139 static int detectNum = 0; 
140 tPrintf(f: "Detect: %d\n", detectNum++); 
141 
142 // This unique_lock is just a more powerful version of lock_guard. Supports subsequent unlocking/locking which 
143 // is presumably needed by wait_for. In any case, wait_for needs this type of lock. 
144 std::unique_lock<std::mutex> lock(Mutex); 
145 bool exitRequested = DetectExitCondition.wait_for(lock&: lock, rtime: std::chrono::milliseconds(DetectPeriod), p: [this]{ return DetectExitRequested; }); 
146 if (exitRequested) 
147 break; 
148 } 
149}

152int tControllerSystem::DetermineGamepadPollingPeriodFromHardwareInfo(tGamepadID g) 
153{ 
154 #ifdef PLATFORM_WINDOWS 
155 
156 int i = int(g); 
157 tPrintf("Gamepad %d ", i); 
158 
159 WinXInputCapabilitiesEx capsEx; 
160 tStd::tMemclr(&capsEx, sizeof(WinXInputCapabilitiesEx)); 
161 if (XInputGetCapabilitiesEx(1, i, 0, &capsEx) == WinErrorSuccess) 
162 { 
163 tVidPid vidpid(uint16(capsEx.vendorId), uint16(capsEx.productId)); 
164 const tContDefn* defn = tLookupContDefn(vidpid); 
165 const char* vendor = defn ? defn->Vendor : "unknown"; 
166 const char* product = defn ? defn->Product : "unknown"; 
167 tPrintf 
168 ( 
169 "connected. vid = 0x%04X pid = 0x%04X Vendor:%s Product:%s\n", 
170 int(capsEx.vendorId), int(capsEx.productId), vendor, product 
171 ); 
172 if (defn && defn->MaxPollingFreq) 
173 return int(1000000.0f/defn->MaxPollingFreq); 
174 } 
175 else 
176 { 
177 tPrintf("not connected\n"); 
178 } 
179 
180 #else 
181 #endif 
182 return 0; 
183}
