27// A tImageKTX object represents and knows how to load a ktx and ktx2 files. In general a Khronos Texture is composed of 
28// multiple layers -- each one a mipmap. It loads the data into tLayers. It can either decode to R8G8B8A8 layers, or leave 
29// the data as-is. Decode from BCn is supported. The layers may be 'stolen' from a tImageKTX so that excessive memcpys 
30// are avoided. After they are stolen the tImageKTX is invalid. Cubemaps and mipmaps are supported. 
31// @todo 1D and 3D textures are not supported yet. 
32// @todo ASTC is not supported yet. 
33class tImageKTX : public tBaseImage 
34{ 
35public: 
36 enum LoadFlag 
37 { 
38 LoadFlag_Decode = 1 << 0, // Decode the ktx texture data into RGBA 32 bit layers. If not set, the layer data will remain unmodified. 
39 LoadFlag_ReverseRowOrder = 1 << 1, // OpenGL uses the lower left as the orig DirectX uses the upper left. Set flag for OpenGL. 
40 
41 // Gamma-correct. Gamma compression using an encoding gamma of 1/2.2. Flag only applies when decode flag set and 
42 // the file has been determined to be in linear-space. Linear-space is often determined by the pixel format 
43 // HDR / floating-point formats (BC6, rgb16f/32f, etc) images. Assumes (colour) data is linear and puts it in 
44 // gamma-space (brighter) for diaplay on a monitor. 
45 LoadFlag_GammaCompression = 1 << 2, 
46 LoadFlag_SRGBCompression = 1 << 3, // Same as above but uses the official sRGB transformation. Linear -> sRGB. Approx encoding gamma of 1/2.4 for part of curve. 
47 LoadFlag_AutoGamma = 1 << 4, // Tries to determine whether to apply sRGB compression based on pixel format. Call GetColourProfile to see if it applied. 
48 LoadFlag_ToneMapExposure = 1 << 5, // Apply exposure value when loading the ktx. Only affects HDR (linear-colour) formats. 
49 LoadFlag_SpreadLuminance = 1 << 6, // For KTX files with a single Red or Luminance component, spread it to all the RGB channels (otherwise red only). Does not spread single-channel Alpha formats. Applies only if decoding a ktx is an R-only or L-only format. 
50 LoadFlag_CondMultFourDim = 1 << 7, // Produce conditional success if image dimension not a multiple of 4. Only checks BC formats, 
51 LoadFlag_CondPowerTwoDim = 1 << 8, // Produce conditional success if image dimension not a power of 2. Only checks BC formats. 
52 LoadFlag_SwizzleBGR2RGB = 1 << 9, // Compressonator stores colours swizzled in their ETC exports. This fixes those files up. 
53 LoadFlags_Default = LoadFlag_Decode | LoadFlag_ReverseRowOrder | LoadFlag_SpreadLuminance | LoadFlag_AutoGamma 
54 };

56 // If an error is encountered loading the resultant object will return false for IsValid. You can call GetLastResult 
57 // to get more detailed information. There are some results that are not full-success that leave the object valid. 
58 // When decoding _and_ reversing row order, most BC 4x4 blocks can be massaged without decompression to fix the row 
59 // order. The more complex ones like BC6 and BC7 cannot be swizzled around like this (well, they probably could be, 
60 // but it's a non-trivial amount of work). 
61 // 
62 // A note on LoadFlag_ReverseRowOrder. tImageKTX tries to perform row-reversing before any decode operation. This is 
63 // often possible even if the KTX texture data is BC-compressed. However, for some of the more complex BC schemes 
64 // (eg. BC6 BC7) this reversal cannot be easily accomplished without a full decode and re-encode which would be 
65 // lossy. In these cases the row-reversal is done _after_ decoding. Unfortunalely decoding may not always be 
66 // requested (for example if you want to pass the image data directly to the GPU memory in OpenGL). In these cases 
67 // tImageKTX will be unable to reverse the rows. You will still get a valid object, but it will be a conditional 
68 // valid (GetStates() will have Conditional_CouldNotFlipRows flag set). You can call also call RowsReversed() to 
69 // see if row-reversal was performed. The conditional is only set if reversal was requested. 
70 // 
71 // Additional parameters may be processed during ktx-loading. Gamma is only used if GammaCompression flag is set. 
72 // Exposure >= 0 (black) and only used if ToneMapExposure flag set. 
73 struct LoadParams 
74 { 
75 LoadParams() { Reset(); } 
76 LoadParams(const LoadParams& src) : Flags(src.Flags), Gamma(src.Gamma), Exposure(src.Exposure) { } 
77 void Reset() { Flags = LoadFlags_Default; Gamma = tMath::DefaultGamma; Exposure = 1.0f; } 
78 LoadParams& operator=(const LoadParams& src) { Flags = src.Flags; Gamma = src.Gamma; Exposure = src.Exposure; return *this; } 
79 
80 uint32 Flags; 
81 float Gamma; 
82 float Exposure; 
83 };

186 // After calling StealLayers the current object will be invalid. This call populates the passed in layer list. If 
187 // the current object is not valid the passed-in layer list is left unmodified. The layer list is appended to. It is 
188 // not emptied if there are layers on the list when passed in. This call gives management of the layers to the 
189 // caller. It does not memcpy and create new layers which is why the object becomes invalid afterwards. If the 
190 // tImageKTX is a cubemap, this function returns false and leaves the object (and list) unmodified. See 
191 // StealCubemapLayers if you want to steal cubemap layers. 
192 bool StealLayers(tList<tLayer>&);

199 // Similar to StealLayers except it steals up to 6 layer-lists if the object is a cubemap. If the tImageKTX
