3// Linked list implementations. There are two implementations in this module. tList is intrusive, tItList (i for 
4// iterator) is non-intrusive. Use the intrusive one for performance and fewer fragmentation issues if possible. 
5// 
6// tList advantages: Faster and less memory fragmentation (one new per object). 
7// tList disadvantages: An object can only be on one list at a time. You must derive from tLink. 
8// 
9// tItList advantages: The same item only in one list at a time. No change in memory image for the objects. Cleaner 
10// iterator syntax similar to the STL containers. Supports the new C++11 range-based for loop syntax. 
11// tItList disadvantages: More memory allocs. Not quite as fast.

60// A tList is an intrusive list implementation. Intrusive results in less memory fragmentation but objects can only be 
61// on one list at a time. A tList may be used in the cdata (global) memory section where zero-initialization is 
62// guaranteed. In this mode it does NOT delete its items on destruction to avoid both the init and de-init fiasco. 
63// tLists may also be used on the stack or heap, in which case it is optional to construct with the ListOwns flag. It 
64// is vital that for zero-init you call the constructor with mode as StaticZero IF you intend to have it populated by 
65// other statically initialized objects before main() is entered. This is because we don't know what items will have 
66// already been put on the list before the constructor is called (again, the C++ init fiasco). The constructor in 
67// StaticZero mode does NOT clear the state variables so it doesn't matter when. 
68// 
69// In static mode, the list does not consider itself to own the things on the list. It is safe to add 'this' 
70// pointers to globally constructed objects for instance. If however you do want the list to delete the items, tList 
71// will allow you to call Empty on a static-zero list. You can also call Reset (no deletes). Clear will be the same 
72// as Reset for static-zero lists. For the implementation, the main difference between Static and External is that 
73// with Static the constructor does not clear the head, tail, and count members because the constructor may be called 
74// after items are added. 
75template<typename T> class tList 
76{ 
77public: 
78 // The default constructor has the list owning the items. List mode is Internal. 
79 tList() : Mode(tListMode::ListOwns), HeadItem(nullptr), TailItem(nullptr), ItemCount(0) { } 
80 
81 // If mode is external the objects will not be deleted when the list is destroyed. You manage item lifetime. 
82 tList(tListMode mode) : Mode(mode) { if (mode != tListMode::StaticZero) { HeadItem = nullptr; TailItem = nullptr; ItemCount = 0; } } 
83  
84 virtual ~tList() { if (Owns()) Empty(); } 
85  
86 T* Insert(T* item); // Insert item at head. Returns item. 
87 T* Insert(T* item, T* here); // Insert item before here. Returns item. 
88 T* Append(T* item); // Append item at tail. Returns item. 
89 T* Append(T* item, T* here); // Append item after here. Returns item. 
90 T* Remove(T* item); // Removes and returns item. 
91 T* Remove(); // Removes and returns head item. 
92 T* Drop(); // Removes and returns tail item. 
93 
94 void Clear() /* Clears the list. Deletes items if list owns them. */ { if (Owns()) Empty(); else Reset(); } 
95 void Reset() /* Resets the list. Never deletes the objects. */ { HeadItem = nullptr; TailItem = nullptr; ItemCount = 0; } 
96 void Empty() /* Empties the list. Always deletes the objects. */ { while (!IsEmpty()) delete Remove(); }

104 T* PrevCirc(const T* here) const /* Circular. Gets item before here. */ { return here->PrevItem ? here->PrevItem : TailItem; }

113 // Sorts the list using the algorithm specified. The supplied compare function should never return true on equal. 
114 // To sort ascending return the truth of a < b. Return a > b to sort in descending order. Returns the number of 
115 // compares performed. 
116 // 
117 // A simple compare function would implement "bool CompareFunc(const T& a, const T& b);" 
118 // If you want to send user information into your compare function, there is no need to create a static or global 
119 // object that is error-prone (and definitely not thread-safe). There is also no need for the Sort function to be 
120 // polluted with an extra user-data argument. Instead use C++ FunctionObjects. A function object is an object that 
121 // implements operator(). As an example: 
122 // 
123 // struct CompareFunctionObject 
124 // { 
125 // bool ascending = true; // Plus whatever other sort-specification data you want. 
126 // bool operator() (const YourType& a, const YourType& b) const 
127 // { 
128 // return ascending ? a < b : a > b; // You implement the compare. 
129 // } 
130 // } 
131 // 
132 // To use this simply do somthing like: 
133 // 
134 // CompareFunctionObject comp; 
135 // comp.ascending = false; 
136 // MyList.Sort(comp); 
137 // 
138 template<typename CompareFunc> int Sort(CompareFunc, tListSortAlgorithm alg = tListSortAlgorithm::Merge); 
139 
140 // Inserts item in a sorted list. It will remain sorted. 
141 template<typename CompareFunc> T* Insert(T* item, CompareFunc); 
142 
143 // This does an O(n) single pass of a bubble sort iteration. Allows the cost of sorting to be distributed over time 
144 // for objects that do not change their order very often. Do the expensive merge sort when the list is initially 
145 // populated, then use this to keep it approximately correct. 
146 // 
147 // The direction is important when doing this type of partial sort. A forward bubble will result in the last object 
148 // in the list being correct after one pass, last two for two passes etc. If getting the front of the list correct 
149 // sooner is more important you'll want to bubble backwards. This is true regardless of whether the compare 
150 // function implements an ascending or descending sort. 
151 // 
152 // maxCompares sets how far into the list to perform possible swaps. If maxCompares == -1 it means go all the way. 
153 // That is, GetNumItems() - 1. Returns number of swaps performed. 
154 template<typename CompareFunc> int Bubble(CompareFunc compare, bool backwards = false, int maxCompares = -1) { if (backwards) return BubbleBackward(compare, maxCompares); else return BubbleForward(compare, maxCompares); } 
155 
156private: 
157 // These return the number of swaps performed. If 0 is returned, the items considered were already in the correct 
158 // order. The maxCompares allows you to limit how many compares are performed. -1 means numItems-1 compares will be 
159 // made (a full sweep). The variable gets clamped to [0, numItems) otherwise. 
160 template<typename CompareFunc> int BubbleForward(CompareFunc, int maxCompares = -1); 
161 template<typename CompareFunc> int BubbleBackward(CompareFunc, int maxCompares = -1); 
162 
163 // These return the number of compares performed. 
164 template<typename CompareFunc> int SortMerge(CompareFunc); 
165 template<typename CompareFunc> int SortBubble(CompareFunc);

168 tListMode Mode; 
169 T* HeadItem; 
170 T* TailItem; 
171 int ItemCount;

269 public:

272 
273 // Iterators may be dereferenced to get to the object. 
274 T& operator*() const { return *((T*)Node->Object); } 
275 T* operator->() const { return (T*)Node->Object; }

279 operator T*() { return GetObject(); }

284 const Iter operator++(int) { Iter curr(*this); Next(); return curr; }

286 const Iter operator++() { Next(); return *this; } 
287 const Iter operator--() { Prev(); return *this; } 
288 const Iter operator+(int offset) const { tAssert(offset >= 0); Iter i = *this; while (offset--) i.Next(); return i; } 
289 const Iter operator-(int offset) const { tAssert(offset >= 0); Iter i = *this; while (offset--) i.Prev(); return i; } 
290 Iter& operator+=(int offset) { tAssert(offset >= 0); while (offset--) Next(); return *this; } 
291 Iter& operator-=(int offset) { tAssert(offset >= 0); while (offset--) Prev(); return *this; } 
292 bool operator==(const Iter& i) const { return (Node == i.Node) && (List == i.List); } 
293 bool operator!=(const Iter& i) const { return (Node != i.Node) || (List != i.List); } 
294 
295 bool IsValid() const { return Node ? true : false; } 
296 void Clear() { Node = nullptr; List = nullptr; } 
297 void Next() { if (Node) Node = Node->Next(); } 
298 void Prev() { if (Node) Node = Node->Prev(); } 
299

301 void NextCirc() { if (Node) Node = Node->Next(); if (!Node) Node = List->Nodes.Head(); } 
302 void PrevCirc() { if (Node) Node = Node->Prev(); if (!Node) Node = List->Nodes.Tail(); } 
303 T* GetObject() const { return (T*)(Node ? Node->Object : nullptr); } 
304

423 for (T* contender = Head(); contender; contender = contender->Next())

542template<typename T> template<typename CompareFunc> inline int tList<T>::SortMerge(CompareFunc compare) 
543{ 
544 if (!HeadItem) 
545 return 0; 
546 
547 // Treat every node as a separate list, completely sorted, starting with 1 element each. 
548 int numNodesPerList = 1; 
549 int numCompares = 0; 
550 
551 while (1) 
552 { 
553 T* p = HeadItem; 
554 HeadItem = nullptr; 
555 TailItem = nullptr; 
556 
557 // Num merges in this loop. 
558 int numMerges = 0; 
559 
560 while (p) 
561 { 
562 numMerges++; 
563 T* q = p; 
564 int numPNodes = 0; 
565 for (int i = 0; i < numNodesPerList; i++) 
566 { 
567 numPNodes++; 
568 q = q->Next(); 
569 if (!q) 
570 break; 
571 } 
572 
573 int numQNodes = numNodesPerList; 
574 
575 // Merge the two lists. 
576 while (numPNodes > 0 || (numQNodes > 0 && q)) 
577 { 
578 // Decide whether next tBaseLink of merge comes from p or q. 
579 T* e; 
580 
581 if (numPNodes == 0) 
582 { 
583 // p is empty; e must come from q. 
584 e = q; 
585 q = q->Next(); 
586 
587 numQNodes--; 
588 } 
589 else if (numQNodes == 0 || !q) 
590 { 
591 // q is empty; e must come from p. 
592 e = p; 
593 p = p->Next(); 
594 
595 numPNodes--; 
596 } 
597 else if (++numCompares && !compare(*q, *p)) 
598 { 
599 // p is lower so e must come from p. 
600 e = p; 
601 p = p->Next(); 
602  
603 numPNodes--; 
604 } 
605 else 
606 { 
607 // First gBaseNode of q is bigger or equal; e must come from q. 
608 e = q; 
609 q = q->Next(); 
610  
611 numQNodes--; 
612 } 
613 
614 // add the next gBaseNode to the merged list. 
615 if (TailItem) 
616 TailItem->NextItem = e; 
617 else 
618 HeadItem = e; 
619 
620 e->PrevItem = TailItem; 
621 TailItem = e; 
622 } 
623 
624 // P and Q have moved numNodesPerList places along. 
625 p = q; 
626 } 
627 TailItem->NextItem = nullptr; 
628 
629 // If we have done only one merge, we're all sorted. 
630 if (numMerges <= 1) 
631 return numCompares; 
632 
633 // Otherwise repeat, merging lists twice the size. 
634 numNodesPerList *= 2; 
635 } 
636 
637 return numCompares; 
638}

659template<typename T> template<typename CompareFunc> inline int tList<T>::BubbleForward(CompareFunc compare, int maxCompares) 
660{ 
661 // @todo Can -1 ever happen? 
662 if (maxCompares == -1) 
663 maxCompares = ItemCount-1; 
664 else if (maxCompares > ItemCount-1) 
665 maxCompares = ItemCount-1; 
666 
667 // If there are no items, only a single item, or no compares are requested, we're done. 
668 if ((ItemCount < 2) || (maxCompares == 0)) 
669 return 0; 
670 
671 T* a = HeadItem; 
672 int numSwaps = 0; 
673 int numCompares = 0; 
674 
675 while ((a != TailItem) && (numCompares < maxCompares)) 
676 { 
677 T* b = a->NextItem; 
678 
679 // We're sorting from lowest to biggest, so if b < a we need to swap them. 
680 if (compare(*b, *a)) 
681 { 
682 // Swap. 
683 if (a->PrevItem) 
684 a->PrevItem->NextItem = b; 
685 
686 if (b->NextItem) 
687 b->NextItem->PrevItem = a; 
688 
689 a->NextItem = b->NextItem; 
690 b->PrevItem = a->PrevItem; 
691 
692 a->PrevItem = b; 
693 b->NextItem = a; 
694 
695 // Fix head and tail if they were involved in the swap. 
696 if (HeadItem == a) 
697 HeadItem = b; 
698 
699 if (TailItem == b) 
700 TailItem = a; 
701 
702 // Since we swapped, a is now correctly ready for the next loop. 
703 numSwaps++; 
704 } 
705 else 
706 { 
707 a = a->NextItem; 
708 } 
709 numCompares++; 
710 } 
711 
712 return numSwaps; 
713}
