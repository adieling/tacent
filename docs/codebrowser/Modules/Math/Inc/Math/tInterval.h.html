3// Intervals and IntervalSets. A tInterval is a set of numbers (over a discrete or continuous domain) that has a start 
4// and an end. Mathematical 'interval notation' is used to represent such a set. In this notation square brackets [] 
5// mean the endpoint is included, and normal brackets / US-parens () means the endpoint is excluded. 
6// For example, [0,5) over integers represents the set { 0, 1, 2, 3, 4 } 
7// 
8// The tIntervalSet class represents collections of possibly disjoint intervals. For example, the interval set: 
9// [0,3)U(10,14) over integers represents the set { 0, 1, 2, 11, 12, 13 }. The tIntervalSet class has the ability to 
10// add new intervals to a set, detect overlaps and/or joins, and represent the new set in the simplest possible form.

37// The tInterval type represents a single interval over integral types. The string form follows the mathematical notation 
38// (a,b), (a,b], [a,b), or [a,b]. In this representation the square brackets mean inclusive and normal brackets 
39// (US parens) are exclusive. 
40// The interval 5 -> [5,5] -> { 5 } 
41// The interval [0,5) -> { 0 1 2 3 4 } 
42// The interval (5,5) -> empty 
43// The interval [5,5) -> empty 
44// The interval (5,5] -> empty 
45// The interval [5,5] -> { 5 } 
46// The interval (4,5] -> { 5 } 
47// The interval (4,5) -> empty 
48// @todo In the future we will rename tInterval to tIntervalDiscrete and tunn it into a template that supports any 
49// integral type: int, uint, int64, etc.

51{

53 tInterval() { Set(a: 0, b: 0, bias: tBias::Inner); } 
54 tInterval(const tInterval& src) { Set(src); } 
55 tInterval(int a, int b, tBias bias = tBias::Full) { Set(a, b, bias); }

57 // String representation is auto-detected. 
58 tInterval(const tString& s) { Set(s); }

60 void Clear() { Set(a: 0, b: 0, bias: tBias::Inner); }

62 // Returns true if the interval is not the empty set (contains nothing). Note that (4,5) is invalid for ints but 
63 // valid for floats/doubles. 
64 bool IsValid() const; 
65

127// This class represents a collection of multiple intervals. The intervals may not overlap, but if they do this class 
128// knows how to simplify the collection to the fewest number of intervals possible. 
129// @todo In future this will be renamed to tIntervalSetDiscrete and templatized to accept any integral type. 
130struct tIntervalSet 
131{ 
132 tIntervalSet() { Clear(); } 
133 tIntervalSet(const tIntervalSet& src) { Set(src); } 
134 tIntervalSet(const tString& s) { Set(s); } 
135 
136 void Clear() { Intervals.Empty(); } 
137 
138 // Returns true if this object has any intervals. They are guaranteed to be valid non-empty ones if it does. 
139 bool IsValid() const { return !Intervals.IsEmpty(); } 
140 bool IsEmpty() const { return Intervals.IsEmpty(); } 
141 
142 void Set(const tIntervalSet& src) { Clear(); for (tItList<tInterval>::Iter i = src.Intervals.First(); i; ++i) Intervals.Append(obj: new tInterval(*i)); } 
143 
144 // String should be in form "[4,6)U[5,8]" or "[4,6)|[5,8]". Think of the | as an 'or' (or U for union). It means a 
145 // value is in the set of intervals if it is in the first interval, _or_ the second, _or) the third etc. This set 
146 // call is 'smart' and will deal with overlaps between intervals. In the example above: 
147 // [4,6)|[5,8] -> [4,8]. You may also pass in a string using the 'Range' notation. See tIntervalRep enum.

154 // is the workhorse of this class allowing the interval-set to be built up consistently and in the simplest form.

175// @todo Implement tIntervalContinuous and tIntervalSetContinuous that are both templates that will support 
176// 'continuous' types like float and double. Although float and double are only approximations of continuous real 
177// numbers, they are much closer than the descrete integral types. See comments in the tInterval implementatons for 
178// an indication of where things will be different. In particular there will be no MakeInclusive ability and joining 
179// will not be necessary -- only properly overlapping intervals may be extended. This is because with a discrete number 
180// there is a well-defined next and previous value. With continuous numbers there is no 'next' number, as no matter 
181// what you choose, there are an infinite set of numbers in between. Basically intervals in the real number domain are 
182// quite different than intervals in the integer domain. 
183// 
184// The tIntervalContinuous classes (over float or double) will support the following example intervals: 
185// The interval 5.0 -> [5.0,5.0] -> { 5.0 } 
186// The interval [0.0,5.0) -> x ε R | 0.0 <= x <= 5.0 
187// The interval (5.0,5.0) -> empty 
188// The interval [5.0,5.0) -> empty 
189// The interval (5,0,5.0] -> empty 
190// The interval [5.0,5.0] -> { 5.0 } 
191// The interval (4.0,5.0] -> x ε R | 4.0 < x <= 5.0 
192// The interval (4.0,5.0) -> x ε R | 4.0 < x < 5.0

201inline bool tMath::tInterval::IsValid() const 
202{ 
203 // For integers any exclusive endpoints can be converted to inclusive. 
204 tInterval incForm(*this); 
205 incForm.MakeInclusive(); 
206 return (incForm.B >= incForm.A); 
207 // For float we check if a == b. If so, we must have outer bias. If not same, b must be > a for it to be valid. 
208 // return ((a == b) && (Bias == tBias::Outer)) || (b > a);

212inline bool tMath::tInterval::Set(const tString& s) 
213{ 
214 Clear(); 
215 
216 // The string should be in the form "[(a,b)]". For convenience, if the string is simply of form "a" it will be 
217 // converted to [a,a]. 
218 tString str(s); 
219 
220 // This handles a single integer by itself and represents it in normal syntax. 
221 if (str.IsNumeric()) 
222 str = tString("[") + str + "," + str + "]"; 
223 
224 tIntervalRep rep = tIntervalRep::Range; 
225 if (str.FindAny(chars: "[]()") != -1) 
226 rep = tIntervalRep::Normal; 
227 
228 // @note We will need '.' here for continuous domain implementation. 
229 if (rep == tIntervalRep::Normal) 
230 str.RemoveAnyNot(theseChars: "[(,0123456789)]-"); 
231 else 
232 str.RemoveAnyNot(theseChars: ",0123456789!-"); 
233 
234 bool inclusiveA = true; 
235 bool inclusiveB = true; 
236 char separator = ','; 
237 if (rep == tIntervalRep::Normal) 
238 { 
239 if ((str[0] != '[') && (str[0] != '(')) 
240 return false; 
241 inclusiveA = (str[0] == '['); 
242 
243 if ((str[str.Length()-1] != ']') && (str[str.Length()-1] != ')')) 
244 return false; 
245 inclusiveB = (str[str.Length()-1] == ']'); 
246 } 
247 else 
248 { 
249 if (str[0] == '!') 
250 inclusiveA = false; 
251 if (str[str.Length()-1] == '!') 
252 inclusiveB = false; 
253 separator = '-'; 
254 } 
255 
256 str.RemoveAny(theseChars: "[()]!"); 
257 if (str.CountChar(c: separator) != 1) 
258 return false; 
259 
260 tString a = str.ExtractLeft(divider: separator); // b is left in str. 
261 A = a.AsInt(base: 10); 
262 B = str.AsInt(base: 10); 
263 
264 if (inclusiveA && inclusiveB) 
265 Bias = tBias::Outer; 
266 else if (!inclusiveA && !inclusiveB) 
267 Bias = tBias::Inner; 
268 else if (inclusiveA) 
269 Bias = tBias::Left; 
270 else 
271 Bias = tBias::Right; 
272 
273 return true; 
274}

302inline tMath::tInterval& tMath::tInterval::MakeInclusive() 
303{ 
304 if (ExclusiveLeft()) 
305 { 
306 A += 1; 
307 Bias = (Bias == tBias::Right) ? tBias::Outer : tBias::Left;

309 
310 if (ExclusiveRight()) 
311 { 
312 B -= 1; 
313 Bias = tBias::Outer;

320inline bool tMath::tInterval::Contains(const tInterval& v) const 
321{ 
322 if (IsEmpty() || v.IsEmpty()) 
323 return false; 
324 
325 #ifdef INTERVAL_FLOAT 
326 // We need to be careful here when the endpoints are equal. It will depend on endpoint exclusivness. 
327 bool leftcon = (A == v.A) ? 
328 ((InclusiveLeft() == v.InclusiveLeft()) ? true : InclusiveLeft()) : 
329 (v.A > A); 
330 
331 bool rghtcon = (B == v.B) ? 
332 ((InclusiveRight() == v.InclusiveRight()) ? true : InclusiveRight()) : 
333 (v.B < B); 
334 #else 
335 tInterval incThis(*this); incThis.MakeInclusive(); 
336 tInterval incTest(v); incTest.MakeInclusive(); 
337 bool leftcon = (incTest.A >= incThis.A); 
338 bool rghtcon = (incTest.B <= incThis.B); 
339 #endif 
340 
341 return (leftcon && rghtcon); 
342}

345inline bool tMath::tInterval::Overlaps(const tInterval& v, bool checkForJoins) const 
346{ 
347 if (IsEmpty() || v.IsEmpty()) 
348 return false; 
349 
350 #ifdef INTERVAL_FLOAT 
351 #else 
352 tInterval incA(*this); incA.MakeInclusive(); 
353 tInterval incB(v); incB.MakeInclusive(); 
354 
355 bool overlap = 
356 tInIntervalII(val: incA.A, min: incB.A, max: incB.B) || 
357 tInIntervalII(val: incA.B, min: incB.A, max: incB.B) || 
358 tInIntervalII(val: incB.A, min: incA.A, max: incA.B) || 
359 tInIntervalII(val: incB.B, min: incA.A, max: incA.B); 
360 
361 if (!overlap && checkForJoins) 
362 { 
363 if ((incA.A == incB.B+1) || (incA.B == incB.A-1)) 
364 overlap = true; 
365 } 
366 return overlap; 
367 #endif 
368}

447inline bool tMath::tIntervalSet::Add(const tInterval& interval) 
448{ 
449 bool modified = false; 
450 
451 // This first step either exits early if the interval we're adding is completely contained in an existing one, 
452 // or it removes existing intervals that are completely inside what we're adding. 
453 tItList<tInterval>::Iter i = Intervals.First(); 
454 while (i.IsValid()) 
455 { 
456 tItList<tInterval>::Iter next = i; 
457 ++next; 
458 
459 if (i->Contains(v: interval)) 
460 return false; 
461 else if (interval.Contains(v: *i)) 
462 Intervals.Remove(iter&: i); 
463 
464 i = next; 
465 } 
466 
467 // Next we find the first and last overlapping intervals. 
468 tItList<tInterval>::Iter frst; 
469 for (auto it = Intervals.First(); it; ++it) 
470 { 
471 // Over integers there is another possibility. The intervals may join even if there is no strict overlap. 
472 if (it->Overlaps(v: interval, checkForJoins: true)) 
473 { 
474 frst = it; 
475 break; 
476 } 
477 } 
478 
479 tItList<tInterval>::Iter last; 
480 for (auto it = Intervals.Last(); it; --it) 
481 { 
482 if (it->Overlaps(v: interval, checkForJoins: true)) 
483 { 
484 last = it; 
485 break; 
486 } 
487 } 
488 
489 // Now we are in 1 of 4 situations: 
490 // 1. We are not overlapping the first or last. (even if first == last or they don't exist). 
491 // 2. We are overlapping the first and last. (works even if first == last). 
492 // 3. We are overlapping the first and not the last. (works even if first == last). 
493 // 4. We are overlapping the last and not the first. (works even if first == last). 
494 
495 // Case 1. Add the interval and we are done. 
496 if (!frst && !last) 
497 { 
498 // We could just append the interval and it would work. However, to keep things organized we can aslo 
499 // decide to insert the new interval at the ordered position ascending from left to right. Since they 
500 // are all disjoint (no overlaps or joins) we only need to check the starting position, A. 
501 tItList<tInterval>::Iter where; 
502 tInterval incInt = interval; incInt.MakeInclusive(); 
503 for (auto i = Intervals.First(); i; ++i) 
504 { 
505 tInterval incI = *i; incI.MakeInclusive(); 
506 if (incInt.A < incI.A) 
507 { 
508 where = i; 
509 break; 
510 } 
511 } 
512 if (where) 
513 Intervals.Insert(obj: new tInterval(interval), here: where); 
514 else 
515 Intervals.Append(obj: new tInterval(interval)); 
516 
517 return true; 
518 } 
519 
520 // Case 2. 
521 else if (frst && last) 
522 { 
523 if (frst != last) 
524 Intervals.Remove(iter&: last); 
525 frst->Encapsulate(v: interval); 
526 } 
527 
528 // Case 3. 
529 else if (frst && !last) 
530 { 
531 frst->Extend(v: interval, allowJoins: true); 
532 } 
533 
534 // Case 3. 
535 else if (last && !frst) 
536 { 
537 last->Extend(v: interval, allowJoins: true); 
538 } 
539 
540 return true; 
541}
