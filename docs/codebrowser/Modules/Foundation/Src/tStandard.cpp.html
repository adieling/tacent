49void* tStd::tMemsrch(void* haystack, int haystackNumBytes, void* needle, int needleNumBytes) 
50{ 
51 if ((haystackNumBytes <= 0) || (needleNumBytes <= 0) || (haystackNumBytes < needleNumBytes)) 
52 return nullptr; 
53 
54 // Serach for the pattern from the first haystack byte (0) to numNeedleBytes from the end. For example, if we are 
55 // seraching for 4 bytes in 8, there will be 5 mem compares of 4 bytes each. 
56 for (int i = 0; i <= haystackNumBytes-needleNumBytes; i++) 
57 { 
58 if (tMemcmp(a: (uint8*)haystack + i, b: needle, numBytes: needleNumBytes) == 0) 
59 return (uint8*)haystack + i; 
60 } 
61 
62 return nullptr; 
63}

66int tStd::tNstrcmp(const char* a, const char* b) 
67{ 
68 const char* origa = a; 
69 const char* origb = b; 
70 
71 bool aStartsDig = a && tIsdigit(c: *a);  
72 bool bStartsDig = b && tIsdigit(c: *b);  
73 
74 // This implementation of tNstrcmp is a modified version of the one written by GitHub user ClangPan. 
75 while (*a && *b) 
76 { 
77 bool aDigit = tIsdigit(c: *a); 
78 bool bDigit = tIsdigit(c: *b); 
79 
80 if (!aDigit && (*a == '-')) 
81 { 
82 ++a; 
83 continue; 
84 } 
85 
86 if (!bDigit && (*b == '-')) 
87 { 
88 ++b; 
89 continue; 
90 } 
91 
92 // We're comparing (possibly multidigit) numbers. 
93 if (aDigit && bDigit) 
94 { 
95 char* enda; 
96 char* endb; 
97 
98 // Get the left number. 
99 int aInt = strtoul(nptr: (char*)a, endptr: &enda, base: 10); 
100 
101 // Get the right number. 
102 int bInt = strtoul(nptr: (char*)b, endptr: &endb, base: 10); 
103 
104 // if the difference is not equal to zero, we have a comparison result 
105 int sign = tMath::tSign(val: aInt - bInt); 
106 if (sign) return sign; 
107 
108 a = enda; 
109 b = endb; 
110 continue; 
111 }  
112 
113 // If only the left char is a digit, we have a result. 
114 if (aDigit) return aStartsDig ? -1 : +1; 
115 
116 // If only the right char is a digit, we have a result. 
117 if (bDigit) return bStartsDig ? +1 : -1; 
118 
119 // compute the difference of both characters 
120 int sign = tMath::tSign(val: tToLower(c: *a) - tToLower(c: *b)); 
121 
122 // If they differ we have a result. 
123 if (sign) return sign; 
124 
125 // Otherwise process the next characters. 
126 ++a; ++b; 
127 } 
128 
129 // If both a and b are at end, we consider letter-case and compare as if we had never done the tToLowers. 
130 if (!(*a) && !(*b)) 
131 return tStrcmp(a: origa, b: origb); 
132 
133 // Now only one of *a or *b are non-zero. 
134 if (*b) return -1; 
135 if (*a) return +1; 
136 
137 return 0; 
138}

159float tStd::tStrtof(const char* s) 
160{ 
161 // Both tStrchr and tStrlen assert on nullptrs so we check here. 
162 if (!s) 
163 return 0.0f; 
164 
165 char* hash = tStrchr(s, c: '#'); 
166 if (hash && (tStrlen(s: hash+1) == 8)) 
167 { 
168 uint32 bin = tStd::tStrtoui32(s: hash+1, base: 16); 
169 return *((float*)(&bin)); 
170 } 
171 
172 return float( tStrtod(s) ); 
173}

176double tStd::tStrtod(const char* s) 
177{ 
178 // tStrlen asserts on nullptrs so we check here. 
179 if (!s) 
180 return 0.0; 
181 
182 int l = tStrlen(s); 
183 if (!l) 
184 return 0.0; 
185 
186 char* hash = tStrchr(s, c: '#'); 
187 if (hash && (tStrlen(s: hash+1) == 16)) 
188 { 
189 uint64 bin = tStrtoui64(s: hash+1, base: 16); 
190 return *((double*)&bin); 
191 } 
192 
193 // This error checking is essential. Sometimes NANs are written in text format to a string. 
194 // Like "nan(snan)". We want these to evaluate to 0.0, not -1 or something else. We allow 
195 // 'e' and 'E' for numbers in exponential form like 3.09E08. 
196 for (int i = 0; i < l; i++) 
197 { 
198 char ch = s[i]; 
199 if 
200 ( 
201 ((ch >= 'a') && (ch <= 'z') && (ch != 'e')) || 
202 ((ch >= 'A') && (ch <= 'Z') && (ch != 'E')) 
203 ) 
204 return 0.0; 
205 } 
206 
207 // Will be 0.0 if there was a problem. 
208 return strtod(nptr: s, endptr: nullptr); 
209}

298int tStd::tUTF8(char8_t* dst, const char16_t* src, int srcLen) 
299{ 
300 // Compute fast worst-case size needed. 
301 // UTF-8 can use up to 3 bytes to encode some codepoints in the BMP (Basic Multilingual Plane). This has 
302 // implications for how much room UTF-8 encoded text could take up from src data that's UTF-16. Eg. 2 char16s could be 
303 // either 2 codepoints in the BMP (6 bytes in UTF-8) or a single codepoint if the second char16 is a surrogate (4 bytes 
304 // in UTF-8). Therefore worst case without inspecting data is 3*numChar16s. 
305 if (!src) 
306 return srcLen * 3; 
307 
308 int total = 0; 
309 while (srcLen > 0) 
310 { 
311 char32_t codepoint; 
312 int read = tUTF::DecodeUtf16(codepoint, src); 
313 srcLen -= read; 
314 src += read; 
315 
316 int written = 0; 
317 if (dst) 
318 { 
319 written = tUTF::EncodeUtf8(dst, codepoint); 
320 dst += written; 
321 } 
322 else 
323 { 
324 // No encoding. Just compute length. 
325 written = tUTF::CalculateUtf8Length(codepoint); 
326 } 
327 total += written; 
328 } 
329 
330 return total; 
331}

364int tStd::tUTF16(char16_t* dst, const char8_t* src, int srcLen) 
365{ 
366 // Compute fast worst-case size needed. 
367 // 1 char8 -> 1 char16. 
368 // 2 char8s (surrogates) -> 1 char16. 
369 // 3 char8s (surrogates) -> also guaranteed 1 char16. 
370 // 4 char8s (surrogates) -> 2 char16s. 
371 // So worst-case is every byte needing 1 whole char16. 
372 if (!src) 
373 return srcLen; 
374 
375 int total = 0; 
376 while (srcLen > 0) 
377 { 
378 char32_t codepoint; 
379 int read = tUTF::DecodeUtf8(codepoint, src); 
380 srcLen -= read; 
381 src += read; 
382 
383 int written = 0; 
384 if (dst) 
385 { 
386 written = tUTF::EncodeUtf16(dst, codepoint); 
387 dst += written; 
388 } 
389 else 
390 { 
391 // No encoding. Just compute length.

679int tUTF::DecodeUtf16(char32_t& codepoint, const char16_t* src) 
680{ 
681 tAssert(src);

688 return 1; 
689 } 
690 
691 // If unmatched low surrogate it's invalid. Return replacement. 
692 if ((high & cSurrogate_Mask16) != cSurrogate_HighVal16) 
693 { 
694 codepoint = tStd::cCodepoint_Replacement; 
695 return 1; 
696 } 
697  
698 char16_t low = src[1]; 
699 
700 // If unmatched high surrogate it's invalid. Return replacement. 
701 if ((low & cSurrogate_Mask16) != cSurrogate_LowVal16) 
702 { 
703 codepoint = tStd::cCodepoint_Replacement; 
704 return 1; 
705 } 
706 
707 // Two correctly matched surrogates if we ade it this far. 
708 // The high bits of the codepoint are the value bits of the high surrogate. 
709 // The low bits of the codepoint are the value bits of the low surrogate. 
710 codepoint = high & cSurrogate_CodepointMask16; 
711 codepoint <<= cSurrogate_CodepointBits; 
712 codepoint |= low & cSurrogate_CodepointMask16; 
713 codepoint += cSurrogate_CodepointOffset;  
714 return 2; 
715}

718int tUTF::EncodeUtf16(char16_t* dst, char32_t codepoint) 
719{ 
720 if (!dst) 
721 return 0; 
722 
723 // If codepoint in the BMP just write the single char16. 
724 if (codepoint <= cCodepoint_LastValidBMP) 
725 { 
726 dst[0] = codepoint; 
727 return 1; 
728 } 
729 
730 codepoint -= cSurrogate_CodepointOffset; 
731 char16_t low = cSurrogate_LowVal16; 
732 low |= codepoint & cSurrogate_CodepointMask32; 
733 
734 codepoint >>= cSurrogate_CodepointBits; 
735 char16_t high = cSurrogate_HighVal16; 
736 high |= codepoint & cSurrogate_CodepointMask32; 
737 
738 dst[0] = high; 
739 dst[1] = low; 
740 return 2;

764int tUTF::DecodeUtf8(char32_t& codepoint, const char8_t* src) 
765{ 
766 tAssert(src); 
767 char8_t leading = src[0]; 
768 int encodingLength = 0; 
769 UTF8Pattern leadingPattern; 
770 
771 bool matches = false; // True if the leading byte matches the current leading pattern. 
772 do 
773 { 
774 encodingLength++; 
775 leadingPattern = UTF8LeadingBytes[encodingLength - 1]; 
776 matches = (leading & leadingPattern.Mask) == leadingPattern.Value; 
777 
778 } while (!matches && (encodingLength < UTF8LeadingBytes_NumElements)); 
779 
780 // If leading byte doesn't match any known pattern it is invalid and we return replacement. 
781 if (!matches) 
782 { 
783 codepoint = tStd::cCodepoint_Replacement; 
784 return encodingLength; 
785 } 
786 
787 codepoint = leading & ~leadingPattern.Mask; 
788 
789 // This loop only ends up running if continuation codeunits found (not ASCII). 
790 for (int i = 1; i < encodingLength; i++) 
791 { 
792 char8_t continuation = src[i]; 
793 
794 // If number of continuation bytes is not the same as advertised on the leading byte it's an invalid encoding 
795 // so we return the replacement. 
796 if ((continuation & cContinuation_UTF8Mask) != cContinuation_UTF8Val) 
797 { 
798 codepoint = tStd::cCodepoint_Replacement; 
799 
800 // I think the best behaviour here is to return how much we processed b4 running into a problem. 
801 // If we returned encodingLength we might skip some input when an invalid is encountered. Hard to say. 
802 return 1+i; 
803 } 
804 
805 codepoint <<= cContinuation_CodepointBits; 
806 codepoint |= continuation & ~cContinuation_UTF8Mask; 
807 } 
808 
809 if 
810 ( 
811 // These are guaranteed to be non-characters by the standard and reuire the replacement. 
812 ((codepoint == tStd::cCodepoint_SpecialNonCharA) || (codepoint == tStd::cCodepoint_SpecialNonCharB)) || 
813 
814 // Surrogates are invalid Unicode codepoints and should only be used in UTF-16. Invalid encoding so return replacement. 
815 ((codepoint <= cCodepoint_LastValidBMP) && ((codepoint & cSurrogate_GenericMask32) == cSurrogate_GenericVal32)) || 
816 
817 // UTF-8 can encode codepoints larger than the Unicode standard allows. If it does it's an invalid encoding and we return the replacement codepoint. 
818 (codepoint > cCodepoint_UnicodeMax) || 
819 
820 // Overlong encodings are considered invalid so we return the replacement codepoint and return the actual number read so we skip the overlong completely. 
821 // We do this last cuz of short-circuit expression evaluation in C++ (calc only called if necessary). 
822 (CalculateUtf8Length(codepoint) != encodingLength) 
823 ) 
824 { 
825 codepoint = tStd::cCodepoint_Replacement; 
826 } 
827 
828 return encodingLength; 
829}

832int tUTF::EncodeUtf8(char8_t* dst, char32_t codepoint) 
833{ 
834 if (!dst) 
835 return 0; 
836 
837 // Write the continuation bytes in reverse order. 
838 int encodeLength = CalculateUtf8Length(codepoint); 
839 for (int contIndex = encodeLength - 1; contIndex > 0; contIndex--) 
840 { 
841 char8_t cont = codepoint & ~cContinuation_UTF8Mask; 
842 cont |= cContinuation_UTF8Val; 
843 dst[contIndex] = cont; 
844 codepoint >>= cContinuation_CodepointBits; 
845 } 
846 
847 // Write the leading byte. 
848 UTF8Pattern pattern = UTF8LeadingBytes[encodeLength - 1]; 
849 char8_t lead = codepoint & ~(pattern.Mask); 
850 lead |= pattern.Value; 
851 dst[0] = lead; 
852 
853 return encodeLength; 
854}
