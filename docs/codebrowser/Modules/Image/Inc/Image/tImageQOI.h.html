3// This class knows how to load and save Quite OK Images (.qoi) files into tPixel arrays. These tPixels may be 'stolen' 
4// by the tPicture's constructor if a qoi file is specified. After the array is stolen the tImageQOI is invalid. This 
5// is purely for performance.

26class tImageQOI : public tBaseImage

33 // The data is copied out of qoiFileInMemory. Go ahead and delete[] after if you want. 
34 tImageQOI(const uint8* qoiFileInMemory, int numBytes) { Load(qoiFileInMemory, numBytes); }

36 // This one sets from a supplied pixel array. If steal is true it takes ownership of the pixels pointer. Otherwise 
37 // it just copies the data out. Sets the colour space to sRGB. Call SetColourSpace after if you wanted linear. 
38 tImageQOI(tPixel4b* pixels, int width, int height, bool steal = false) { Set(pixels, width, height, steal); }

40 // Sets from a single frame. 
41 tImageQOI(tFrame* frame, bool steal = true) { Set(frame, steal); } 
42 
43 // Constructs from a tPicture. 
44 tImageQOI(tPicture& picture, bool steal = true) { Set(picture, steal); }

46 virtual ~tImageQOI() { Clear(); }

48 // Clears the current tImageQOI before loading. Returns success. If false returned, object is invalid. 
49 bool Load(const tString& qoiFile); 
50 bool Load(const uint8* qoiFileInMemory, int numBytes);

52 // This one sets from a supplied pixel array. If steal is true it takes ownership of the pixels pointer. Otherwise 
53 // it just copies the data out. After this call the objects ColourSpace is set to sRGB. If the data was all linear 
54 // you can call SetColourSpace() manually afterwards. 
55 bool Set(tPixel4b* pixels, int width, int height, bool steal = false) override;

63 enum class tFormat 
64 { 
65 Invalid, // Invalid must be 0. 
66 BPP24, // 24 bit colour. 
67 BPP32, // 24 bit colour with 8 bits opacity in the alpha channel. 
68 Auto // Save function will decide format. BPP24 if all image pixels are opaque and BPP32 otherwise. 
69 }; 
70 
71 struct SaveParams 
72 { 
73 SaveParams() { Reset(); } 
74 SaveParams(const SaveParams& src) : Format(src.Format), ColourProfile(src.ColourProfile) { } 
75 void Reset() { Format = tFormat::Auto; ColourProfile = tColourProfile::Auto; } 
76 SaveParams& operator=(const SaveParams& src) { Format = src.Format; ColourProfile = src.ColourProfile; return *this; } 
77 tFormat Format; 
78 tColourProfile ColourProfile; // QOI supports lRGB and sRGB. 
79 }; 
80 
81 // Saves the tImageQOI to the file specified. The type of filename must be "qoi". If tFormat is Auto, this 
82 // function will decide the format. BPP24 if all image pixels are opaque and BPP32 otherwise. Returns the format 
83 // that the file was saved in, or tFormat::Invalid if there was a problem. Since Invalid is 0, you can use an 'if'. 
84 // The colour-space is also saved with the file. If space is set to auto it uses whatever the current space is in 
85 // this object. If not set to auto, it overrides the space for the saved file. Setting it to invalid uses sRGB. 
86 tFormat Save(const tString& qoiFile, tFormat, tColourProfile = tColourProfile::Auto) const; 
87 tFormat Save(const tString& qoiFile, const SaveParams& = SaveParams()) const; 
88 
89 // After this call no memory will be consumed by the object and it will be invalid. 
90 void Clear() override; 
91 bool IsValid() const override { return Pixels ? true : false; } 
92 
93 int GetWidth() const { return Width; } 
94 int GetHeight() const { return Height; } 
95 
96 // All pixels must be opaque (alpha = 255) for this to return true. 
97 bool IsOpaque() const; 
98 
99 // After this call you are the owner of the pixels and must eventually delete[] them. This tImageQOI object is 
100 // invalid afterwards. 
101 tPixel4b* StealPixels(); 
102 tFrame* GetFrame(bool steal = true) override; 
103 tPixel4b* GetPixels() const { return Pixels; } 
104 
105private: 
106 int Width = 0; 
107 int Height = 0; 
108 tPixel4b* Pixels = nullptr; 
109};
