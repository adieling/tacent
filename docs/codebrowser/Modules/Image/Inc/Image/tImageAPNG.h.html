66 enum class tFormat 
67 { 
68 Invalid, // Invalid must be 0. 
69 BPP24, // RGB. 24 bit colour. 
70 BPP32, // RGBA. 24 bit colour and 8 bits opacity in the alpha channel. 
71 Auto // Save function will decide format. BPP24 if all image pixels are opaque and BPP32 otherwise. 
72 };

85 // Saves the tImageAPNG to the APNG file specified. The type of filename must be PNG or APNG. PNG is allowed because 
86 // the way apng files are specified they can have the png extension and still be read by non-apng-aware loaders. 
87 // If tFormat is Auto, this function will decide the format. BPP24 if all image pixels are opaque and BPP32 
88 // otherwise. Returns the format that the file was saved in, or tFormat::Invalid if there was a problem. Since 
89 // Invalid is 0, you can use an 'if'. OverrideframeDuration is in milliseconds. Set to >= 0 to override all frames. 
90 tFormat Save(const tString& apngFile, tFormat, int overrideframeDuration = -1) const; 
91 tFormat Save(const tString& apngFile, const SaveParams& = SaveParams()) const;

115 // Since some apng files may have a .png extension, it is hand to quickly be able to tell if a particular .png 
116 // file is an apng. Probably no one will ever read this comment, but the Mozilla apng people should probably not 
117 // have insisted that apngs be encoded in pngs. In any case, this slightly crappy code cannot guarantee that a 
118 // return value of true means it is an apng (although such a false positive is extremely unlikely). Even in these 
119 // cases, it just means the APNG reading code will be used -- it will still successfully extract the single frame. 
120 // 
121 // The preference is, however, that non-apng files be loaded by tImagePNG. It is faster and reads the src format 
122 // better than APngDis, which could be further modified but is unfamiliar code. 
123 static bool IsAnimatedPNG(const tString& pngFile);

132inline bool tImage::tImageAPNG::IsOpaque() const 
133{ 
134 for (tFrame* frame = Frames.Head(); frame; frame = frame->Next()) 
135 if (!frame->IsOpaque()) 
136 return false; 
137 
138 return true; 
139}

142inline tFrame* tImage::tImageAPNG::StealFrame(int frameNum) 
143{ 
144 tFrame* f = GetFrame(frameNum); 
145 if (!f) 
146 return nullptr; 
147 
148 return Frames.Remove(item: f); 
149}

152inline void tImage::tImageAPNG::StealFrames(tList<tFrame>& frames) 
153{ 
154 while (tFrame* frame = Frames.Remove()) 
155 frames.Append(item: frame); 
156}

159inline tFrame* tImage::tImageAPNG::GetFrame(int frameNum) 
160{ 
161 if ((frameNum >= Frames.GetNumItems()) || (frameNum < 0)) 
162 return nullptr; 
163 
164 tFrame* f = Frames.First(); 
165 while (frameNum--) 
166 f = f->Next(); 
167 
168 return f; 
169}
