16#pragma once 
17#include <Foundation/tString.h> 
18#include <Foundation/tList.h> 
19#include <System/tThrow.h> 
20namespace tPipeline 
21{ 
22 
23 
24class tRule : public tLink<tRule> 
25{ 
26public: 
27 tRule() : Target(), Dependencies(), Clean(false), Config(tConfig::Default) { } 
28 virtual ~tRule() { Dependencies.Empty(); } 
29 virtual void Build() { } // You may override this to build the rule. 
30 virtual char* GetName() const { return nullptr; } 
31 
32 // If clean is set to true, the OutOfDate() function will always return true. 
33 void SetClean(bool clean = true) { Clean = clean; } 
34 
35 enum class tConfig 
36 { 
37 Default, 
38 Debug, 
39 Develop, 
40 Release, 
41 Ship 
42 // Whatever else is needed can just be added. 
43 }; 
44 
45 // The build function can optionally get access to the configuration variable set by this function. 
46 void SetConfig(tConfig config = tConfig::Default) { Config = config; } 
47 
48 // Clears the dependencies and sets the target. 
49 void SetTarget(const tString& fullTargetName); 
50 
51 // Adds a dependency. If the dependency doesn't exist a tRuleError object is thrown and the dependency is not 
52 // added. If the dependency was already added, this function does nothing. 
53 void AddDependency(const tString& fullDepName); 
54 
55 // Adds a dependency. The string pointer is dereferenced and copied. If the dependency doesn't exist a tRuleError 
56 // object is thrown and the dependency is not added. If the dependency was already added, this function does 
57 // nothing. 
58 void AddDependency(tStringItem* fullDepName); 
59 
60 // Adds multiple dependencies. The list is left empty and the strings are managed by the tRule. If any dependency 
61 // doesn't exist a tRuleError object is thrown and that dependency is not added. All dependencies that do exist 
62 // will be added if they aren't already added. 
63 void AddDependencies(tList<tStringItem>& deps); 
64 
65 // Adds multiple dependencies. Does not act recursively on the directory. Extension can be something like "txt". 
66 // Throws a tRuleError if there are any problems. All dependencies that can be added will be. Checks for duplicate 
67 // deps. Ignores hidden files and folders and will not add them or recurse into them. 
68 void AddDependencyDir(const tString& dir, const tString& extension = tString()); 
69 
70 // Adds multiple dependencies. Acts recursively on the directory. Extension can be something like "txt". Throws a 
71 // tRuleError if there are any problems. All dependencies that can be added will be. Checks for duplicate deps. 
72 // Ignores hidden files and folders and will not add them or recurse into them. 
73 void AddDependencyDirRec(const tString& dir, const tString& extension = tString()); 
74 
75 // Returns true if target has been specified and target doesn't exist or is older than any dependency or if a clean 
76 // build is requested (the latter only if checkCleanFlag == true). Returns false if there is no need to build. 
77 // Throws a tRuleError if any dependency doesn't exist. 
78 bool OutOfDate(bool checkCleanFlag = true); 
79 
80 // Here are some aliases so you don't have to type as much. 
81 void AddDep(const tString& fullDepName) { AddDependency(fullDepName); } 
82 void AddDep(tStringItem* fullDepName) { AddDependency(fullDepName); } 
83 void AddDeps(tList<tStringItem>& deps) { AddDependencies(deps); } 
84 void AddDepDir(const tString& dir, const tString& filter = "*.*") { AddDependencyDir(dir, extension: filter); } 
85 void AddDepDirRec(const tString& dir, const tString& filter = "*.*") { AddDependencyDirRec(dir, extension: filter); } 
86 
87protected: 
88 tString Target; 
89 tList<tStringItem> Dependencies; 
90 bool Clean; 
91 tConfig Config; 
92 
93private: 
94 bool MaybeAddToDependenciesCaseInsensitive(const tString&); 
95}; 
96 
97 
98}

101struct tRuleError : public tError 
102{ 
103 tRuleError(const char* format, ...); 
104 tRuleError(const tString& m) : tError("[tRule] ") { Message += m; } 
105 tRuleError() : tError("[tRule] ") { } 
106};
