134// For these conversion calls, unknown digit characters for the supplied base are ignored. If base is not E [2, 36], the 
135// base in which to interpret the string is determined by passing a prefix in the string. Base 10 is used if no specific 
136// prefix is found. Although these functions take in UTF-8 strings (chat8_t*), a well-formed source string will only 
137// include ASCII characters like digits, negative signs, prefix letters etc. Again, unknown characters are ignored, or, 
138// in the case of the 'strict' variants, cause the return value to be false. 
139// Base prefixes in use: 
140// 
141// Base 16 prefixes: x X 0x 0X # 
142// Base 10 prefixes: d D 0d 0D 
143// Base 8 prefixes: o O 0o 0O @ 
144// Base 4 prefixes: n N 0n 0N (N for Nybble) 
145// Base 2 prefixes: b B 0b 0B ! 
146// 
147// Negative/positive symbol may only be used with base 10 strings: eg. "d-769" or just "-769". The behaviour follows 
148// this recipe: If the base is invalid (not between 2 and 36) the first and maybe second characters are examined to 
149// determine base. Next, invalid characters are removed (implementation may just ignore them). Invalid characters 
150// include + and - for non base 10. Finally the conversion is performed. Valid digits for base 36 are 0-9, a-z, and A-Z. 
151// In all string-to-int functions, 0 is returned if there is no conversion. This can happen if all characters are 
152// invalid, the passed in string is null, or the passed in string is empty. 
153// 
154// If base is E (1, 36] AND a prefix is supplied, the supplied base is used. Using the prefix instead would be 
155// ambiguous. For example, "0x" is a valid base 36 number but "0x" is also a prefix. If you supply a prefix it must 
156// match the base or you will get undefined results. eg "0xff" or "xff" with base=36 interprets the 'x', correctly, as 
157// 33. "0xff" with base=16 works too because 'x' is an invalid hex digit and gets ignored. The '0' is leading so also 
158// does not interfere. The same behaviour holds for all prefixes and bases. Finally, do not use base=-1 on strings 
159// that are not either base 16, 10, 8, 4, or 2. eg. "0XF" is a valid base-34 number, and we wouldn't want it 
160// interpreted as base 16 is that's not what you intended. 
161template <typename IntegralType> IntegralType tStrtoiT(const char*, int base = -1);

261// These functions convert to/from the 3 main unicode encodings. Note that most text in Tacent is 
262// assumed to be UTF-8. These are provided so external or OS-specific calls can be made when they expect non-UTF-8 
263// input, and when results are supplied, converted back to UTF-8. 
264// 
265// Null termination is not part of UTF encoding. These work on arrays of code-units (charN types). Not null terminated. 
266// 1) If (only) dst is nullptr, returns the exact number of code-units (charNs) needed for dst. 
267// 2) If src is nullptr, dst is ignored and srcLen is used to return a fast, worst-case number of codeunits (charNs) 
268// needed for dst assuming no overlong encoding. This second method is fast because the contents of src are not 
269// inspected, but it often gives conservative (larger) results.

491template<typename IntegralType> inline IntegralType tStd::tStrtoiT(const char8_t* str, int base) 
492{ 
493 if (!str || (*str == '\0')) 
494 return IntegralType(0); 
495 
496 int len = tStrlen(s: str); 
497 const char8_t* start = str; 
498 const char8_t* end = str + len - 1; 
499 
500 if ((base < 2) || (base > 36)) 
501 base = -1; 
502 
503 if (base == -1) 
504 { 
505 // Base is -1. Need to determine based on prefix. 
506 if ((len > 1) && (*start == '0')) 
507 start++; 
508 
509 if ((*start == 'x') || (*start == 'X') || (*start == '#')) 
510 base = 16; 
511 else if ((*start == 'd') || (*start == 'D')) 
512 base = 10; 
513 else if ((*start == 'o') || (*start == 'O') || (*start == '@')) 
514 base = 8; 
515 else if ((*start == 'n') || (*start == 'N')) 
516 base = 4; 
517 else if ((*start == 'b') || (*start == 'B') || (*start == '!')) 
518 base = 2; 
519 
520 if (base == -1) 
521 base = 10; 
522 else 
523 start++; 
524 } 
525 
526 IntegralType val = 0; 
527 IntegralType colVal = 1; 
528 for (const char8_t* curr = end; curr >= start; curr--) 
529 { 
530 if ((*curr == '-') && (base == 10)) 
531 { 
532 val = -val; 
533 continue; 
534 } 
535 
536 int digVal = -1; 
537 if (*curr >= '0' && *curr <= '9') 
538 digVal = *curr - '0'; 
539 else if (*curr >= 'a' && *curr <= 'z') 
540 digVal = 10 + (*curr - 'a'); 
541 else if (*curr >= 'A' && *curr <= 'Z') 
542 digVal = 10 + (*curr - 'A'); 
543 
544 if ((digVal == -1) || (digVal > base-1)) 
545 continue; 
546 
547 val += IntegralType(digVal)*colVal; 
548 colVal *= IntegralType(base); 
549 } 
550 
551 return val; 
552}
