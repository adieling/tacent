20void tObject::Save(tChunkWriter& chunk) const 
21{ 
22 chunk.Begin(chunkID: tChunkID::Scene_Object); 
23 { 
24 chunk.Begin(chunkID: tChunkID::Core_ID); 
25 chunk.Write(obj: ID); 
26 chunk.End(); 
27 
28 chunk.Begin(chunkID: tChunkID::Core_Name); 
29 chunk.Write(s: Name); 
30 chunk.End(); 
31 
32 Attributes.Save(chunk); 
33 } 
34 chunk.End(); 
35}

38void tObject::Load(const tChunk& objChunk) 
39{ 
40 Clear(); 
41 tAssert(objChunk.ID() == tChunkID::Scene_Object); 
42 
43 for (tChunk chunk = objChunk.First(); chunk.Valid(); chunk = chunk.Next()) 
44 { 
45 switch (chunk.ID()) 
46 { 
47 case tChunkID::Core_ID: 
48 chunk.GetItem(item&: ID); 
49 break; 
50 
51 case tChunkID::Core_Name: 
52 chunk.GetItem(item&: Name); 
53 break; 
54 
55 case tChunkID::Scene_AttributeList: 
56 Attributes.Load(chunk); 
57 break; 
58 } 
59 } 
60}
