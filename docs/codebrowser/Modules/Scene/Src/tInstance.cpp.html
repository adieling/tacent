22void tInstance::Save(tChunkWriter& chunk) const 
23{ 
24 chunk.Begin(chunkID: tChunkID::Scene_Instance); 
25 { 
26 tObject::Save(chunk); 
27 
28 chunk.Begin(chunkID: tChunkID::Scene_InstanceParameters); 
29 { 
30 chunk.Write(obj: ObjectType); 
31 chunk.Write(obj: ObjectID); 
32 chunk.Write(Transform); 
33 } 
34 chunk.End(); 
35 } 
36 chunk.End(); 
37}

40void tInstance::Load(const tChunk& instChunk) 
41{ 
42 tAssert(instChunk.ID() == tChunkID::Scene_Instance); 
43 Clear(); 
44 
45 for (tChunk chunk = instChunk.First(); chunk.Valid(); chunk = chunk.Next()) 
46 { 
47 switch (chunk.ID()) 
48 { 
49 case tChunkID::Scene_Object: 
50 tObject::Load(chunk); 
51 break; 
52 
53 case tChunkID::Scene_InstanceParameters: 
54 { 
55 chunk.GetItem(item&: ObjectType); 
56 chunk.GetItem(item&: ObjectID); 
57 chunk.GetItem(item&: Transform); 
58 break; 
59 } 
60 } 
61 } 
62}
