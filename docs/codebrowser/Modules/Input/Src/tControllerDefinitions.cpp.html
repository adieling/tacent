32const tInput::tContDefn* tInput::tLookupContDefn(const tVidPid& vidpid) 
33{ 
34 if (!ContDictPopulated) 
35 { 
36 tPopulateContDict(); 
37 ContDictPopulated = true; 
38 } 
39 
40 return ContDict.GetValue(key: vidpid); 
41}
