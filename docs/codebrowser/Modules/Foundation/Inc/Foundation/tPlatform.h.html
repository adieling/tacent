204template<typename T> inline T tGetSwapEndian(const T& val) 
205{ 
206 union endianAccessor 
207 { 
208 T Data; 
209 uint8 Bytes[sizeof(T)]; 
210 }; 
211 endianAccessor& src = (endianAccessor&)val; 
212 endianAccessor dst; 
213 for (int d = 0; d < int(sizeof(T)); d++) 
214 dst.Bytes[d] = src.Bytes[sizeof(T) - d - 1]; 
215 
216 return dst.Data; 
217} 
218 
219 
220template<typename T> inline void tSwapEndian(T& val) 
221{ 
222 union endianAccessor 
223 { 
224 T Data; 
225 uint8 Bytes[sizeof(T)]; 
226 }; 
227 endianAccessor& srcDst = (endianAccessor&)val; 
228 int numBytes = sizeof(T); 
229 for (int s = 0; s < (numBytes >> 1); s++) 
230 { 
231 uint8& a = srcDst.Bytes[s]; 
232 uint8& b = srcDst.Bytes[numBytes - 1 - s]; 
233 uint8 t = a; 
234 a = b; 
235 b = t; 
236 } 
237}

240template<typename T> inline void tSwapEndian(T* a, int numItems) 
241{ 
242 for (int i = 0; i < numItems; i++) 
243 tSwapEndian(a[i]); 
244}

247inline constexpr uint16 tSwapEndian16(uint16 val) 
248{ 
249 uint16 u0 = val & 0x00FF; 
250 uint16 u1 = (val >> 8) & 0x00FF; 
251 return (uint16(u1) | (uint16(u0) << 8)); 
252} 
253 
254 
255inline constexpr uint32 tSwapEndian32(uint32 val) 
256{ 
257 uint32 u0 = val & 0x000000FF; 
258 uint32 u1 = (val >> 8) & 0x000000FF; 
259 uint32 u2 = (val >> 16) & 0x000000FF; 
260 uint32 u3 = (val >> 24) & 0x000000FF; 
261 return (uint32(u3) | (uint32(u2) << 8) | (uint32(u1) << 16) | (uint32(u0) << 24)); 
262}

265inline tEndianness tGetEndianness() 
266{ 
267 union 
268 { 
269 uint8 c[2]; 
270 uint16 u; 
271 } a; 
272 
273 a.c[1] = 0; 
274 a.c[0] = 1; 
275 
276 if (a.u == 1) 
277 return tEndianness::Little; 
278 
279 return tEndianness::Big; 
280}
