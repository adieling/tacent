3// This class knows how to load and save PNG files. It does zero processing of image data. It knows the details of the 
4// png file format and loads the data into a tPixel array. These tPixels may be 'stolen' by the tPicture's constructor 
5// if a png file is specified. After the array is stolen the tImagePNG is invalid. This is purely for performance.

29 enum LoadFlags 
30 { 
31 // Gamma-correct. Gamma compression using an encoding gamma of 1/2.2. Assumes (colour) data is linear and puts 
32 // it in gamma-space (brighter) for diaplay on a monitor. Png files at 16 bpc are in linear space. Png files at 
33 // 8 bpc abd sRGB. 
34 LoadFlag_GammaCompression = 1 << 0, 
35 LoadFlag_SRGBCompression = 1 << 1, // Same as above but uses the official sRGB transformation. Linear -> sRGB. Approx encoding gamma of 1/2.4 for part of curve. 
36 LoadFlag_AutoGamma = 1 << 2, // Applies sRGB compression for 16 bpc png images. Call GetColourSpace to see final colour profile. 
37 
38 // If a png is 16 bpc you can force it to load into an 8 bpc buffer with this flag. 
39 LoadFlag_ForceToBpc8 = 1 << 3, 
40 
41 // Crazily some PNG files are actually JPG/JFIF files inside. I don't much like supporting this, but some 
42 // software (ms-paint for example), will happily load such an invalid png. The world would be better if app 
43 // developers wouldn't save things with the wrong extension, but they get away with it because other software 
44 // loads this junk... and now this library is yet another. 
45 LoadFlag_AllowJPG = 1 << 4, 
46 LoadFlags_Default = LoadFlag_ForceToBpc8 | LoadFlag_AllowJPG 
47 };

54 LoadParams& operator=(const LoadParams& src) { Flags = src.Flags; Gamma = src.Gamma; return *this; } 
55 uint32 Flags; 
56 float Gamma; 
57 };

98 enum class tFormat 
99 { 
100 Invalid, // Invalid must be 0. 
101 BPP24_RGB_BPC8, // 24-bit RGB. 3 8-bit components. 
102 BPP32_RGBA_BPC8, // 32-bit RGBA. 4 8-bit components. 
103 BPP48_RGB_BPC16, // 48-bit RGB. 3 16-bit components. 
104 BPP64_RGBA_BPC16, // 64-bit RGBA. 4 16-bit components. 
105 
106 // Save function will decide format. If image is 8-bpc, RGB_24BIT_8BPC if all image pixels are opaque and 
107 // RGBA_32BIT_8BPC otherwise. If image is 16-bpc, RGB_48BIT_16BPC for opaque, RGB_64BIT_16BPC otherwise. 
108 Auto 
109 };

111 struct SaveParams

113 SaveParams() { Reset(); } 
114 SaveParams(const SaveParams& src) : Format(src.Format) { } 
115 void Reset() { Format = tFormat::Auto; } 
116 SaveParams& operator=(const SaveParams& src) { Format = src.Format; return *this; } 
117 tFormat Format;

120 // Saves the tImagePNG to the PNG file specified. The type of filename must be PNG. If tFormat is Auto, this 
121 // function will decide the format. If the internal buffer is 8-bpc it will choose between BPP24 and BPP32 depending 
122 // on opacity (BPP24 is all pixels are opaque). In the internal buffer is 16-bpc it chooses between BPP48 and BPP64. 
123 // When tFormat is explicitly one of the BPPNN choices, it may need to convert the data. For example, if the 
124 // internal buffer is 8-bpc and you choose to save BPP48 or BPP64 (both 16-bpc formats), a conversion must take 
125 // place. Returns the format that the file was saved in, or tFormat::Invalid if there was a problem. Since Invalid 
126 // is 0, you can use an 'if' to check success. 
127 tFormat Save(const tString& pngFile, tFormat) const; 
128 tFormat Save(const tString& pngFile, const SaveParams& = SaveParams()) const;
