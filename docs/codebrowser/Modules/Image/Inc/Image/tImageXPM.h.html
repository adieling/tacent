1// tImageXPM.h

3// This class knows how to load and save an X-Windows Pix Map (.xpm) file. It knows the details of the xpm file format 
4// and loads the data into a tPixel array. These tPixels may be 'stolen' by the tPicture's constructor if a xpm file is 
5// specified. After the array is stolen the tImageXPM is invalid. This is purely for performance.

17#pragma once 
18#include <Foundation/tString.h> 
19#include <Math/tColour.h> 
20#include <Image/tPixelFormat.h> 
21#include <Image/tBaseImage.h> 
22namespace tImage 
23{ 
24 
25 
26class tImageXPM : public tBaseImage 
27{ 
28public: 
29 // Creates an invalid tImageXPM. You must call Load manually. 
30 tImageXPM() { } 
31 
32 tImageXPM(const tString& xpmFile) { Load(xpmFile); } 
33 
34 // The data is copied out of xpmFileInMemory. Go ahead and delete after if you want. 
35 tImageXPM(const uint8* xpmFileInMemory, int numBytes) { Load(xpmFileInMemory, numBytes); } 
36 
37 // This one sets from a supplied pixel array. If steal is true it takes ownership of the pixels pointer. Otherwise 
38 // it just copies the data out. 
39 tImageXPM(tPixel4b* pixels, int width, int height, bool steal = false) { Set(pixels, width, height, steal); } 
40 
41 // Sets from a single frame. 
42 tImageXPM(tFrame* frame, bool steal = true) { Set(frame, steal); } 
43 
44 // Constructs from a tPicture. 
45 tImageXPM(tPicture& picture, bool steal = true) { Set(picture, steal); } 
46 
47 virtual ~tImageXPM() { Clear(); } 
48 
49 // Clears the current tImageXPM before loading. Returns success. If false returned, object is invalid. 
50 bool Load(const tString& xpmFile); 
51 bool Load(const uint8* xpmFileInMemory, int numBytes); 
52 
53 // This one sets from a supplied pixel array. If steal is true it takes ownership of the pixels pointer. Otherwise 
54 // it just copies the data out. 
55 bool Set(tPixel4b*, int width, int height, bool steal = false) override; 
56 
57 // Sets from a single frame. 
58 bool Set(tFrame*, bool steal = true) override; 
59 
60 // Sets from a tPicture. 
61 bool Set(tPicture& picture, bool steal = true) override; 
62 
63 // After this call no memory will be consumed by the object and it will be invalid. 
64 void Clear() override; 
65 bool IsValid() const override { return Pixels ? true : false; } 
66 
67 int GetWidth() const { return Width; } 
68 int GetHeight() const { return Height; } 
69 
70 bool IsOpaque() const { return true; } 
71 
72 // After this call you are the owner of the pixels and must eventually delete[] them. This tImageXPM object is 
73 // invalid afterwards. 
74 tPixel4b* StealPixels(); 
75 tFrame* GetFrame(bool steal = true) override; 
76 tPixel4b* GetPixels() const { return Pixels; } 
77 
78private: 
79 int Width = 0; 
80 int Height = 0; 
81 tPixel4b* Pixels = nullptr; 
82};

88inline void tImageXPM::Clear()

90 Width = 0; 
91 Height = 0; 
92 delete[] Pixels;
