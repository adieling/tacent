3// This knows how to load and save a Radiance High Dynamic Range image (.hdr or .rgbe). It knows the details of the hdr 
4// file format and loads the data into a tPixel array. These tPixels may be 'stolen' by the tPicture's constructor if 
5// an HDR file is specified. After the array is stolen the tImageHDR is invalid. This is purely for performance.

40#pragma once 
41#include <Foundation/tString.h> 
42#include <Math/tColour.h> 
43#include <Image/tPixelFormat.h> 
44#include <Image/tBaseImage.h> 
45namespace tImage 
46{ 
47 
48 
49class tImageHDR : public tBaseImage 
50{ 
51public: 
52 struct LoadParams 
53 { 
54 LoadParams() { Reset(); } 
55 void Reset(); 
56 float Gamma; 
57 int Exposure; 
58 }; 
59 
60 // Creates an invalid tImageHDR. You must call Load manually. 
61 tImageHDR() { } 
62 tImageHDR(const tString& hdrFile, const LoadParams& loadParams = LoadParams()) { Load(hdrFile, loadParams); } 
63 
64 // hdrFileInMemory can be deleted after this runs. 
65 tImageHDR(uint8* hdrFileInMemory, int numBytes, const LoadParams& loadParams = LoadParams()) { Load(hdrFileInMemory, numBytes, loadParams); } 
66 
67 // This one sets from a supplied pixel array. It just reads the data (or steals the array if steal set). 
68 tImageHDR(tPixel4b* pixels, int width, int height, bool steal = false) { Set(pixels, width, height, steal); } 
69 
70 // Sets from a single frame. 
71 tImageHDR(tFrame* frame, bool steal = true) { Set(frame, steal); } 
72 
73 // Constructs from a tPicture. 
74 tImageHDR(tPicture& picture, bool steal = true) { Set(picture, steal); } 
75 
76 virtual ~tImageHDR() { Clear(); } 
77 
78 // Clears the current tImageHDR before loading. If false returned object is invalid. 
79 bool Load(const tString& hdrFile, const LoadParams& = LoadParams()); 
80 bool Load(uint8* hdrFileInMemory, int numBytes, const LoadParams& = LoadParams()); 
81 
82 // This one sets from a supplied pixel array. 
83 bool Set(tPixel4b* pixels, int width, int height, bool steal = false) override; 
84 
85 // Sets from a single frame. 
86 bool Set(tFrame*, bool steal = true) override; 
87 
88 // Sets from a tPicture. 
89 bool Set(tPicture& picture, bool steal = true) override; 
90 
91 // After this call no memory will be consumed by the object and it will be invalid. 
92 void Clear() override; 
93 bool IsValid() const override { return Pixels ? true : false; } 
94 
95 int GetWidth() const { return Width; } 
96 int GetHeight() const { return Height; } 
97 
98 // After this call you are the owner of the pixels and must eventually delete[] them. This tImageHDR object is 
99 // invalid afterwards. 
100 tPixel4b* StealPixels(); 
101 tFrame* GetFrame(bool steal = true) override; 
102 tPixel4b* GetPixels() const { return Pixels; } 
103 
104private: 
105 bool LegacyReadRadianceColours(tPixel4b* scanline, int length); // Older hdr files use this scanline format. 
106 bool ReadRadianceColours(tPixel4b* scanline, int length); // Most hdr files use the new scanline format. This will call the old as necessary. 
107 bool ConvertRadianceToGammaCorrected(tPixel4b* scan, int len); 
108 static void AdjustExposure(tPixel4b* scan, int len, int adjust); 
109 
110 void PutB(int v) { *WriteP++ = uint8(v); } 
111 uint8 GetB() { return *ReadP++; } 
112 void UngetB(int v) { *(--ReadP) = v; } 
113 
114 int Width = 0; 
115 int Height = 0; 
116 tPixel4b* Pixels = nullptr; 
117 
118 // Read and write pointers used during processing. 
119 uint8* ReadP = nullptr; 
120 uint8* WriteP = nullptr; 
121 
122 // While it would be more efficient to share these tables between instances, we need thread safety. 
123 void SetupGammaTables(float gamma); 
124 void CleanupGammaTables(); 
125 
126 uint8* MantissaTable = nullptr; 
127 uint8* ExponentTable = nullptr; 
128 uint8 (*GammaTable)[256] = nullptr; 
129 
130 // Constants. 
131 const static int MaxGammaShift = 31; 
132 const static int MinScanLen = 8; // Minimum scanline length for encoding. 
133 const static int MaxScanLen = 0x7FFF; // Maximum scanline length for encoding. 
134 const static int MinRunLen = 4; 
135 const static int ExpXS = 128; // Excess used for exponent. 
136};

151 Width = 0;
