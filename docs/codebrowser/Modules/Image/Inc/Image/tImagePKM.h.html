1// tImagePKM.h

3// This class knows how to load and save Ericsson's ETC1/ETC2/EAC PKM (.pkm) files. The pixel data is stored in a 
4// tLayer. If decode was requested the layer will store raw pixel data. The layer may be 'stolen'. IF it is the 
5// tImagePKM is invalid afterwards. This is purely for performance.

27class tImagePKM : public tBaseImage 
28{ 
29public: 
30 enum LoadFlag 
31 { 
32 LoadFlag_Decode = 1 << 0, // Decode the pkm texture data into RGBA 32 bit. If not set, the pixel data will remain unmodified. 
33 
34 // The remaining flags only apply when decode flag set. ReverseRowOrder is guaranteed to work if decoding, and 
35 // guaranteed to not work if not decoding. 
36 LoadFlag_ReverseRowOrder = 1 << 1, // OpenGL uses the lower left as the orig DirectX uses the upper left. Set flag for OpenGL. 
37 LoadFlag_GammaCompression = 1 << 2, // Gamma-correct. Gamma compression using an encoding gamma of 1/2.2. 
38 LoadFlag_SRGBCompression = 1 << 3, // Same as above but uses the official sRGB transformation. Linear -> sRGB. Approx encoding gamma of 1/2.4 for part of curve. 
39 LoadFlag_AutoGamma = 1 << 4, // Tries to determine whether to apply sRGB compression based on pixel format. Call GetColourSpace to see if it applied. 
40 LoadFlag_SpreadLuminance = 1 << 5, // For PKM files with a single Red, spread it to all the RGB channels (otherwise red only). Applies only if decoding a pkm that is an R-only format. 
41 LoadFlags_Default = LoadFlag_Decode | LoadFlag_ReverseRowOrder 
42 }; 
43 
44 struct LoadParams 
45 { 
46 LoadParams() { Reset(); } 
47 LoadParams(const LoadParams& src) : Flags(src.Flags), Gamma(src.Gamma) { } 
48 
49 void Reset() { Flags = LoadFlags_Default; Gamma = tMath::DefaultGamma; } 
50 LoadParams& operator=(const LoadParams& src) { Flags = src.Flags; Gamma = src.Gamma; return *this; } 
51 
52 uint32 Flags; 
53 float Gamma; // Used iff decoding. 
54 }; 
55 
56 // Creates an invalid tImagePMK. You must call Load manually. 
57 tImagePKM() { } 
58 tImagePKM(const tString& pkmFile, const LoadParams& params = LoadParams()) { Load(pkmFile, params); } 
59 
60 // The data is copied out of pkmFileInMemory. Go ahead and delete after if you want. 
61 tImagePKM(const uint8* pkmFileInMemory, int numBytes, const LoadParams& params = LoadParams()) { Load(pkmFileInMemory, numBytes, params); } 
62 
63 // This one sets from a supplied pixel array. If steal is true it takes ownership of the pixels pointer. Otherwise 
64 // it just copies the data out. 
65 tImagePKM(tPixel4b* pixels, int width, int height, bool steal = false) { Set(pixels, width, height, steal); } 
66 
67 // Sets from a single frame. 
68 tImagePKM(tFrame* frame, bool steal = true) { Set(frame, steal); } 
69 
70 // Constructs from a tPicture. 
71 tImagePKM(tPicture& picture, bool steal = true) { Set(picture, steal); } 
72 
73 virtual ~tImagePKM() { Clear(); } 
74 
75 // Clears the current tImagePKM before loading. Returns success. If false returned, object is invalid. 
76 bool Load(const tString& pkmFile, const LoadParams& params = LoadParams()); 
77 bool Load(const uint8* pkmFileInMemory, int numBytes, const LoadParams& params = LoadParams()); 
78 
79 // This one sets from a supplied pixel array. If steal is true it takes ownership of the pixels pointer. Otherwise 
80 // it just copies the data out. 
81 bool Set(tPixel4b* pixels, int width, int height, bool steal = false) override; 
82 
83 // Sets from a single frame. 
84 bool Set(tFrame*, bool steal) override; 
85 
86 // Sets from a tPicture. 
87 bool Set(tPicture& picture, bool steal = true) override; 
88 
89 // After this call no memory will be consumed by the object and it will be invalid. 
90 void Clear() override; 
91 bool IsValid() const override { return (Layer && Layer->IsValid()); } 
92 
93 int GetWidth() const { return Layer ? Layer->Width : 0; } 
94 int GetHeight() const { return Layer ? Layer->Height : 0; } 
95 
96 // If decoded all pixels must be opaque (alpha = 255) for this to return true. 
97 // If not decoded it returns false if the pixel format supports transparency. 
98 bool IsOpaque() const; 
99 
100 // After the steal call you are the owner of the layer and must eventually delete it. This tImageASTC object is 
101 // invalid afterwards. 
102 tLayer* StealLayer() { tLayer* layer = Layer; Layer = nullptr; return layer; } 
103 tLayer* GetLayer() const { return Layer; } 
104 tFrame* GetFrame(bool steal = true) override; 
105 
106private: 
107 // We store the data in a tLayer because that's the container we use for pixel data than may be in any format. 
108 // The user of tImagePKM is not required to decode, so we can't just use a tPixel array. 
109 tLayer* Layer = nullptr; 
110};
