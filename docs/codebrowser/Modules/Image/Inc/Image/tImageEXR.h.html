3// This knows how to load and save OpenEXR images (.exr). It knows the details of the exr high dynamic range 
4// file format and loads the data into a tPixel array. These tPixels may be 'stolen' by the tPicture's constructor if 
5// an EXR file is specified. After the array is stolen the tImageEXR is invalid. This is purely for performance.

106inline void tImageEXR::LoadParams::Reset() 
107{ 
108 Gamma = tMath::DefaultGamma; 
109 Exposure = 1.0f; 
110 Defog = 0.0f; 
111 KneeLow = 0.0f; 
112 KneeHigh = 3.5f; 
113}

116inline bool tImageEXR::IsOpaque() const 
117{ 
118 for (tFrame* frame = Frames.Head(); frame; frame = frame->Next()) 
119 if (!frame->IsOpaque()) 
120 return false; 
121 
122 return true; 
123}

126inline tFrame* tImageEXR::StealFrame(int frameNum) 
127{ 
128 tFrame* f = GetFrame(frameNum); 
129 if (!f) 
130 return nullptr; 
131 
132 return Frames.Remove(item: f); 
133} 
134 
135 
136inline void tImage::tImageEXR::StealFrames(tList<tFrame>& frames) 
137{ 
138 while (tFrame* frame = Frames.Remove()) 
139 frames.Append(item: frame); 
140}

142 
143inline tFrame* tImageEXR::GetFrame(int frameNum) 
144{ 
145 if ((frameNum >= Frames.GetNumItems()) || (frameNum < 0)) 
146 return nullptr; 
147 
148 tFrame* f = Frames.First(); 
149 while (frameNum--) 
150 f = f->Next(); 
151 
152 return f; 
153}

155 
156inline void tImageEXR::Clear() 
157{ 
158 while (tFrame* frame = Frames.Remove()) 
159 delete frame; 
160 
161 tBaseImage::Clear();
