120inline tFrame* tImage::tImageWEBP::StealFrame(int frameNum) 
121{ 
122 tFrame* f = GetFrame(frameNum); 
123 if (!f) 
124 return nullptr; 
125 
126 return Frames.Remove(item: f); 
127} 
128 
129 
130inline void tImage::tImageWEBP::StealFrames(tList<tFrame>& frames) 
131{ 
132 while (tFrame* frame = Frames.Remove()) 
133 frames.Append(item: frame); 
134}

137inline tFrame* tImage::tImageWEBP::GetFrame(int frameNum) 
138{ 
139 if ((frameNum >= Frames.GetNumItems()) || (frameNum < 0)) 
140 return nullptr; 
141 
142 tFrame* f = Frames.First(); 
143 while (frameNum--) 
144 f = f->Next(); 
145 
146 return f; 
147}

150inline void tImageWEBP::Clear() 
151{ 
152 while (tFrame* frame = Frames.Remove()) 
153 delete frame; 
154 BackgroundColour = tColour4b::white;

156 tBaseImage::Clear();
