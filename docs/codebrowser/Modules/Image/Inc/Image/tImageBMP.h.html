81 enum class tFormat 
82 { 
83 Invalid, // Invalid must be 0. 
84 BPP24, // RGB. 24 bit colour. 
85 BPP32, // RGBA. 24 bit colour and 8 bits opacity in the alpha channel. 
86 Auto // Save function will decide format. BPP24 if all image pixels are opaque and BPP32 otherwise. 
87 }; 
88 
89 struct SaveParams 
90 { 
91 SaveParams() { Reset(); } 
92 SaveParams(const SaveParams& src) : Format(src.Format) { } 
93 void Reset() { Format = tFormat::Auto; } 
94 SaveParams& operator=(const SaveParams& src) { Format = src.Format; return *this; } 
95 tFormat Format; 
96 }; 
97 
98 // Saves the tImageBMP to the bmp file specified. The filetype must be "bmp". If tFormat is Auto, this function 
99 // will decide the format. BPP24 if all image pixels are opaque and BPP32 otherwise. Returns the format that the 
100 // file was saved in, or tFormat::Invalid if there was a problem. Since Invalid is 0, you can use an 'if'. 
101 tFormat Save(const tString& bmpFile, tFormat) const; 
102 tFormat Save(const tString& bmpFile, const SaveParams& = SaveParams()) const; 
103 
104 // After this call no memory will be consumed by the object and it will be invalid. 
105 void Clear() override; 
106 bool IsValid() const override { return Pixels ? true : false; } 
107 
108 int GetWidth() const { return Width; } 
109 int GetHeight() const { return Height; } 
110 
111 // All pixels must be opaque (alpha = 1) for this to return true. 
112 bool IsOpaque() const;

114 // After this call you are the owner of the pixels and must eventually delete[] them. This tImageBMP object is 
115 // invalid afterwards. 
116 tPixel4b* StealPixels(); 
117 tFrame* GetFrame(bool steal = true) override; 
118 tPixel4b* GetPixels() const { return Pixels; }
