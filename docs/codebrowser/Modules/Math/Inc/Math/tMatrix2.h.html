23struct tMatrix2 : public tMat2

32 void Set(const tMat2& m) { tSet(d&: *this, s: m); }

36 void Get(float* a) const { tGet(a, m: *this); }

40 void Zero() { tZero(d&: *this); } 
41 void Zero(tCompBit c) { tZero(d&: *this, c); }

54 tMatrix2& operator=(const tMat2& m) { tSet(d&: *this, s: m); return *this; }

61 // The multiplication by a vector is not declared as a friend since we treat vectors as column vectors only, 
62 // therefore the vector multiplication is not commutable. 
63 tMatrix2& operator*=(float a) { tMul(m&: *this, a); return *this; } 
64 tMatrix2& operator*=(const tMat2& a) { tMul(m&: *this, b: a); return *this; } 
65 const tMatrix2 operator*(const tMat2& a) const { tMatrix2 d; tMul(d, a: *this, b: a); return d; } 
66 const tVec2 operator*(const tVec2& a) const { tVec2 d; tMul(d, a: *this, b: a); return d; } 
67 inline friend const tMatrix2 operator*(float a, const tMat2& b) { tMatrix2 d; tMul(d, a, b); return d; } 
68 inline friend const tMatrix2 operator*(const tMat2& a, float b) { tMatrix2 d; tMul(d, a, b); return d; } 
69 tMatrix2& operator/=(float a) { tDiv(m&: *this, a); return *this; } 
70 const tMatrix2 operator/(float a) const { tMatrix2 d; tDiv(d, a: *this, b: a); return d; } 
71 
72 inline friend bool operator==(const tMat2& a, const tMat2& b) { return tEqual(a, b); } 
73 inline friend bool operator!=(const tMat2& a, const tMat2& b) { return tNotEqual(a, b); } 
74 operator const float*() { return E; }

77 float operator[](int i) const { return E[i]; }
