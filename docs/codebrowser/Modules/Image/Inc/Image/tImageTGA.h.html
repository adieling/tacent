30 enum LoadFlags 
31 { 
32 LoadFlag_None = 0, 
33 
34 // The most common way to interpret the alpha channel is as opacity (0.0 is fully transarent and 1.0 is fully 
35 // opaque). However there are some 16-bit TGAs (5551 with 1-bit alpha) in the wild that are saved with a 0 in 
36 // the alpha channel and are expected to be visible. The TGA specification is a bit vague on this point: 
37 // "If the pixel depth is 16 bits, the topmost bit is reserved for transparency." This statement was probably 
38 // intended to mean the topmpst bit was for the 'attribute'/alpha channel and should be interpreted as opacity. 
39 // In any case, these files exist, so the LoadFlag_AlphaOpacity flag is available to be disabled if necessary. 
40 // LoadFlag_AlphaOpacity Present : Interpret alpha normally (as opacity). 0 = transparent. 1 = opaque. 
41 // LoadFlag_AlphaOpacity Not Present: Interpret alpha reversed (as transparency). 0 = opaque. 1 = transparent. 
42 LoadFlag_AlphaOpacity = 1 << 0, 
43 LoadFlags_Default = LoadFlag_AlphaOpacity 
44 }; 
45 
46 struct LoadParams 
47 { 
48 LoadParams() { Reset(); } 
49 LoadParams(const LoadParams& src) : Flags(src.Flags) { } 
50 void Reset() { Flags = LoadFlags_Default; } 
51 LoadParams& operator=(const LoadParams& src) { Flags = src.Flags; return *this; } 
52 uint32 Flags; 
53 };

90 enum class tFormat 
91 { 
92 Invalid, // Invalid must be 0. 
93 BPP24, // 24 bit colour. 
94 BPP32, // 24 bit colour with 8 bits opacity in the alpha channel. 
95 Auto // Save function will decide format. BPP24 if all image pixels are opaque and BPP32 otherwise. 
96 };

115 // Saves the tImageTGA to the Targa file specified. The type of filename must be "tga". If tFormat is Auto, this 
116 // function will decide the format. BPP24 if all image pixels are opaque and BPP32 otherwise. Returns the format 
117 // that the file was saved in, or tFormat::Invalid if there was a problem. Since Invalid is 0, you can use an 'if'. 
118 tFormat Save(const tString& tgaFile, tFormat, tCompression = tCompression::RLE) const;
