25// A tImageDDS object represents and knows how to load a dds file. In general a DirectDrawSurface is composed of 
26// multiple layers -- each one a mipmap. It loads the data into tLayers. It can either decode to R8G8B8A8 layers, or leave 
27// the data as-is. Decode from BCn is supported. The layers may be 'stolen' from a tImageDDS so that excessive memcpys 
28// are avoided. After they are stolen the tImageDDS is invalid. Cubemaps and mipmaps are supported. 
29// @todo 1D and 3D textures are not supported yet. 
30class tImageDDS : public tBaseImage 
31{ 
32public: 
33 enum LoadFlag 
34 { 
35 LoadFlag_Decode = 1 << 0, // Decode the dds texture data into RGBA 32 bit layers. If not set, the layer data will remain unmodified. 
36 LoadFlag_ReverseRowOrder = 1 << 1, // OpenGL uses the lower left as the orig DirectX uses the upper left. Set flag for OpenGL. 
37 
38 // Gamma-correct. Gamma compression using an encoding gamma of 1/2.2. Flag only applies when decode flag set for 
39 // HDR / floating-point formats (BC6, rgb16f/32f, etc) images. Assumes (colour) data is linear and puts it in 
40 // gamma-space (brighter) for diaplay on a monitor. 
41 LoadFlag_GammaCompression = 1 << 2, 
42 LoadFlag_SRGBCompression = 1 << 3, // Same as above but uses the official sRGB transformation. Linear -> sRGB. Approx encoding gamma of 1/2.4 for part of curve. 
43 LoadFlag_AutoGamma = 1 << 4, // Tries to determine whether to apply sRGB compression based on pixel format. Call GetColourSpace to see if it applied. 
44 LoadFlag_ToneMapExposure = 1 << 5, // Apply exposure value when loading the dds. Only affects HDR (linear-colour) formats. 
45 LoadFlag_SpreadLuminance = 1 << 6, // For DDS files with a single Red or Luminance component, spread it to all the RGB channels (otherwise red only). Does not spread single-channel Alpha formats. Applies only if decoding a dds is an R-only or L-only format. 
46 LoadFlag_CondMultFourDim = 1 << 7, // Produce conditional success if image dimension not a multiple of 4. Only checks BC formats, 
47 LoadFlag_CondPowerTwoDim = 1 << 8, // Produce conditional success if image dimension not a power of 2. Only checks BC formats. 
48 LoadFlag_StrictLoading = 1 << 9, // If set even mildly ill-formed dds files will not load. 
49 LoadFlag_SwizzleBGR2RGB = 1 << 10, // Compressonator stores colours swizzled in their ETC exports. This fixes those files up. 
50 LoadFlags_Default = LoadFlag_Decode | LoadFlag_ReverseRowOrder | LoadFlag_SpreadLuminance | LoadFlag_AutoGamma 
51 };

53 // If an error is encountered loading the resultant object will return false for IsValid. You can call GetLastResult 
54 // to get more detailed information. There are some results that are not full-success that leave the object valid. 
55 // When decoding _and_ reversing row order, most BC 4x4 blocks can be massaged without decompression to fix the row 
56 // order. The more complex ones like BC6 and BC7 cannot be swizzled around like this (well, they probably could be, 
57 // but it's a non-trivial amount of work). 
58 // 
59 // A note on LoadFlag_ReverseRowOrder. tImageDDS tries to perform row-reversing before any decode operation. This is 
60 // often possible even if the DDS texture data is BC-compressed. However, for some of the more complex BC schemes 
61 // (eg. BC6 BC7) this reversal cannot be easily accomplished without a full decode and re-encode which would be 
62 // lossy. In these cases the row-reversal is done _after_ decoding. Unfortunalely decoding may not always be 
63 // requested (for example if you want to pass the image data directly to the GPU memory in OpenGL). In these cases 
64 // tImageDDS will be unable to reverse the rows. You will still get a valid object, but it will be a conditional 
65 // valid (GetStates() will have Conditional_CouldNotFlipRows flag set). You can call also call RowsReversed() to 
66 // see if row-reversal was performed. The conditional is only set if reversal was requested. 
67 // 
68 // Additional parameters may be processed during dds-loading. Gamma is only used if GammaCompression flag is set. 
69 // Exposure >= 0 (black) and only used if ToneMapExposure flag set. 
70 struct LoadParams 
71 { 
72 LoadParams() { Reset(); } 
73 LoadParams(const LoadParams& src) : Flags(src.Flags), Gamma(src.Gamma), Exposure(src.Exposure) { } 
74 void Reset() { Flags = LoadFlags_Default; Gamma = tMath::DefaultGamma; Exposure = 1.0f; } 
75 LoadParams& operator=(const LoadParams& src) { Flags = src.Flags; Gamma = src.Gamma; Exposure = src.Exposure; return *this; } 
76 
77 uint32 Flags; 
78 float Gamma; 
79 float Exposure; 
80 };

103 // The state of the tImage is a combination of one or more of the following enumerants. 
104 // The values of the enum are used as bit indices into a bitfield.

107 // The tImageDDS is considered valid. May be combined with the conditionals below. 
108 Valid, 
109 
110 // Conditional valid. Valid bit still set. 
111 FirstConditional, 
112 Conditional_CouldNotFlipRows = FirstConditional, 
113 Conditional_PitchXORLinearSize, // Possible if strict loading not set. 
114 Conditional_IncorrectPixelFormatSpec, // Possible if strict loading not set. 
115 Conditional_DimNotMultFourBC, 
116 Conditional_DimNotPowerTwoBC, 
117 LastConditional = Conditional_DimNotPowerTwoBC, 
118

193 // After calling StealLayers the current object will be invalid. This call populates the passed in layer 
194 // list. If the current object is not valid the passed-in layer list is left unmodified. The layer list is appended 
195 // to. It is not emptied if there are layers on the list when passed in. This call gives management of the layers 
196 // to the caller. It does not memcpy and create new layers which is why the object becomes invalid afterwards. If 
197 // the tImageDDS is a cubemap, this function returns false and leaves the object (and list) unmodified. See 
198 // StealCubemapLayers if you want to steal cubemap layers. 
199 bool StealLayers(tList<tLayer>&);

206 // Similar to StealLayers except it steals up to 6 layer-lists if the object is a cubemap. If the tImageDDS
