64// Interval Notation. When you see [a,b] or (a,b) the square bracket means include the endpoint and the rounded brackets 
65// mean exclude. See http://www.coolmath.com/algebra/07-solving-inequalities/03-interval-notation-01. When a function 
66// takes a bias argument, a Low bias will cause the return value to include the lower extent of the interval and exclude 
67// the high extent. A High bias will exclude the low end and include the high end. As a notational convenience When a 
68// function takes a bias argument, we'll write the interval as [(a,b)] and the enum below shows how to interpret it: 
69enum class tBias 
70{ 
71 Full, Outer = Full, // [a,b] 
72 Low, Left = Low, // [a,b) 
73 High, Right = High, // (a,b] 
74 Center, Inner = Center // (a,b) 
75};

344 inline int tGCDRec(int a, int b) 
345 { 
346 // This is the Euclidean algorithm: 
347 // b<a : gdc(a,b) = gdc(a-b,b). 
348 // Eg. gdc(16,4) == gdc(12,4) == gdc(8,4) == gdc(4,4) (done). 
349 // Instead of looping through the subracts one by one, we can just use the remainder / mod operator. 
350 if (b == 0) 
351 return a; 
352  
353 return tGCDRec(a: b, b: a % b); 
354 } 
355}

358inline int tMath::tGCD(int a, int b) 
359{ 
360 // Both zeroes is undefined. Any number divides zero, but since the other number is also zero, the 
361 // most reasonable it to return the biggest number we can. 
362 if ((a == 0) && (b == 0)) 
363 return MaxInt; 
364 
365 if (a < 0) a = -a; 
366 if (b < 0) b = -b; 
367 return tGCDRec(a, b); 
368}

371inline int tMath::tLCM(int a, int b) 
372{ 
373 // lcm(a,b) = |a| * (|b|/gcd(a,b)) 
374 // Since tGCD returns MaxInt for tGCD(0,0), we don't need to worry about divide by zero. 
375 if (a < 0) a = -a; 
376 if (b < 0) b = -b; 
377 return a * ( b / tGCD(a,b) );

411inline int tMath::tFindFirstClearBit(uint8 v) 
412{ 
413 // Find the first zero bit. The operation we do is log2((a xor (a+1)) +1) 
414 // This is guaranteed to be a power of two. 
415 uint8 freeBit = (v ^ (v+1)) + 1; 
416 
417 // If 0 it means nothing was found. 
418 if (!freeBit) 
419 return -1; 
420 
421 // Now get the log in base 2 of freeBit and wrap if position is 0. 
422 int c = tMath::tLog2(v: freeBit) ; 
423 if (c == 0) 
424 c = 8; 
425 
426 // This is the first cleared index in the bit array (from the LSB). 
427 return c-1; 
428}

472inline int tMath::tLog2(uint32 x) 
473{ 
474 if (x == 0) 
475 return -1; 
476 
477 // See "http://graphics.stanford.edu/~seander/bithacks.html#IntegerLog". 
478 const uint32 b[] = { 0xAAAAAAAA, 0xCCCCCCCC, 0xF0F0F0F0, 0xFF00FF00, 0xFFFF0000 }; 
479 int c = (x & b[0]) != 0; 
480 
481 c |= ((x & b[4]) != 0) << 4; 
482 c |= ((x & b[3]) != 0) << 3; 
483 c |= ((x & b[2]) != 0) << 2; 
484 c |= ((x & b[1]) != 0) << 1; 
485 return c;
