3// A tLayer is a data container for texture pixel data. It is used only by the tTexture class to store image data. It 
4// contains data that may be in a variety of hardware-ready formats, like dxt5. It is primarily used to store the 
5// multiple mipmap layers of a texture. Main members are width, height, pixel format, and a function to compute data 
6// size based on those three variables. It knows how to save and load itself in tChunk format. 
7// 
8// tLayers may have any width and height in [1, MaxLayerDimension]. If the pixel format is block-based (4x4 pixels) 
9// the tLayer still allows smaller than 4 width and height. However, a whole block is still needed so the number of 
10// bytes will be the block size for the particular BC format. For example, a 1x1 BC1 format layer still needs 8 bytes. 
11// A 5x5 BC format layer would need 4 blocks (same as an 8x8). The tLayer does not place further constraints on width 
12// and height. A higher level system, for example, may want to ensure power-of-two sizes, or multiple of 4, but that 
13// shouldn't and doesn't happen here.

38 // Constructs the layer from the chunk. Normally it copies the layer data from the chunk. However, if ownData is 
39 // false no mem-copy is performed and the layer just points into the chunk data. In this case it is vital that the 
40 // chunk object exists for the entire lifespan of the layer. 
41 tLayer(const tChunk& chunk, bool ownsData = true) { Load(chunk, ownData: ownsData); } 
42 
43 // Constructs a layer having the supplied width, height, and number of bytes. If steal memory is true, the data 
44 // array is passed off to this class and it will manage the array deleting it when necessary. If steal is false, 
45 // the data gets mem-copied into this object. 
46 tLayer(tPixelFormat fmt, int width, int height, uint8* data, bool steal = false) { Set(format: fmt, width, height, data, steal); } 
47 
48 tLayer(const tLayer& src) { Set(src); } 
49 virtual ~tLayer() { Clear(); }

51 bool IsValid() const { return Data ? true : false; }

53 void Set(tPixelFormat format, int width, int height, uint8* data, bool steal = false); 
54 void Set(const tLayer& layer);

56 // Returns the size of the data in bytes by reading the Width, Height, and PixelFormat. For block-compressed format 
57 // the data size will be a multiple of the block size in bytes. BC 4x4 blocks may be different sizes, whereas ASTC 
58 // block size is always 16 bytes. eg. a 1x1 BC1 format layer still needs 8 bytes. A 5x5 BC1 format layer would need 
59 // a whole 4 blocks (same as an 8x8) and would yield 32 bytes. 
60 int GetDataSize() const; 
61 
62 void Save(tChunkWriter&) const; 
63 
64 // Loads the layer from a chunk. Any previous layer data gets destroyed. The bool ownData works the same as with 
65 // the constructor. 
66 void Load(const tChunk&, bool ownData); 
67 
68 // Frees internal layer data and makes the layer invalid. 
69 void Clear() { PixelFormat = tPixelFormat::Invalid; Width = Height = 0; if (OwnsData) delete[] Data; Data = nullptr; OwnsData = true; } 
70 
71 // This just checks the pixel format to see if it supports alpha. It does NOT check the data. 
72 bool IsOpaqueFormat() const { return tImage::tIsOpaqueFormat(format: PixelFormat); } 
73 
74 uint8* StealData() { if (!OwnsData) return nullptr; OwnsData = false; return Data; }

76 // An invalid layer is never considered equal to another layer even if the other layer is also invalid. Whether the 
77 // layer owns the data is considered irrelevant for equivalency purposes. 
78 bool operator==(const tLayer&) const; 
79 bool operator!=(const tLayer& src) const { return !(*this == src); } 
80 
81 // The following functions are a convenience if the layer pixel format happens to be R8G8B8A8. For example, 
82 // block compressed images may have been told to decompress in which case the layers they contain will be R8G8B8A8. 
83 int GetIndex(int x, int y) const { tAssert((x >= 0) && (y >= 0) && (x < Width) && (y < Height) && (PixelFormat == tPixelFormat::R8G8B8A8)); return y * Width + x; } 
84 tPixel4b GetPixel(int x, int y) const { return ((tPixel4b*)Data)[ GetIndex(x, y) ]; }

101inline void tLayer::Set(tPixelFormat format, int width, int height, uint8* data, bool steal) 
102{ 
103 Clear(); 
104 if (!width || !height || !data) 
105 return; 
106 
107 PixelFormat = format; 
108 Width = width; 
109 Height = height; 
110 Data = nullptr; 
111 OwnsData = true; 
112 
113 if (steal) 
114 { 
115 Data = data; 
116 } 
117 else 
118 { 
119 int dataSize = GetDataSize(); 
120 Data = new uint8[dataSize]; 
121 tStd::tMemcpy(dest: Data, src: data, numBytes: dataSize); 
122 } 
123}

126inline void tLayer::Set(const tLayer& layer) 
127{ 
128 if (&layer == this) 
129 return; 
130 
131 PixelFormat = layer.PixelFormat; 
132 Width = layer.Width; 
133 Height = layer.Height; 
134 OwnsData = layer.OwnsData; 
135 
136 if (OwnsData) 
137 { 
138 int dataSize = layer.GetDataSize(); 
139 Data = new uint8[dataSize]; 
140 tStd::tMemcpy(dest: Data, src: layer.Data, numBytes: dataSize); 
141 } 
142 else 
143 { 
144 Data = layer.Data; 
145 } 
146}

149inline int tLayer::GetDataSize() const 
150{ 
151 if (!Width || !Height || (PixelFormat == tPixelFormat::Invalid)) 
152 return 0; 
153 
154 // Non-block-compressed textures are considered as having a single pixel per block. 
155 int blockW = tGetBlockWidth(PixelFormat); 
156 int blockH = tGetBlockHeight(PixelFormat); 
157 tAssert((blockW > 0) && (blockH > 0)); 
158 
159 int numBlocks = tGetNumBlocks(blockWH: blockW, imageWH: Width) * tGetNumBlocks(blockWH: blockH, imageWH: Height); 
160 int bytesPerBlock = 0; 
161 
162 // tGetBytesPerBlock _could_ also handle packed formats but for palettized formats 
163 // I think we still need to use tGetBitsPerPixel. 
164 if (tIsBCFormat(format: PixelFormat) || tIsASTCFormat(format: PixelFormat) || tIsPVRFormat(format: PixelFormat)) 
165 bytesPerBlock = tGetBytesPerBlock(PixelFormat); 
166 else 
167 bytesPerBlock = tGetBitsPerPixel(PixelFormat) >> 3; 
168 
169 // @todo I think we're currently in trouble here if we call tGetBitsPerPixel with palettized that 
170 // returns a non multiple of 8. The code below attempts to deal with this... but should be revisited. 
171 // It basically works out the total bits, and makes sure we have enough bytes to store them all even 
172 // if that total is not divisible by 8. 
173 if (bytesPerBlock == 0) 
174 { 
175 int totalBits = tGetBitsPerPixel(PixelFormat) * numBlocks; 
176 int totalBytesCeil = (totalBits/8) + ((totalBits%8) ? 1 : 0); 
177 return totalBytesCeil; 
178 } 
179 
180 return numBlocks * bytesPerBlock; 
181}
