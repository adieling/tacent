3// Small float (less than 32-bit) representations. Includes a 16-bit half-sized float (tHalf) and a couple of packed 
4// formats using 10, 11, and 14-bit floats. Arithmetic is not directly supported by these classes, they are used for 
5// simply converting raw bit batterns to and from 32-BIT IEEE-754 floats (which will have hardware support anyways). 
6// 
7// As a reminder, IEEE-754 floats have 1 sign bit, 8 exponent bits, and 23 mantissa bits, They are usually encoded with 
8// an implicit +1 in the mantissa, but support 'denormals' for small values (sub-normal) where there is no implied +1. 
9// All denormals are sub-normal in IEEE-754. 
10// 
11// For the 16-bit half-precision floats the conversion functions are based on code from Fabian "ryg" Giesen in the 
12// public domain: "I hereby place this code in the public domain, as per the terms of the CC0 
13// license: https://creativecommons.org/publicdomain/zero/1.0/" 
14// float -> half : https://gist.github.com/rygorous/2156668 
15// half -> float : https://gist.github.com/rygorous/2144712 
16// All modifications from the original should be considerd ISC as seen below. 
17// 
18// For the 32-bit packed M9M9M9E5 type the implementation is based on 
19// https://registry.khronos.org/OpenGL/extensions/EXT/EXT_texture_shared_exponent.txt 
20// There is no licence information provided, but the comment reads: "This source code provides ANSI C routines. It 
21// assumes the C 'float' data type is stored with the IEEE 754 32-bit floating-point format." Authors appear to be one 
22// or more of the following: Contact: Mark J. Kilgard, NVIDIA Corporation (mjk 'at' nvidia.com). Contributors: Pat Brown 
23// NVIDIA, Jon Leech, and Bruce Merry ARM. The extensions appear to be from an area of the site for implementers with 
24// this note: "No fee or agreement is needed to implement a Khronos API or use it in a product but you may not claim 
25// compliance and may not use its trademark and logo." 
26// 
27// In the following code you may see left shifts of negative signed integers. This is commonly thought to be undefined 
28// in C++, but it shouldn't be... and as of c++20 the standard does not even draw a distinction between negative and 
29// positive signed int left-shifts, which is good. I am considering it well-defined to left-shift a negative: 
30// Since C++20: 
31// The value of a << b is the unique value congruent to a * 2b modulo 2N where N is the number of bits in the return 
32// type (that is, bitwise left shift is performed and the bits that get shifted out of the destination type are 
33// discarded). The value of a >> b is a/2b, rounded towards negative infinity (in other words, right shift on signed a 
34// is arithmetic right shift).

205inline uint16 FloatToHalfRaw(float f) 
206{ 
207#ifdef TACENT_HALF_FLOAT_RTNE 
208 tFP32U f32(f); 
209 uint32 f32inf = 0xFF << 23; // Pos infinity. All 8 exponent bits set. 
210 uint32 f16max = (0x007F + 0x0010) << 23; 
211 tFP32U denormMagic (((0x007Fu - 0x000Fu) + (23u - 10u) + 1u) << 23); 
212 uint32 sgnmsk = 0x80000000; 
213 uint16 out = 0x0000; 
214 uint32 sign = f32.Raw & sgnmsk; 
215 f32.Raw ^= sign; 
216 
217 // All the integer compares in this function can be safely compiled into signed compares since all operands are 
218 // below 0x80000000. 
219 if (f32.Raw >= f16max) 
220 { 
221 out = (f32.Raw > f32inf) ? 0x7e00 : 0x7c00; // Result is Inf or NaN (all exponent bits set). NaN->qNaN and Inf->Inf. 
222 } 
223 else 
224 { 
225 if (f32.Raw < (0x00000071 << 23)) // (De)normalized number or zero. 
226 { 
227 // Resulting FP16 is subnormal or zero. Use a magic value to align our 10 mantissa bits at the bottom of 
228 // the float. As long as FP addition is round-to-nearest-even this just works. 
229 f32.Flt += denormMagic.Flt; 
230 out = f32.Raw - denormMagic.Raw; // One integer subtract of the bias later, we have our final float! 
231 } 
232 else 
233 { 
234 uint32 mantOdd = (f32.Raw >> (23u - 10u)) & 0x00000001; // Is resulting mantissa is odd? 
235 f32.Raw += ((0x0F - 0x7F) << 23) + 0x0FFF; // Update exponent, rounding bias part 1. 
236 f32.Raw += mantOdd; // Rounding bias part 2. 
237 out = f32.Raw >> 13; // Take the bits! 
238 } 
239 } 
240 
241 out |= sign >> 16; 
242 return out; 
243#else 
244 // Based off of float_to_half_fast3. Rounds to +inf. 
245 tFP32U f32(f); 
246 uint32 f32inf = 0xFFu << 23; // Pos infinity. All 8 exponent bits set. 
247 uint32 f16inf = 0x1Fu << 23; // Pos infinity. All 5 exponent bits set. 
248 tFP32U f32magic ( 0x0Fu << 23 ); 
249 uint32 f32sgnmsk = 0x80000000u; 
250 uint32 rndmask = ~0x00000FFFu; 
251 uint16 out = 0x0000u; 
252 
253 uint32 sign = f32.Raw & f32sgnmsk; 
254 f32.Raw ^= sign; 
255 
256 if (f32.Raw >= f32inf) // Inf or NaN (all exponent bits set) 
257 { 
258 out = (f32.Raw > f32inf) ? 0x7e00u : 0x7c00u;// NaN->qNaN and Inf->Inf 
259 } 
260 else // (De)normalized number or zero. 
261 { 
262 f32.Raw &= rndmask; 
263 f32.Flt *= f32magic.Flt; 
264 f32.Raw -= rndmask; 
265 
266 if (f32.Raw > f16inf) // Clamp to signed infinity if overflew. 
267 f32.Raw = f16inf; 
268 
269 out = f32.Raw >> 13; // Take the bits! 
270 } 
271 
272 out |= sign >> 16; 
273 return out; 
274#endif 
275}

278inline float HalfRawToFloat(uint16 raw) 
279{ 
280 // Based off of half_to_float_fast4. 
281 const tFP32U magic ( 0x71u << 23 ); 
282 const uint32 shiftedExp = 0x7c00 << 13; // Exponent mask after shift. 
283 tFP32U out ( (raw & 0x7FFFu) << 13 ); // Exponent/mantissa bits. 
284 uint32 exp = shiftedExp & out.Raw; // Just the exponent. 
285 out.Raw += (0x7F - 15) << 23; // Exponent adjust. 
286 
287 if (exp == shiftedExp) // Handle exponent special cases. First check for Inf/NaN. 
288 { 
289 out.Raw += (0x80 - 0x10) << 23; // Extra exp adjust. 
290 } 
291 else if (exp == 0) // Zero/Denormal. 
292 { 
293 out.Raw += 1 << 23; // Extra exp adjust. 
294 out.Flt -= magic.Flt; // Renormalize. 
295 } 
296 
297 out.Raw |= (raw & 0x8000) << 16; // Sign bit. 
298 return out.Flt; 
299}

302inline void tPackedF11F11F10::Set(float x, float y, float z) 
303{ 
304 tMath::tiClampMin(val&: x, min: 0.0f); 
305 tMath::tiClampMin(val&: y, min: 0.0f); 
306 tMath::tiClampMin(val&: z, min: 0.0f); 
307 
308 // SEEEEEMMMMMMMMMM 
309 uint16 xhalf = FloatToHalfRaw(f: x); 
310 uint16 yhalf = FloatToHalfRaw(f: y); 
311 uint16 zhalf = FloatToHalfRaw(f: z); 
312 
313 // The masks and shifts below remove the sign bit. No sign for F11F11F10. The mantissa for each float gets 
314 // truncated. Negatives would become positive except we clamped above. 
315 // EEEE EMMM MMMM MMM0 0000 0000 0000 0000 & 1111 1111 1110 0000 0000 0000 0000 0000 
316 // 0000 0000 00SE EEEE MMMM MMMM MM00 0000 & 0000 0000 0001 1111 1111 1100 0000 0000 
317 // 0000 0000 0000 0000 0000 0SEE EEEM MMMM & 0000 0000 0000 0000 0000 0011 1111 1111 
318 uint32 rx = (xhalf << 17) & 0xFFE00000; 
319 uint32 ry = (yhalf << 6 ) & 0x001FFC00; 
320 uint32 rz = (zhalf >> 5 ) & 0x000003FF; 
321 Raw = rx | ry | rz; 
322}

403inline void M999E5::FloatToComponents(int32& xm, int32& ym, int32& zm, int32& expShared, float x, float y, float z) 
404{ 
405 float xc = tMath::tClamp(val: x, min: 0.0f, max: M999E5::MAX); 
406 float yc = tMath::tClamp(val: y, min: 0.0f, max: M999E5::MAX); 
407 float zc = tMath::tClamp(val: z, min: 0.0f, max: M999E5::MAX); 
408 float maxxyz = tMath::tMax(a: xc, b: yc, c: zc); 
409 expShared = tMath::tMax(a: -M999E5::EXP_BIAS-1, b: M999E5::FloorLog2(f: maxxyz)) + 1 + M999E5::EXP_BIAS; 
410 tAssert((expShared <= M999E5::MAX_VALID_BIASED_EXP) && (expShared >= 0)); 
411 
412 // This pow function could be replaced by a table. 
413 double denom = tMath::tPow(a: 2.0, b: expShared - M999E5::EXP_BIAS - 9); 
414 int32 maxm = int(tMath::tFloor(v: maxxyz / denom + 0.5)); 
415 tAssert(maxm <= M999E5::MAX_MANTISSA+1); 
416 if (maxm == M999E5::MAX_MANTISSA+1) 
417 { 
418 denom *= 2.0; 
419 expShared += 1; 
420 tAssert(expShared <= M999E5::MAX_VALID_BIASED_EXP); 
421 } 
422 
423 xm = int32(tMath::tFloor(v: xc/denom + 0.5)); 
424 ym = int32(tMath::tFloor(v: yc/denom + 0.5)); 
425 zm = int32(tMath::tFloor(v: zc/denom + 0.5)); 
426 tAssert((xm <= M999E5::MAX_MANTISSA) && (xm >= 0)); 
427 tAssert((ym <= M999E5::MAX_MANTISSA) && (ym >= 0)); 
428 tAssert((zm <= M999E5::MAX_MANTISSA) && (zm >= 0)); 
429}
