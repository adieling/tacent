23tVisualStudioSolution::tVisualStudioSolution(const tString& slnFile) : 
24 Directory(), 
25 ProjectFiles() 
26{ 
27 if (!tSystem::tFileExists(file: slnFile)) 
28 throw tSolutionError("Cannot process solution file [%s] for dependencies.", slnFile.Pod()); 
29 
30 int numBytes = tSystem::tGetFileSize(file: slnFile); 
31 if (numBytes <= 0) 
32 return; 
33 
34 Directory = tSystem::tGetDir(path: slnFile); 
35 
36 // Load the file into a string. 
37 tString fileData(numBytes); 
38 tSystem::tLoadFile(file: slnFile, buffer: (uint8*)fileData.Text()); 
39 tString extension(".vcxproj"); 
40 
41 int startIndex = 0; 
42 int extIndex = fileData.FindString(str: extension, start: startIndex); 
43 while (extIndex != -1) 
44 { 
45 int startOfProjFile = fileData.FindChar(c: '"', reverse: true, start: extIndex); 
46 if (startOfProjFile == -1) 
47 break; 
48 startOfProjFile++; 
49 
50 int endOfProjFile = extIndex + extension.Length(); 
51 endOfProjFile--; 
52 
53 tString vcxProjFile(256); 
54 for (int c = 0; c < (endOfProjFile-startOfProjFile)+1; c++) 
55 vcxProjFile[c] = fileData[startOfProjFile+c]; 
56 
57 vcxProjFile.Replace(search: '\\', replace: '/'); 
58 vcxProjFile = tSystem::tGetSimplifiedPath(path: Directory + vcxProjFile); 
59 
60 ProjectFiles.Append(item: new tStringItem(vcxProjFile)); 
61 
62 startIndex = extIndex+1; 
63 extIndex = fileData.FindString(str: extension, start: startIndex); 
64 } 
65}

68tVisualStudioProject::tVisualStudioProject(const tString& projFile) : 
69 Directory(), 
70 Files() 
71{ 
72 if (!tSystem::tFileExists(file: projFile)) 
73 throw tSolutionError("Cannot process project file [%s] for dependencies.", projFile.Pod()); 
74 
75 int numBytes = tSystem::tGetFileSize(file: projFile); 
76 if (numBytes <= 0) 
77 return; 
78 
79 Directory = tSystem::tGetDir(path: projFile); 
80 
81 // Load the file into a string. 
82 tString fileData(numBytes); 
83 tSystem::tLoadFile(file: projFile, buffer: (uint8*)fileData.Text()); 
84 tString tagA("ClCompile Include="); 
85 tString tagB("ClInclude Include="); 
86 
87 int startIndex = 0; 
88 int tagAIndex = fileData.FindString(str: tagA, start: startIndex); 
89 int tagBIndex = fileData.FindString(str: tagB, start: startIndex); 
90 int tagIndex = tMath::tMin(a: tagAIndex, b: tagBIndex); 
91 while (tagIndex != -1) 
92 { 
93 int startOfFile = fileData.FindChar(c: '"', reverse: false, start: tagIndex); 
94 if (startOfFile == -1) 
95 break; 
96 startOfFile++; 
97 
98 int endOfFile = fileData.FindChar(c: '"', reverse: false, start: startOfFile); 
99 endOfFile--; 
100 
101 tString vcFile(256); 
102 for (int c = 0; c < (endOfFile-startOfFile)+1; c++) 
103 vcFile[c] = fileData[startOfFile+c]; 
104 
105 vcFile.Replace(search: '\\', replace: '/'); 
106 vcFile = tSystem::tGetSimplifiedPath(path: Directory + vcFile); 
107 
108 Files.Append(item: new tStringItem(vcFile)); 
109 
110 startIndex = endOfFile; 
111 tagAIndex = fileData.FindString(str: tagA, start: startIndex); 
112 tagBIndex = fileData.FindString(str: tagB, start: startIndex); 
113 tagIndex = tMath::tMin(a: tagAIndex, b: tagBIndex); 
114 } 
115}
