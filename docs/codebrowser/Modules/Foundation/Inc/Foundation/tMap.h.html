1// tMap.h 
2// 
3// A map class (dictionary) that keeps track of keys and associated values, both of which may be any type. The 
4// requirements are: The key type must be copyable, comparable, and convertable to a uint32. The value type must be 
5// copyable and should have a default constructor. tMap is implemented as a hash-table with lists to resolve collisions 
6// and has expected O(1) running time for insertions and value retrievals. The hash table automatically grows when a 
7// threshold percentage of the hash table is used (defaulting to 60%). Keys are unique -- the last value assigned to a 
8// key is the one stored in the tMap. 
9// 
10// You may iterate through a tMap to retrieve all keys and values. Range-based for loops are supported. Note that this 
11// is slightly less efficient than iterating through a tList though, as empty nodes in the hash table are visited.

31{ 
32public:

38 // found. The value will be default constructed. GetValue returns nullptr if the key can't be found. 
39 V* GetValue(const K&); 
40 V& GetInsert(const K&); 
41 V& operator[](const K&); 
42 bool Remove(const K&);

52 struct Pair 
53 { 
54 Pair(const Pair& pair) : Key(pair.Key), Value(pair.Value) { } 
55 
56 // Note that values-types like int are default-constructed as 0 by Value(). 
57 Pair(const K& key) : Key(key), Value() { } 
58 K Key; V Value;

60 struct HashTableItem 
61 { 
62 tItList<Pair> Pairs; 
63 }; 
64 void Rekey(int newSize); 
65 int NumItems; 
66 int HashTableSize;

71public: 
72 // If needed you can use this iterator to query every key and/or value in the tMap. Note that the keys and values 
73 // are stored unordered. Further, since not all hash table entries will have valid data, it is slightly less 
74 // efficient to iterate through a tMap vs interating through, for example, a tItList, since empties must be skipped. 
75 class Iter 
76 { 
77 public: 
78 Iter() : Map(nullptr), TableIndex(-1), PairIter() { } 
79 Iter(const Iter& src) : Map(src.Map), TableIndex(src.TableIndex), PairIter(src.PairIter) { } 
80 
81 bool IsValid() const { return PairIter.IsValid(); } 
82 void Clear() { Map = nullptr; TableIndex = -1; PairIter.Clear(); } 
83 void Next(); 
84 V& Value() const { return PairIter->Value; } 
85 K& Key() const { return PairIter->Key; } 
86 
87 Iter& operator*() { return *this; } 
88 Iter* operator->() const { return this; } 
89 operator bool() const { return PairIter.IsValid(); } 
90 Iter& operator=(const Iter& i) { if (this != &i) { Map = i.Map; TableIndex = i.TableIndex; PairIter = i.PairIter; } return *this; } 
91 
92 // Use ++iter instead of iter++ when possible. Since the hash table is unordered, there's no point in offering 
93 // both forward and backwards iteration. Therefore there's only First, Next, operator++ etc. 
94 const Iter operator++(int) { Iter curr(*this); Next(); return curr; } 
95 const Iter operator++() { Next(); return *this; } 
96 const Iter operator+(int offset) const { Iter i = *this; while (offset--) i.Next(); return i; } 
97 Iter& operator+=(int offset) { tAssert(offset >= 0); while (offset--) Next(); return *this; } 
98 bool operator==(const Iter& i) const { return (Map == i.Map) && (TableIndex == i.TableIndex) && (PairIter == i.PairIter); } 
99 bool operator!=(const Iter& i) const { return (Map != i.Map) || (TableIndex != i.TableIndex) || (PairIter != i.PairIter); } 
100 
101 private: 
102 friend class tMap; 
103 Iter(const tMap<K,V>* map, int tableIndex, typename tItList<Pair>::Iter iter) : Map(map), TableIndex(tableIndex), PairIter(iter) { } 
104 const tMap<K,V>* Map; 
105 int TableIndex; 
106 typename tItList<Pair>::Iter PairIter; 
107 }; 
108 
109public: 
110 Iter First() const; 
111 Iter begin() const /* For range-based iteration supported by C++11. */ { return First(); } 
112 Iter end() const /* For range-based iteration supported by C++11. */ { return Iter(this, -1, typename tItList<Pair>::Iter()); } 
113};

119template<typename K, typename V> inline tMap<K,V>::tMap(int initialLog2Size, float rekeyPercent) 
120{ 
121 NumItems = 0; 
122 
123 tAssert(initialLog2Size >= 0); 
124 HashTableSize = 1 << initialLog2Size; 
125 
126 tAssert(rekeyPercent >= 0.0f); 
127 RekeyPercent = rekeyPercent; 
128 
129 HashTableEntryCount = 0; 
130 HashTable = new HashTableItem[HashTableSize];

140template<typename K, typename V> inline V& tMap<K,V>::GetInsert(const K& key) 
141{ 
142 // Do we need to grow the hash table? 
143 if (GetHashTablePercent() >= RekeyPercent) 
144 Rekey(newSize: 2*HashTableSize); 
145 
146 uint32 hash = uint32(key); 
147 int hashBits = tMath::tLog2(x: HashTableSize); 
148 hash = hash & (0xFFFFFFFF >> (32-hashBits)); 
149 tAssert(hash < HashTableSize); 
150 
151 HashTableItem& item = HashTable[hash]; 
152 for (Pair& pair : item.Pairs) 
153 { 
154 if (pair.Key == key) 
155 return pair.Value; 
156 } 
157 
158 if (item.Pairs.IsEmpty()) 
159 HashTableEntryCount++; 
160 NumItems++; 
161 return item.Pairs.Append(new Pair(key))->Value; 
162}

165template<typename K, typename V> inline V* tMap<K,V>::GetValue(const K& key) 
166{ 
167 uint32 hash = uint32(key); 
168 int hashBits = tMath::tLog2(x: HashTableSize); 
169 hash = hash & (0xFFFFFFFF >> (32-hashBits)); 
170 tAssert(hash < HashTableSize); 
171 
172 HashTableItem& item = HashTable[hash]; 
173 for (Pair& pair : item.Pairs) 
174 { 
175 if (pair.Key == key) 
176 return &pair.Value; 
177 } 
178 
179 return nullptr; 
180}

189template<typename K, typename V> inline bool tMap<K,V>::Remove(const K& key) 
190{ 
191 uint32 hash = uint32(key); 
192 int hashBits = tMath::tLog2(x: HashTableSize); 
193 hash = hash & (0xFFFFFFFF >> (32-hashBits)); 
194 tAssert(hash < HashTableSize); 
195 
196 HashTableItem& item = HashTable[hash]; 
197 
198 for (auto iter = item.Pairs.First(); iter; iter++) 
199 { 
200 if (iter->Key == key) 
201 { 
202 delete item.Pairs.Remove(iter); 
203 if (item.Pairs.IsEmpty()) 
204 HashTableEntryCount--; 
205 NumItems--; 
206 return true; 
207 } 
208 } 
209 
210 return false; 
211}

214template<typename K, typename V> inline void tMap<K,V>::Rekey(int newSize) 
215{ 
216 tAssert((newSize > HashTableSize) && tMath::tIsPower2(newSize)); 
217 HashTableItem* newTable = new HashTableItem[newSize]; 
218 
219 // Loop throught existing keys and rekey them into the new table 
220 for (int i = 0; i < HashTableSize; i++) 
221 { 
222 for (Pair& pair : HashTable[i].Pairs) 
223 { 
224 uint32 hashNew = uint32(pair.Key); 
225 int hashBitsNew = tMath::tLog2(x: newSize); 
226 hashNew = hashNew & (0xFFFFFFFF >> (32-hashBitsNew)); 
227 
228 tAssert(hashNew < newSize); 
229 newTable[hashNew].Pairs.Append(new Pair(pair)); 
230 } 
231 HashTable[i].Pairs.Clear(); 
232 } 
233 
234 HashTableSize = newSize; 
235 delete[] HashTable; 
236 HashTable = newTable; 
237}

240template<typename K, typename V> inline void tMap<K,V>::Iter::Next() 
241{ 
242 // If we can just advance the PairIter, we're done. 
243 PairIter++; 
244 if (PairIter.IsValid()) 
245 return; 
246 
247 // Try next hash table entries until we either find a non-empty one or reach the end of the table. 
248 while (++TableIndex < Map->HashTableSize) 
249 { 
250 HashTableItem& item = Map->HashTable[TableIndex]; 
251 if (!item.Pairs.IsEmpty()) 
252 { 
253 PairIter = item.Pairs.First(); 
254 return; 
255 } 
256 }; 
257 
258 // It is vital to have 'this' be the same as end() here, as ranged-based for loops must return false when 
259 // comparing the last Next() with end() using != operator. This is why must set the index to -1. To make 
260 // sure it matches end(). 
261 TableIndex = -1; 
262 PairIter = typename tItList<Pair>::Iter();  
263}

266template<typename K, typename V> inline typename tMap<K,V>::Iter tMap<K,V>::First() const 
267{ 
268 for (int tableIndex = 0; tableIndex < HashTableSize; tableIndex++) 
269 { 
270 HashTableItem& item = HashTable[tableIndex]; 
271 if (!item.Pairs.IsEmpty()) 
272 return Iter(this, tableIndex, item.Pairs.First()); 
273 } 
274 return Iter(this, -1, typename tItList<Pair>::Iter()); 
275}
