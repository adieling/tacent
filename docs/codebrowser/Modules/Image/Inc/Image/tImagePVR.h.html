6// class supports V1, V2, and V3 pvr files.

26// A tImagePVR object represents and knows how to load a pvr file. A pvr file is a container format much like ktx or 
27// dds. It comes in 3 different versions: V1, V2, and V3. All three use the same pvr extension. This class loads the 
28// data into tLayers. It can either decode to R8G8B8A8 layers, or leave the data as-is. There are many pixel formats 
29// supported by pvr V3 files including ASTC, BCn, and the PVRTC formats. The PVRTC formats also come in two main 
30// versions: V1 and V2. If a V1 format is used, the file should be checked to ensure it is POT. The layers may be 
31// 'stolen' from a tImagePVR so that excessive memcpys are avoided. After they are stolen the tImagePVR is invalid. 
32// Cubemaps, mipmaps, texture arrays, and 3D textures are supported. 
33class tImagePVR : public tBaseImage 
34{ 
35public: 
36 enum LoadFlag 
37 { 
38 LoadFlag_Decode = 1 << 0, // Decode the pvr texture data into RGBA 32-bit layers. If not set, the layer data will remain unmodified. 
39 LoadFlag_ReverseRowOrder = 1 << 1, // OpenGL uses the lower left as the orig DirectX uses the upper left. Set flag for OpenGL. 
40 
41 // Gamma-correct. Gamma compression using an encoding gamma of 1/2.2. Flag only applies when decode flag set for 
42 // HDR / floating-point formats (BC6, rgb16f/32f, etc) images. Assumes (colour) data is linear and puts it in 
43 // gamma-space (brighter) for diaplay on a monitor. 
44 LoadFlag_GammaCompression = 1 << 2, 
45 LoadFlag_SRGBCompression = 1 << 3, // Same as above but uses the official sRGB transformation. Linear -> sRGB. Approx encoding gamma of 1/2.4 for part of curve. 
46 LoadFlag_AutoGamma = 1 << 4, // Tries to determine whether to apply sRGB compression based on pixel format. Call GetColourSpace to see if it applied. 
47 LoadFlag_ToneMapExposure = 1 << 5, // Apply exposure value when loading. Only affects HDR (linear-colour) formats. 
48 LoadFlag_SpreadLuminance = 1 << 6, // For files with a single Red or Luminance component, spread it to all the RGB channels (otherwise red only). Does not spread single-channel Alpha formats. Applies only if decoding an R-only or L-only format. 
49 LoadFlag_CondMultFourDim = 1 << 7, // Produce conditional success if image dimension not a multiple of 4. Only checks BC formats, 
50 LoadFlag_StrictLoading = 1 << 8, // If set ill-formed files will not load. Specifically if format is PVRTC (not PVRTC2) the texture must be POT if this flag set. 
51 LoadFlag_MetaDataOrient = 1 << 9, // Undo orientation transformations in pvr3 image as indicated by pvr3 meta-data. Works iff decoding or already in RGBA. 
52 
53 LoadFlags_Default = LoadFlag_Decode | LoadFlag_ReverseRowOrder | LoadFlag_SpreadLuminance | LoadFlag_AutoGamma | LoadFlag_StrictLoading | LoadFlag_MetaDataOrient 
54 };

56 // If an error is encountered loading the resultant object will return false for IsValid. You can call GetLastResult 
57 // to get more detailed information. There are some results that are not full-success that leave the object valid. 
58 // When decoding _and_ reversing row order, most BC 4x4 blocks can be massaged without decompression to fix the row 
59 // order. The more complex ones like BC6 and BC7 cannot be swizzled around like this (well, they probably could be, 
60 // but it's a non-trivial amount of work). 
61 // 
62 // A note on LoadFlag_ReverseRowOrder. tImagePVR tries to perform row-reversing before any decode operation. This is 
63 // often possible even if the PVR texture data is BC-compressed. However, for some of the more complex BC schemes 
64 // (eg. BC6 BC7) this reversal cannot be easily accomplished without a full decode and re-encode which would be 
65 // lossy. In these cases the row-reversal is done _after_ decoding. Unfortunalely decoding may not always be 
66 // requested (for example if you want to pass the image data directly to the GPU memory in OpenGL). In these cases 
67 // tImagePVR will be unable to reverse the rows. You will still get a valid object, but it will be a conditional 
68 // valid (GetStates() will have Conditional_CouldNotFlipRows flag set). You can call also call RowsReversed() to 
69 // see if row-reversal was performed. The conditional is only set if reversal was requested. 
70 // 
71 // Additional parameters may be processed during loading. Gamma is only used if GammaCompression flag is set. 
72 // Exposure >= 0 (black) and only used if ToneMapExposure flag set. 
73 struct LoadParams 
74 { 
75 LoadParams() { Reset(); } 
76 LoadParams(const LoadParams& src) : Flags(src.Flags), Gamma(src.Gamma), Exposure(src.Exposure), MaxRange(src.MaxRange) { } 
77 void Reset() { Flags = LoadFlags_Default; Gamma = tMath::DefaultGamma; Exposure = 1.0f; MaxRange = 8.0f; } 
78 LoadParams& operator=(const LoadParams& src) { Flags = src.Flags; Gamma = src.Gamma; Exposure = src.Exposure; MaxRange = src.MaxRange; return *this; } 
79 
80 uint32 Flags; 
81 float Gamma; 
82 float Exposure; 
83 float MaxRange; // Used for RGBM and RGBD only. 
84 };

245 // This is lifted from the PVR3 spec. It is a superset of the PVR1 and PVR2 structure. 
246 // 
247 // for each MIP-Map Level in MIP-Map Count 
248 // for each Surface in Num. Surfaces 
249 // for each Face in Num. Faces 
250 // for each Slice in Depth 
251 // for each Row in Height 
252 // for each Pixel in Width 
253 // Byte data[Size_Based_On_PixelFormat] 
254 // 
255 // The data ordering is different for PVR1/2 files but will fit info PVR3 structure 
256 // as listed above. 
257 // 
258 // for each Surface in Num. Surfaces 
259 // for each Face in 6 
260 // for each MIP-Map Level in MIP-Map Count 
261 // for each Row in Height 
262 // for each Pixel in Width 
263 // Byte data[Size_Based_On_PixelFormat] 
264 // 
265 // In this loader we'll leverage tLayers which deal with the Width, Height, and Byte data. 
266 // We'll simply create enough tLayer slots for NumSurfaces*NumFaces*NumMipmaps*Depth. 
267 // Rather than either the V1/2 or V3 ordering we'll be using: 
268 // 
269 // Surfaces 
270 // Faces 
271 // Mipmaps 
272 // Slices 
273 // Layers 
274 int NumSurfaces = 0; // For storing arrays of image data. 
275 int NumFaces = 0; // For cubemaps. Cubemaps are always specified using a left-handed coord system even when using the OpenGL functions. 
276 int NumMipmaps = 0; 
277 
278 int Depth = 0; // Number of slices. 
279 int Width = 0; 
280 int Height = 0; 
281 
282 int NumLayers = 0; 
283 tLayer** Layers = nullptr; // Always nullptr if NumLayers is 0.
