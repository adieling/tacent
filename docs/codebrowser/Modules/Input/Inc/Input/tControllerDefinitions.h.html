45// The VidPid acts as a key when retreiving information for a controller. 
46struct tVidPid 
47{ 
48 tVidPid() : VID(0), PID(0) { } 
49 tVidPid(uint16 vid, uint16 pid) : VID(vid), PID(pid) { } 
50 uint16 VID; // Vendor ID. 
51 uint16 PID; // Product ID. 
52 
53 // These are needed so that a VidPid can be used as a key in a tMap. 
54 explicit operator uint32() { return (VID << 16) | PID; } 
55 explicit operator uint32() const { return (VID << 16) | PID; } 
56}; 
57inline bool operator==(const tVidPid& a, const tVidPid& b) { return (a.VID == b.VID) && (a.PID == b.PID); }

60// The controller component displacement technology. 
61enum class tDispTech 
62{ 
63 UNK = -1, // Unknown. 
64 POT, // Potentiometer. Physical contact. May drift. 
65 HAL, // Hall effect. No physical contact. 
66 TMR // Tunnel MagnetoResistance. No physical contact. 
67};

70// The controller definition stored in the dictionary. Unfortunately some controllers have multiple ways to connect with 
71// different latency characteristics. Usually the 2.4GHz dongles are pretty close to wired. Also quite often bluetooth 
72// connected controllers aren't recognized by XInput (plus BT has a lot more latency). For this reason the wired 
73// definitions are used in the table below. 
74struct tContDefn 
75{ 
76 // I want to keep tContDefn as an aggregrate type. For this reason if you need to set the parameters to a reasonable 
77 // 'generic' set of values, call this function. 
78 void SetGeneric(); 
79 void Clear(); 
80 
81 const char* Vendor; 
82 const char* Product; 
83 
84 // Polling frequency in Hz. From this an appropriate polling period can be computed. This is the "max" frequency in 
85 // the sense that a particular controller may poll at a different frequency for buttons vs analog inputs like 
86 // triggers and sticks. However, the xinput API reads everything at once so we need a polling rate that is as fast 
87 // as the fastest component polling rate in the controller. 
88 int32 MaxPollingFreq; 
89 
90 // Displacement tech used by the joysticks. 
91 tDispTech DispTechSticks; 
92 
93 // Displacement tech used by the triggers. 
94 tDispTech DispTechTriggers; 
95  
96 // Stick dead zone expressed as a percent [0.0,1.0]. Assumes multiple sticks have the same deat-zone. 
97 float StickDeadZone; 
98 
99 // Trigger dead zone expressed as percent [0.0, 1.0]. 
100 float TriggerDeadZone; 
101  
102 // Latency here is the average time in milliseconds between an input change and the controller reporting the change. 
103 float LatencyAxes; 
104 float LatencyButtons; 
105 
106 // Jitter represents inconsistencies in the time-domain reporting of input changes. It is listed here as the 
107 // standard deviation of latency measurements in milliseconds. There is a 68.2% probability of a sample being 
108 // within +- one standard deviation from the average latency. Jitter values can be used to control the 'tau' value 
109 // in a low-pass filter. Such a filter mitigate both jitter and noise. 
110 float JitterAxes; 
111 float JitterButtons; 
112};

127inline void tInput::tContDefn::SetGeneric() 
128{ 
129 Vendor = "Unknown"; 
130 Product = "Generic"; 
131 MaxPollingFreq = 125; 
132 DispTechSticks = tDispTech::UNK; 
133 DispTechTriggers = tDispTech::UNK; 
134 
135 // The XInput deadzones are really too big (like 24%). We're choosing 15% for a generic controller. 
136 StickDeadZone = 0.15f; 
137 TriggerDeadZone = 0.05f; 
138 LatencyAxes = 5.50f; 
139 LatencyButtons = 5.50f; 
140 JitterAxes = 2.25f; 
141 JitterButtons = 2.25f; 
142}
