34int64 tSystem::tGetHardwareTimerFrequency() 
35{ 
36 if (!HPCFrequency) 
37 { 
38 #if defined(PLATFORM_WINDOWS) 
39 int64 freq; 
40 QueryPerformanceFrequency((WinLargeInteger*)&freq); 
41 HPCFrequency = freq; 
42 #else 
43 HPCFrequency = 1000000000ll; 
44 #endif 
45 } 
46 
47 return HPCFrequency;

51int64 tSystem::tGetHardwareTimerCount() 
52{ 
53 #if defined(PLATFORM_WINDOWS) 
54 int64 count; 
55 QueryPerformanceCounter((WinLargeInteger*)&count); 
56 return count; 
57 
58 #else 
59 struct timespec ts; 
60 int err = clock_gettime(CLOCK_MONOTONIC_RAW, tp: &ts); 
61 if (err) 
62 return 0; 
63 int64 nanoSecs = (1000000000ll * ts.tv_sec) + ts.tv_nsec; 
64 return nanoSecs; 
65 
66 #endif 
67}

96tString tSystem::tConvertTimeToString(std::tm timePoint, tTimeFormat format) 
97{ 
98 tString tstr(256); 
99 
100 switch (format) 
101 { 
102 case tTimeFormat::Standard: 
103 // Eg. 2020-01-14 01:47:12 
104 strftime(s: (char*)tstr.Text(), maxsize: 256, format: "%F %T", tp: &timePoint); 
105 break; 
106 
107 case tTimeFormat::Extended: 
108 // Eg. Tuesday January 14 2020 - 01:36:34 
109 strftime(s: (char*)tstr.Text(), maxsize: 256, format: "%A %B %d %G - %T", tp: &timePoint); 
110 break; 
111 
112 case tTimeFormat::Short: 
113 // Eg. Tue Jan 14 14:38:58 2020 
114 strftime(s: (char*)tstr.Text(), maxsize: 256, format: "%c", tp: &timePoint); 
115 break; 
116 
117 case tTimeFormat::Filename: 
118 // Eg. 2023-02-14-23-55-09 
119 strftime(s: (char*)tstr.Text(), maxsize: 256, format: "%F-%H-%M-%S", tp: &timePoint); 
120 break; 
121 } 
122 
123 return tstr; 
124}

127void tSystem::tSleep(int milliSeconds) 
128{ 
129 #if defined(PLATFORM_WINDOWS) 
130 ::Sleep(milliSeconds); 
131 
132 #else 
133 struct timespec ts; 
134 ts.tv_sec = milliSeconds / 1000; 
135 ts.tv_nsec = 1000000*(milliSeconds - (ts.tv_sec*1000)); 
136 nanosleep(requested_time: &ts, remaining: nullptr); 
137 
138 #endif 
139}

142float tSystem::tGetTime() 
143{ 
144 #ifdef PLATFORM_WINDOWS 
145 int64 freq = tGetHardwareTimerFrequency(); 
146 
147 // This is organized like this to keep precision as high as possible. 
148 int64 elapsedCount = tGetHardwareTimerCount() - HPCStartCount; 
149 int64 wholeSeconds = elapsedCount / freq; 
150 int64 remainder = elapsedCount % freq; 
151 return float(wholeSeconds) + float(remainder)/float(freq); 
152  
153 #else 
154 int64 microSecs = (tGetHardwareTimerCount() - HPCStartCount) / 1000ll; 
155 return float(microSecs) / 1000000.0f; 
156 
157 #endif 
158}

202void tSystem::tTimer::PrintHighPrecisionConversionTable(const char* outputFile) 
203{ 
204 double inSeconds[int(tUnit::tTime::NumTimeUnits)] = 
205 { 
206 5.39E-44, // PlankTime 
207 6.97E-24, // Chronon 
208 1.0E-18, // Atosecond 
209 1.0E-15, // Femtosecond 
210 1.0E-12, // Picosecond 
211 1.0E-9, // Nanosecond 
212 1.0E-6, // Microsecond 
213 1.0E-3, // Millisecond 
214 1.0/60.0, // Tick 
215 1.0, // Second 
216 0.3, // She 
217 3.0 + 1.0/3.0, // Helek 
218 60.0, // Minute 
219 3600.0, // Hour 
220 86400.0, // Day 
221 604800.0, // Week 
222 1209600.0, // Fortnight 
223 31536000.0, // Year 
224 31536000.0, // Annum 
225 3153600000.0, // Century 
226 31536000000.0, // Millennium 
227 7884000000000000.0, // GalacticYear 
228 }; 
229 
230 tFileHandle file = tSystem::tOpenFile(file: outputFile, mode: "wt"); 
231 
232 tfPrintf(dest: file, format: "// "); 
233 for (int n = 0; n < int(tUnit::tTime::NumTimeUnits); n++) 
234 { 
235 int count = tfPrintf(dest: file, format: "%s", tUnit::GetUnitName(tUnit::tTime(n))); 
236 for (int space = 0; space < (25-count); space++) 
237 tfPrintf(dest: file, format: " "); 
238 } 
239 tfPrintf(dest: file, format: "\n"); 
240 
241 for (int from = 0; from < int(tUnit::tTime::NumTimeUnits); from++) 
242 { 
243 int count = tfPrintf(dest: file, format: "/* %s */ ", tUnit::GetUnitName(tUnit::tTime(from))); 
244 for (int space = 0; space < (21-count); space++) 
245 tfPrintf(dest: file, format: " "); 
246 
247 tfPrintf(dest: file, format: "{"); 
248 for (int to = 0; to < int(tUnit::tTime::NumTimeUnits); to++) 
249 { 
250 // From -> to. t = f * t/f. 
251 tfPrintf(dest: file, format: " %019.16E,", inSeconds[from]/inSeconds[to]); 
252 } 
253 tfPrintf(dest: file, format: " },\n"); 
254 } 
255 tSystem::tCloseFile(f: file); 
256};
