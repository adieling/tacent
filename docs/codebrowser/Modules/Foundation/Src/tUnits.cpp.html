99tUnit::tTime tUnit::GetTimeUnit(const char* unitName) 
100{ 
101 for (int u = 0; u < int(tTime::NumTimeUnits); u++) 
102 { 
103 if (!tStd::tStricmp(a: unitName, b: TimeUnitNames[u])) 
104 return tTime(u); 
105 } 
106 
107 return tTime::Unspecified; 
108}
