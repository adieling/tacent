3// Implementations for low and high pass filters.

5// LowPass 
6// 
7// A simple first order low-pass filter using exponential smoothing. Uses a weight E [0.0, 1.0] to combine the current 
8// input with the previous filtered value -- the bigger the weight, the less lag and less filtering (higher cutoff 
9// frequency). Specifying the weight directly is not useful as it doesn't take into account how often the update 
10// function is called. For this reason delta-time and cutoff-frequency are used instead. Cutoff-frequency determines the 
11// frequency at which signals begin to be attenuated (reduced in amplitude). Lower cutoff frequencies result in more 
12// smoothing/jitter-reduction but introduce more lag. 
13// 
14// Optionally the delta-time and a time-constant (Tau) can be used. Tau is related to cutoff-frequency and represents 
15// the time (in seconds) it takes for the filter's output to reach approximately 63% of a raw input value. Tau can be 
16// useful if you have imperical jitter data on a hardware sensor that specifies jitter in seconds or milli-seconds. 
17// 
18// In some cases you know the update period or frequency and it doesn't change between updates. In some cases the time 
19// delta between updates is variable. To accomodate this there are separate classes for fixed or dynamic time-steps. 
20// Classes with Fix in the name should be used if you know the timestep and it is constant. Classes with Dyn should be 
21// used if you need to supply the delta-time each update. There are also variants that use either float or double (for 
22// more accuracy) -- these have either Dbl or Flt in their name.

41// An abstract base class for 1st order low-pass filters. 
42template<typename T> class tLowPassFilter 
43{ 
44public: 
45 tLowPassFilter(T initialValue) : Value(initialValue) { } 
46  
47 // Call this after Update if it's difficult to store the result immediately. 
48 T GetValue() const { return Value; } 
49 virtual void SetCutoffFreq(T cutoffFreq) = 0; 
50 virtual void SetTau(T tau) = 0; 
51 
52protected: 
53 T ComputeTau(T cutoffFreq) const { return T(1.0) / (tMath::TwoPi * cutoffFreq); } 
54 T ComputeWeight(T dt, T tau) const { return tSaturate(T(1.0) - tMath::tExp(-dt / tau)); } 
55 T ComputeWeightFast(T dt, T tau) const { return tSaturate(dt / (tau + dt)); } 
56 T UpdateValue(T weight, T inputValue) { Value = weight*inputValue + (T(1.0) - weight)*Value; return Value; } 
57 
58 // The current filtered value. 
59 T Value; 
60};

63template <typename T> class tLowPassFilterFixed : public tLowPassFilter<T> 
64{ 
65public: 
66 // If tauAttenuation is false attenuation is specified as a cutoffFrequency. If tauAttenuation is true attenuation 
67 // is specified as 'tau'. Cutoff-frequency is in Hz above which the signal is attenuated (reduced in amplitude). Tau 
68 // is specified in seconds and is the time it takes for the filter's output to reach approximately 63% of a raw 
69 // input value. FixedDeltaTime is in seconds. 
70 tLowPassFilterFixed(T fixedDeltaTime, T attenuation, bool tauAttenuation = false, T initialValue = T(0.0)); 
71 
72 // Given the new value returns the filtered value. Call this every fixedDeltaTime seconds. 
73 T Update(T inputValue); 
74 
75 void SetCutoffFreq(T cutoffFreq) override; 
76 void SetTau(T tau) override; 
77 
78private: 
79 // We need to bring required base class functions into scope. The base template class hasn't been instantiated yet. 
80 using tLowPassFilter<T>::ComputeTau; 
81 using tLowPassFilter<T>::ComputeWeight; 
82 using tLowPassFilter<T>::UpdateValue; 
83 
84 T FixedDeltaTime; // Stored so we can adjust weight dynamically. 
85 T Weight; // Internal only. You cannot get or set this directly. 
86};

88typedef tLowPassFilterFixed<double> tLowPassFilter_FixDbl;

129template<typename T> inline tMath::tLowPassFilterFixed<T>::tLowPassFilterFixed(T fixedDeltaTime, T attenuation, bool tauAttenuation, T initialValue) : 
130 tLowPassFilter<T>(initialValue), 
131 FixedDeltaTime(fixedDeltaTime) 
132{ 
133 // Calculate the weight. It doesn't change since fixedDeltaTime is unchanging. 
134 tiClampMin<T>(attenuation, T(tMath::Epsilon)); 
135 T tau = tauAttenuation ? attenuation : ComputeTau(attenuation); 
136 Weight = ComputeWeight(FixedDeltaTime, tau); 
137}
