27void tContGamepad::StartPolling(int pollingPeriod, tGamepadID gamepadID) 
28{ 
29 tAssert((pollingPeriod >= 1) && (gamepadID != tGamepadID::Invalid) && (gamepadID != tGamepadID::NumGamepads)); 
30 
31 // If it's already running do nothing. We also don't update the period if we're already running. 
32 if (IsPolling()) 
33 return; 
34 
35 // PollingPeriod and PollingGamepadID is only ever set before the thread starts and doesn't change. We don't need to Mutex protect it. 
36 PollingGamepadID = gamepadID; 
37 PollingPeriod = pollingPeriod; 
38 PollingThread = std::thread(&tContGamepad::Poll, this); 
39 
40}

43void tContGamepad::StopPolling() 
44{ 
45 // If it's already stopped so do nothing. 
46 if (!IsPolling()) 
47 return; 
48 
49 // We don't want to wait around to exit the polling thread to finish its sleep cycle so we use a condition_variable 
50 // that uses PollingExitRequested as the predecate. 
51 { 
52 const std::lock_guard<std::mutex> lock(Mutex); 
53 PollingExitRequested = true; 
54 } 
55 
56 // Joins back up to the detection thread. 
57 PollingThread.join(); 
58 PollingGamepadID = tGamepadID::Invalid; 
59}

62void tContGamepad::Poll() 
63{ 
64 while (true) 
65 { 
66 #ifdef PLATFORM_WINDOWS 
67 WinXInputState state; 
68 tStd::tMemclr(&state, sizeof(WinXInputState)); 
69 
70 // Get the state of the controller from XInput. From what I can tell reading different gamepads on different 
71 // threads does not require a mutex to protect this call. Only this thread instance will read this particular 
72 // controller and so two calls to XInputGetState for the same controller will never happed at the same time. 
73 WinDWord result = XInputGetState(int(PollingGamepadID), &state); 
74 
75 if (result == WinErrorSuccess) 
76 { 
77 // Controller connected. We can read its state and update components. 
78 if (state.dwPacketNumber != PollingPacketNumber) 
79 { 
80 // The set calls to the components below are all mutex protected internally. Main thread may read input 
81 // unit values so we require the protection. 
82 
83 // Note there is a bit more precision for the neg integral values. -32768 to 32767 maps to [-1.0, 1.0] 
84 int16 rawLX = state.Gamepad.sThumbLX; 
85 float axisLXNorm = (rawLX < 0) ? float(rawLX)/32768.0f : float(rawLX)/32767.0f; 
86 tPrintf("LX: %05.3f ", axisLXNorm); 
87 // @todo Set LStick X. 
88 
89 int16 rawLY = state.Gamepad.sThumbLY; 
90 float axisLYNorm = (rawLY < 0) ? float(rawLY)/32768.0f : float(rawLY)/32767.0f; 
91 tPrintf("LY: %05.3f ", axisLXNorm); 
92 // @todo Set LStick Y. 
93 
94 bool rawLB = (state.Gamepad.wButtons & XINPUT_GAMEPAD_LEFT_THUMB) ? true : false; 
95 tPrintf("LB: %s\n", rawLB ? "down" : "up"); 
96 // @todo Set LStick B. 
97 
98 int16 rawRX = state.Gamepad.sThumbRX; 
99 float axisRXNorm = (rawRX < 0) ? float(rawRX)/32768.0f : float(rawRX)/32767.0f; 
100 // @todo Set RStick X. 
101 
102 int16 rawRY = state.Gamepad.sThumbRY; 
103 float axisRYNorm = (rawRY < 0) ? float(rawRY)/32768.0f : float(rawRY)/32767.0f; 
104 // @todo Set RStick Y. 
105 
106 bool rawRB = (state.Gamepad.wButtons & XINPUT_GAMEPAD_RIGHT_THUMB) ? true : false; 
107 
108 float leftTriggerNorm = float(state.Gamepad.bLeftTrigger) / 255.0f; 
109 LTrigger.SetDisplacementRaw(leftTriggerNorm); 
110 
111 float rightTriggerNorm = float(state.Gamepad.bRightTrigger) / 255.0f; 
112 RTrigger.SetDisplacementRaw(rightTriggerNorm); 
113 
114 PollingPacketNumber = state.dwPacketNumber; 
115 } 
116 } 
117 else 
118 { 
119 // Controller is not connected. This can happen if the detection thread of the controller system has not 
120 // realized yet that the controller was disconnected (and stopped the polling thread). In this case we can 
121 // just stop polling. Note that exiting the loop does not stop the thread from being joinable. It is still 
122 // considered connected until the detection thread calls StopPolling. In the mean time, it only makes sense 
123 // to reset the components. 
124 LTrigger.Reset(); 
125 RTrigger.Reset(); 
126 break; 
127 } 
128 #endif 
129 
130 // This unique_lock is just a more powerful version of lock_guard. Supports subsequent unlocking/locking which 
131 // is presumably needed by wait_for. In any case, wait_for needs this type of lock. 
132 std::unique_lock<std::mutex> lock(Mutex); 
133 bool exitRequested = PollingExitCondition.wait_for(lock&: lock, rtime: std::chrono::nanoseconds(PollingPeriod), p: [this]{ return PollingExitRequested; }); 
134 if (exitRequested) 
135 break; 
136 } 
137}
