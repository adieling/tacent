16#pragma once 
17#include <mutex> 
18#include <Foundation/tFundamentals.h> 
19#include "Input/tUnit.h" 
20namespace tInput

22 
23 
24// A continuous displacement unit is a container for a float in [0.0, 1.0]. 
25class tUnitContinuousDisp : public tUnit 
26{ 
27public: 
28 tUnitContinuousDisp(const tName& name, std::mutex& mutex) : tUnit(name, mutex) { } 
29 virtual ~tUnitContinuousDisp() { } 
30 
31 // Read by the main system update to sent change notification events. 
32 // May also be used directly by client code in main thread. 
33 float GetDisplacement() const 
34 { 
35 std::lock_guard<std::mutex> lock(Mutex); 
36 return Displacement; 
37 } 
38 
39 void Reset() 
40 { 
41 SetDisplacementRaw(0.0f); 
42 SetDisplacement(0.0f); 
43 } 
44 
45private: 
46 friend class tCompTrigger; 
47 
48 // Called by the the controller polling thread. 
49 void SetDisplacementRaw(float disp) 
50 { 
51 std::lock_guard<std::mutex> lock(Mutex); 
52 tMath::tiClamp(val&: disp, min: 0.0f, max: 1.0f); 
53 DisplacementRaw = disp; 
54 } 
55 
56 // Called by the the controller component in the update function of the main thread. It is the component that does 
57 // anti-jitter and dead zone to read the raw value and convert it to the actual. 
58 void SetDisplacement(float disp) 
59 { 
60 tAssert(tMath::tInInterval(disp, 0.0f, 1.0f)); 
61 std::lock_guard<std::mutex> lock(Mutex); 
62 DisplacementRaw = disp; 
63 } 
64 
65 float DisplacementRaw = 0.0f; // Mutex protected. 
66 float Displacement = 0.0f; // Mutel protected. 
67}; 
68 
69 
70}
