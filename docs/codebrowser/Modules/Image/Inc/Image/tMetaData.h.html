43 LongitudeDMS, // string Degrees, Minutes, Seconds, Direction. Eg. 160°59'4"W

45 AltitudeRelRef, // string Relative altitude ground reference. Applies to AltitudeRel value. 
46 // "Above Ground" : Reference data unavailable. Assume above ground. 
47 // "Above Sea Level" : Ground is above sea level. 
48 // "Below Sea Level" : Ground is below sea level. 
49 AltitudeRel, // float Relative altitude in meters. Relative to AltitudeRelRef. 
50 Roll, // float Flight roll in degrees.

52 Yaw, // float Flight yaw in degrees.

59 GPSTimeStamp, // string UTC Date and time as string in format "YYYY-MM-DD hh:mm:ss". 
60 // It's possible one of the YYYY-MM-DD or hh:mm:ss parts is missing. You will get nothing/invalid 
61 // or "YYYY-MM-DD" or "hh:mm:ss" or both: "YYYY-MM-DD hh:mm:ss" depending on what is available.

64 ShutterSpeed, // float Units s^-1. Reciprocal of exposure time. If not set, computed. 
65 ExposureTime, // float Exposure time in seconds. Reciprocal of ShutterSpeed. If not set, computed. 
66 ExposureBias, // float Exposure bias. APEX units. 
67 FStop, // float F/Stop. Unitless. Ratio of the lens focal length to the diameter of the entrance pupil. 
68 ExposureProgram,// uint32 Exposure Program. 
69 // 0: Not Defined. 
70 // 1: Manual. 
71 // 2: Normal Program. 
72 // 3: Aperture Priority. 
73 // 4: Shutter Priority. 
74 // 5: Creative Program. 
75 // 6: Action Program. 
76 // 7: Portrait Mode. 
77 // 8: Landscape Mode. 
78 ISO, // uint32 Equivalent ISO film speed rating. 
79 Aperture, // float APEX units. 
80 Brightness, // float Average scene luminance of whole image. APEX units. 
81 MeteringMode, // uint32 Metering Mode. 
82 // 0: Unknown. 
83 // 1: Average. 
84 // 2: Center Weighted Average. 
85 // 3: Spot. 
86 // 4: Multi-spot. 
87 // 5: Pattern. 
88 // 6: Partial. 
89 FlashHardware, // uint32 Flash hardware present. 
90 // 0: Flash Present. 
91 // 1: No Flash Present. 
92 FlashUsed, // uint32 Flash used. 
93 // 0: No. 
94 // 1: Yes. 
95 FlashStrobe, // uint32 Flash strobe detection. 
96 // 0: No Detector. 
97 // 1: Reserved. 
98 // 2: Strobe Return Light Not Detected. 
99 // 3: Strobe Return Light Detected. 
100 FlashMode, // uint32 Flash camera mode. 
101 // 0: Unknown. 
102 // 1: Compulsory Flash Firing. 
103 // 2: Compulsory Flash Suppression. 
104 // 3: Auto. 
105 FlashRedEye, // uint32 Red eye reduction. 
106 // 0: No Red-Eye Reduction or Unknown. 
107 // 1: Red-Eye Reduction. 
108 FocalLength, // float Lens focal length in mm. Always > 0 if set. 
109 Orientation, // uint32 Orientation. 
110 // Note the descriptions below describe the transformations that are present in the data in the current file. 
111 // Reverse ops in reverse order to obtain an untransformed image. 
112 // 0: Unspecified. 
113 // 1: NoTransforms. Image data is not mirrored or rotated. 
114 // 2: Flip-Y. Image is mirrored about vertical axis (right <-> left). 
115 // 3: Flip-XY. Same as 180 Degrees Rotation. 
116 // 4: Flip-X. The image is mirrored about horizontal axis (top <-> bottom). 
117 // 5: Rot-CW90 Flip-Y. The image is rotated 90 degrees clockwise and then flipped about verical axis. 
118 // 6: Rot-ACW90. The image is rotated 90 degrees anti-clockwise. 
119 // 7: Rot-ACW90 Flip-Y. The image is rotated 90 degrees clockwise and then flipped about verical axis. 
120 // 8: Rot-CW90. The image is rotated 90 degrees anti-clockwise. 
121 LengthUnit, // uint32 The length unit used for XPixelsPerUnit and YPixelsPerUnit. AKA: ResolutionUnit. 
122 // 1: Not Specified. 
123 // 2: Inch. 
124 // 3: cm. 
125 XPixelsPerUnit, // float Horizontal pixels per DistanceUnit. AKA: XResolution.

128 ImageWidth, // uint32 Width in pixels. 
129 ImageHeight, // uint32 Height in pixels. 
130 ImageWidthOrig, // uint32 Original image width in pixels. 
131 ImageHeightOrig,// uint32 Original image height in pixels. 
132 DateTimeChange, // string Date and time the image was changed. "YYYY-MM-DD hh:mm:ss". 
133 DateTimeOrig, // string Date and time of original image. 
134 DateTimeDigit, // string Date and time the image was digitized. 
135 
136 // Authoring Note Tags 
137 Software, // string Software used to edit image. 
138 Description, // string Image description. 
139 Copyright, // string Copyright notice.

149// A single piece of image metadata. Could be more memory efficient, but hardly seems worth it. 
150struct tMetaDatum 
151{ 
152 enum class DatumType { Invalid = -1, Uint32, Float, String }; 
153 tMetaDatum() : Type(DatumType::Invalid) { } 
154 tMetaDatum(const tMetaDatum& src) { Set(src); } 
155 
156 void Clear() { Type = DatumType::Invalid; String.Clear(); } 
157 void Set(const tMetaDatum& src); 
158 void Set(uint32 v) { Type = DatumType::Uint32; Uint32 = v; } 
159 void Set(float v) { Type = DatumType::Float; Float = v; } 
160 void Set(const tString& v) { Type = DatumType::String; String = v; } 
161 bool IsValid() const { return (Type != DatumType::Invalid); } 
162 bool IsSet() const { return (Type != DatumType::Invalid); } 
163 
164 // Two invalid datums are always considered equal. If both are valid, only then are values compared. 
165 bool operator==(const tMetaDatum& src) const; 
166 bool operator!=(const tMetaDatum& src) const { return !(*this == src); } 
167 tMetaDatum& operator=(const tMetaDatum& src) { Set(src); return *this; } 
168 
169 DatumType Type; 
170 union { uint32 Uint32; float Float; }; 
171 tString String; 
172};

227inline void tImage::tMetaDatum::Set(const tMetaDatum& src) 
228{ 
229 Type = src.Type; 
230 switch (Type) 
231 { 
232 case DatumType::Uint32: Uint32 = src.Uint32; return; 
233 case DatumType::Float: Float = src.Float; return; 
234 case DatumType::String: String = src.String; return; 
235 }

239inline bool tImage::tMetaDatum::operator==(const tMetaDatum& src) const 
240{ 
241 if ((Type == DatumType::Invalid) && (src.Type == DatumType::Invalid)) 
242 return true; 
243 
244 if (Type != src.Type) 
245 return false; 
246 
247 tAssert(Type == src.Type); 
248 switch (Type) 
249 { 
250 case DatumType::Uint32: return Uint32 == src.Uint32; 
251 case DatumType::Float: return Float == src.Float; 
252 case DatumType::String: return String == src.String; 
253 } 
254 return false; 
255}

267inline bool tImage::tMetaData::Set(const tMetaData& src) 
268{

270 for (int d = 0; d < int(tMetaTag::NumTags); d++)

272 
273 return IsValid();

277inline bool tImage::tMetaData::operator==(const tMetaData& src) const 
278{ 
279 if (!IsValid() && !src.IsValid()) 
280 return true; 
281 
282 if (NumTagsValid != src.NumTagsValid) 
283 return false; 
284 
285 for (int d = 0; d < int(tMetaTag::NumTags); d++) 
286 if (Data[d] != src.Data[d]) 
287 return false; 
288 
289 return true; 
290}
