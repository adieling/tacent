24// This is the type you should use most of the time for 2D vectors. Extends the POD type with operators etc. Note the 
25// intentional empty default constructor. 
26struct tVector2 : public tVec2 
27{ 
28 tVector2() { } 
29 tVector2(const tVec2& v) { tSet(d&: *this, s: v); } 
30 tVector2(const tVec3& v) /* Discards v.z. */ { tSet(d&: *this, s: v); } 
31 tVector2(const tVec4& v) /* Discards v.z and v.w. */ { tSet(d&: *this, s: v); } 
32 tVector2(float xy) /* Sets both x and y to xy. */ { tSet(d&: *this, xy); } 
33 tVector2(float x, float y) { tSet(d&: *this, x, y); } 
34 tVector2(const float* a) { tSet(d&: *this, a); } 
35 const tVec2& Pod() const { return *this; } 
36 
37 void Set(const tVec2& v) { tSet(d&: *this, s: v); } 
38 void Set(const tVec3& v) { tSet(d&: *this, s: v); } 
39 void Set(const tVec4& v) { tSet(d&: *this, s: v); } 
40 void Set(float xy) { tSet(d&: *this, xy); } 
41 void Set(float x, float y) { tSet(d&: *this, x, y); } 
42 void Set(const float* a) { tSet(d&: *this, a); } 
43 void Get(float& x, float& y) const { tGet(x, y, s: *this); } 
44 void Get(float* a) const { tGet(a, s: *this); } 
45 
46 void Zero() { tZero(d&: *this); } 
47 void Zero(comp_t c) { tZero(d&: *this, c); } 
48 bool IsZero() const { return tIsZero(v: (tVec2&)*this); } 
49 bool IsZero(comp_t c) const { return tIsZero(v: *this, c); } 
50 bool ApproxEqual(const tVec2& v, float e = Epsilon) const { return tApproxEqual(a: *this, b: v, e); } 
51 bool ApproxEqual(const tVec2& v, comp_t c, float e = Epsilon) const { return tApproxEqual(a: *this, b: v, c, e); } 
52 
53 float LengthSq() const { return tLengthSq(v: *this); } 
54 float Length() const { return tLength(v: *this); } 
55 float LengthFast() const { return tLengthFast(v: *this); } 
56 
57 void Normalize() { tNormalize(v&: *this); } 
58 float NormalizeGetLength() /* Returns pre-normalized length. */ { return tNormalizeGetLength(v&: *this); } 
59 bool NormalizeSafe() /* Returns success. */ { return tNormalizeSafe(v&: *this); } 
60 float NormalizeSafeGetLength() /* Returns pre-normalized length. 0.0f if no go. */ { return tNormalizeSafeGetLength(v&: *this); } 
61 void NormalizeScale(float s) /* Normalize then scale. Resizes the vector. */ { tNormalizeScale(v&: *this, s); } 
62 bool NormalizeScaleSafe(float s) { return tNormalizeScaleSafe(v&: *this, s); } 
63 void NormalizeFast() { tNormalizeFast(v&: *this); } 
64 bool NormalizeSafeFast() { return tNormalizeSafeFast(v&: *this); } 
65 
66 void Lerp(const tVec2& a, const tVec2& b, float t) /* Also extrapolates. See tLerp comments. */ { tLerp(d&: *this, a, b, t); } 
67 inline static int GetNumComponents() { return 2; }

69 // Friend is used for commutative binary operators where one of the operators is a built-in type. This essentially 
70 // puts these operators outside of the class where they belong. If the other operand type is user-defined, the 
71 // operator should be completely outside as there would be ambiguity as to which one owned the friend declaration. 
72 tVector2& operator=(const tVec2& v) { tSet(d&: *this, s: v); return *this; } 
73 tVector2& operator=(const tVec3& v) { tSet(d&: *this, s: v); return *this; } 
74 tVector2& operator=(const tVec4& v) { tSet(d&: *this, s: v); return *this; } 
75 
76 // Addition is not defined for types other than the same type as the object itself, so no friend needed. 
77 tVector2& operator+=(const tVec2& a) { tAdd(v&: *this, a); return *this; } 
78 const tVector2 operator+(const tVec2& a) const { tVector2 d; tAdd(d, a: *this, b: a); return d; } 
79 
80 tVector2& operator-=(const tVec2& a) { tSub(v&: *this, a); return *this; } 
81 const tVector2 operator-(const tVec2& a) const { tVector2 d; tSub(d, a: *this, b: a); return d; } 
82 const tVector2 operator-() const { tVector2 d; tNeg(d, a: *this); return d; } 
83 
84 tVector2& operator*=(float a) { tMul(v&: *this, a); return *this; } 
85 inline friend const tVector2 operator*(const tVec2& a, float b) { tVector2 d; tMul(d, a, b); return d; } 
86 inline friend const tVector2 operator*(float a, const tVec2& b) { tVector2 d; tMul(d, a, b); return d; } 
87 float operator*(const tVec2& a) const /* Dot (inner) product. Same as Transpose(this) * a. */ { return tDot(a: *this, b: a); } 
88 
89 // Divide is not commutative so no need for friend. 
90 tVector2& operator/=(float a) { tDiv(v&: *this, a); return *this; } 
91 const tVector2 operator/(float a) const { tVector2 d; tDiv(d, a: *this, b: a); return d; } 
92 
93 inline friend bool operator==(const tVec2& a, const tVec2& b) { return tEqual(a, b); } 
94 inline friend bool operator!=(const tVec2& a, const tVec2& b) { return tNotEqual(a, b); } 
95 operator const float*() { return E; } 
96 operator const float*() const { return E; } 
97 float& operator[](int i) { return E[i]; } 
98 float operator[](int i) const { return E[i]; }

101 const static tVector2 i; // Basis vector (1, 0). 
102 const static tVector2 j; // Basis vector (0, 1). 
103 const static tVector2 one; // One vector (1, 1). 
104};
