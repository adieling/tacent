64// Interval Notation. When you see [a,b] or (a,b) the square bracket means include the endpoint and the rounded brackets 
65// mean exclude. See http://www.coolmath.com/algebra/07-solving-inequalities/03-interval-notation-01. When a function 
66// takes a bias argument, a Low bias will cause the return value to include the lower extent of the interval and exclude 
67// the high extent. A High bias will exclude the low end and include the high end. As a notational convenience When a 
68// function takes a bias argument, we'll write the interval as [(a,b)] and the enum below shows how to interpret it: 
69enum class tBias 
70{ 
71 Full, Outer = Full, // [a,b] 
72 Low, Left = Low, // [a,b) 
73 High, Right = High, // (a,b] 
74 Center, Inner = Center // (a,b) 
75};

341 inline int tGCDRec(int a, int b) 
342 { 
343 // This is the Euclidean algorithm: 
344 // b<a : gdc(a,b) = gdc(a-b,b). 
345 // Eg. gdc(16,4) == gdc(12,4) == gdc(8,4) == gdc(4,4) (done). 
346 // Instead of looping through the subracts one by one, we can just use the remainder / mod operator. 
347 if (b == 0) 
348 return a; 
349  
350 return tGCDRec(a: b, b: a % b); 
351 } 
352}

355inline int tMath::tGCD(int a, int b) 
356{ 
357 // Both zeroes is undefined. Any number divides zero, but since the other number is also zero, the 
358 // most reasonable it to return the biggest number we can. 
359 if ((a == 0) && (b == 0)) 
360 return MaxInt; 
361 
362 if (a < 0) a = -a; 
363 if (b < 0) b = -b; 
364 return tGCDRec(a, b); 
365}

368inline int tMath::tLCM(int a, int b) 
369{ 
370 // lcm(a,b) = |a| * (|b|/gcd(a,b)) 
371 // Since tGCD returns MaxInt for tGCD(0,0), we don't need to worry about divide by zero. 
372 if (a < 0) a = -a; 
373 if (b < 0) b = -b; 
374 return a * ( b / tGCD(a,b) );

408inline int tMath::tFindFirstClearBit(uint8 v) 
409{ 
410 // Find the first zero bit. The operation we do is log2((a xor (a+1)) +1) 
411 // This is guaranteed to be a power of two. 
412 uint8 freeBit = (v ^ (v+1)) + 1; 
413 
414 // If 0 it means nothing was found. 
415 if (!freeBit) 
416 return -1; 
417 
418 // Now get the log in base 2 of freeBit and wrap if position is 0. 
419 int c = tMath::tLog2(v: freeBit) ; 
420 if (c == 0) 
421 c = 8; 
422 
423 // This is the first cleared index in the bit array (from the LSB). 
424 return c-1; 
425}

469inline int tMath::tLog2(uint32 x) 
470{ 
471 if (x == 0) 
472 return -1; 
473 
474 // See "http://graphics.stanford.edu/~seander/bithacks.html#IntegerLog". 
475 const uint32 b[] = { 0xAAAAAAAA, 0xCCCCCCCC, 0xF0F0F0F0, 0xFF00FF00, 0xFFFF0000 }; 
476 int c = (x & b[0]) != 0; 
477 
478 c |= ((x & b[4]) != 0) << 4; 
479 c |= ((x & b[3]) != 0) << 3; 
480 c |= ((x & b[2]) != 0) << 2; 
481 c |= ((x & b[1]) != 0) << 1; 
482 return c; 
483}
