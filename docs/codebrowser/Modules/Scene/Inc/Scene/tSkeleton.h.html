94inline void tJoint::Clear() 
95{ 
96 tObject::Clear(); 
97 Translation.Zero(); 
98 Orientation = tMath::tQuaternion::unit; 
99 JointScale.Set(x: 1.0f, y: 1.0f, z: 1.0f, w: 1.0f); 
100 ParentID = tObject::InvalidID; 
101 NumChildren = 0; 
102 delete Children; 
103 Children = nullptr;
