44// on all filesystems: YYYY-MM-DD-HH-MM-SS. 
45enum class tTimeFormat 
46{ 
47 Standard, // Eg. 2020-01-14 01:47:12

61class tTimer 
62{ 
63public: 
64 // Creates a timer. You can specify whether the timer is constructed started or not. Internally a floating-point 
65 // (double) member is used to keep track of the elapsed time. You can choose the units it represents, which has 
66 // implications for overall timer precision. In general, use an internal unit that matches your domain - timing 
67 // oscillations of visible light? Use nanoseconds. If unit is Unspecified seconds are used. 
68 tTimer(bool start = true, tUnit::tTime unit = tUnit::tTime::Second) : UnitInternal(unit), Running(start), Time(0.0f) { if (UnitInternal == tUnit::tTime::Unspecified) UnitInternal = tUnit::tTime::Second; } 
69 
70 // Call this frequently. If Unit is not specified, internal units are used. 
71 void Update(float timeElapsed, tUnit::tTime = tUnit::tTime::Unspecified); 
72 
73 // Does nothing (idempotent) if timer already started (or stopped). 
74 void Start() { Running = true; } 
75 void Stop() { Running = false; } 
76 void Reset(bool start = true) { Time = 0.0; Running = start; } 
77 
78 // Returns the time in the units of Unit. If Unit is not specified, uses internal units. 
79 float GetTime(tUnit::tTime = tUnit::tTime::Unspecified) const; 
80 bool IsRunning() const { return Running; } 
81 tUnit::tTime GetInternalUnit() const { return UnitInternal; } 
82 
83 static double Convert(double time, tUnit::tTime from, tUnit::tTime to); 
84 
85private: 
86 // @todo Add getters that format the time into a number of different string formats.

89 tUnit::tTime UnitInternal; 
90 bool Running; 
91 double Time;

94 // For developers only. Easily add a new unit and recreate the unit conversion table. 
95 static void PrintHighPrecisionConversionTable(const char* outputFile); 
96};
