3// A tFixInt is a fixed sized integer type that may be larger than the native types. It is not a 'big' integer class 
4// in that the size of the int is fixed at compile time. This loss of dynamic precision is well worth the efficiency 
5// gains and simple memory layout that a fixed size affords. tFixInt is ideal for storing integer values where a uint32 
6// or uint64 is not big enough. The tFixIntU is the unsigned version of a tFixInt. tFixIntU is a superclass of a 
7// tFixInt. This header also typedefs some commonly used sizes. Specifically it allows one to use the types tint128, 
8// tint256, tint512, tuint128, tuint256, and tuint512 simply by including this header. 
9// 
10// Comparisons 
11// tBitArray - Use when you want to store a large number of bits and you don't know how many at compile-time. 
12// This type os primatily for storage and access to a large number of bits. Not many bitwise or 
13// mathematical operators. 
14// tBitField - Use when know how many bits at compile-time and you want bitwise logic opertors like and, or, xor, 
15// shift, not, etc. Good for storing a fixed number of flags or channels etc. 
16// tFixInt - Use when you want full mathematical operations like any built-in integral type. Size must be known at 
17// compile time and must be a multiple of 32 bits. You get + - / * etc as well as all bitwise logic ops. 
18// You can construct a tFixInt from a tBitField of the same size.

606{

609 { 
610 shiftcnt = 1 / shiftcnt; 
611 tFixIntU<N> r; 
612 r.MakeMax(); 
613 return r; 
614 } 
615 
616 shiftcnt = a.FindHighestBitSet() - b.FindHighestBitSet(); 
617 if (shiftcnt > 0) 
618 b <<= shiftcnt; 
619 
620 if (b > a) 
621 { 
622 b >>= 1; 
623 --shiftcnt; 
624 }

628 {

634 b >>= 1; 
635 --shiftcnt; 
636 }

645{ 
646 // Special version of division that's fast but only works with small divisors. 'a' must be positive. 
647 tAssert(b); 
648 uint32 temp = 0; 
649 bool seenNonZero = false; 
650 tFixInt<N> result = 0; 
651 
652 for (int i = a.GetNumElements()-1; i >= 0; i--) 
653 { 
654 if (seenNonZero || a.GetElement(i) != 0) 
655 { 
656 uint32 sh1 = a.GetElement(i) >> 16; 
657 temp = (temp << 16) + sh1; 
658 sh1 = temp / b; 
659 temp %= b; 
660 
661 uint32 sh2 = (a.GetElement(i) << 16) >> 16; 
662 temp = (temp << 16) + sh2; 
663 sh2 = temp / b; 
664 temp %= b; 
665 
666 result.SetElement(i, (sh1 << 16) | sh2); 
667 seenNonZero = true; 
668 } 
669 } 
670 if (remainder) 
671 *remainder = int(temp); 
672 
673 return result; 
674}
