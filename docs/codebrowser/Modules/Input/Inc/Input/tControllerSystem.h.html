26class tControllerSystem 
27{ 
28public: 
29 // PollingPeriod is in nanoseconds. PollingControllerDetectionPeriod is in milliseconds. A value of 0 means 
30 // auto-determine both the pollingPeriod and the pollingControllerDetectionPeriod. In auto-determine mode 
31 // pollingControllerDetectionPeriod is set to 1 second and pollingPeriod tries to detect the controller 
32 // manufacturers and use an appropriate pollingPeriod. For example the polling rate of an 8BitDo is 1000Hz so it 
33 // will set the pollingPeriod to 1ms. 
34 // 
35 // Note that currently there is a single polling thread for all controllers (not per controller). This means that 
36 // it will choose the lowest period for all detected contollers and may change as contollers are unplugged and 
37 // plugged in. If controller details can't be determined or pollingPeriod is 0, 8ms (125Hz) is used. This is the 
38 // XBoxOne controller (gamepad) polling rate. 
39 tControllerSystem 
40 ( 
41 int pollingPeriod_ns = 0, int pollingControllerDetectionPeriod_ms = 0 
42 ); 
43 virtual ~tControllerSystem(); 
44 
45 // Call this periodically from the main thread loop. When this is called any callbacks are executed and all 
46 // controller state is updated. 
47 void Update(); 
48 
49 tContGamepad& GetGetpad(tGamepadID gid) 
50 { 
51 return Gamepads[int(gid)]; 
52 } 
53 
54private: 
55 // This function runs on a different thread. 
56 void Detect(); 
57 
58 // Scrapes gamepad capabilities/hardware-info to try to determine the optimal polling period in ns for the specified 
59 // gampepad ID. If it fails, which is not unlikely, returns 0. We use nano-seconds because some controllers can poll 
60 // in the multiple thousands of Hz. 
61 int DetermineGamepadPollingPeriodFromHardwareInfo(tGamepadID); 
62 
63 // This mutex protects PollExitRequested, the gamepad Connected state variable, and all tUnit members in the 
64 // components of the gamepads -- ditto for other controller types when we get around to implementing them. 
65 std::mutex Mutex; 
66 
67 bool PollingPeriodAutoDetect = false; 
68 int PollingPeriod = 8000; // In nanoseconds. 8000ns -> 125Hz. 
69 int DetectPeriod = 1000; // In milliseconds. 
70 
71 // To simplify the implementation we are going to support up to precisely 4 gamepads. This matches the maximum 
72 // supported by xinput on windows and restricts the number of gamepads on Linux to 4, which seems perfectly 
73 // reasonable. By simply having an array of gamepads that are always present it also makes reading controller values 
74 // a simple process. Just loop through the controllers and ignore any that are in the disconnected state. 
75 // Parts of tContGamepad mutex protected: the connected bool and tUnit values in the components. 
76 std::vector<tContGamepad> Gamepads; 
77 
78 // The DetectExitRequested predicate is required to avoid spurious wakeups. Mutex protected. 
79 bool DetectExitRequested = false; 
80 std::condition_variable DetectExitCondition; 
81 std::thread DetectThread; 
82};
